;
; THIS MODULE IS DESIGNED TO ALLOW A FORTRAN PROGRAM TO ACCESS THE
; 'SORTER.DAT' FILE, AND MAKE USE OF THE 'SORTER.SAV' PROGRAM TO
; SORT RECORDS. REFER TO DOCUMENTATION ON 'SORTER.SAV' FOR FORMAT
; OF THE DATA FILE-
;
; THE ENTRY POINTS TO THE MODULE ARE -
;
;	SRTFIN	---	INITIALIZE MODULE. THIS ENTRY OPENS THE
;			SORTER.DAT FILE AND RESETS POINTERS ETC.
;			MUST BE CALLED PRIOR TO ANY OTHER CALLS TO
;			MODULE.
;
;	SRTFOR	---	USED TO ACCESS/INVOKE FACILITIES PROVIDED
;			BY MODULE.
.SBTTL	MODIFICATIONS
;
; 11-FEB-80	MOVE CODE INTO RIGHT PSECT
;

;
;
;
.SBTTL	DOCUMENTATION	- CALLING SEQUENCES
;
;
; THE FIRST ARGUMENT IS AN OPCODE FUNCTION, WHICH DETERMINES THE OPERATION
; TO BE DONE. THE CALLING SEQUENCES ARE -
;
;	CALL SRTFOR (0)
;
;		RESETS/REWINDS THE DATA FILE
;
;	CALL SRTFOR (1)
;
;		CLOSES OUT THE DATA FILE. THE CHANNEL IS PURGED.
;
;	CALL SRTFOR (2,ISIZE,NUMREC,ISTATUS,ICHAIN)
;
;		RETURNS PARAMETERS TO CALLER. 'ISIZE' IS THE RECORD
;		SIZE IN BYTES, 'NUMREC' IS THE NUMBER OF RECORDS,
;		'ISTATUS' IS THE STATUS FLAG SET UP WITH THE LAST
;		PUT CALL, AND 'ICHAIN' IS NON-ZERO IFF THE PROGRAM
;		WAS CHAINED TO.
;
;	CALL SRTFOR (3,ISIZE,NUMREC,ISTATUS,NAME)
;
;		PUTS THE SPECIFIED PARAMETERS INTO THE DATA FILE.
;		THE 'NAME' VARIABLE IS AN ASCIZ STRING, WHICH IS
;		THE NAME OF THE FILE TO BE CHAINED AFTER THE RECORDS
;		ARE SORTED.
;
;	CALL SRTFOR (4,BUF,NUM)
;
;		MOVE 'NUM' BYTE FROM 'BUF' TO THE DATA FILE.
;
;	CALL SRTFOR (5,BUF,NUM)
;
;		MOVE 'NUM' BYTES FROM THE FILE TO 'BUF'.
;
;	CALL SRTFOR (6)
;
;		SORT RECORDS IN FILE. THIS CALL CLOSES OUT THE
;		FILE, AND CHAINS THE 'SORTER.SAV'. 'SORTER.SAV'
;		WILL CHAIN TO THE PROGRAM WHOSE NAME WAS PASSED OVER
;		IN THE LAST PUT CALL WHEN THE SORTING IS DONE.
;
;
; THE NEXT PAGE CONTAINS A FORTRAN PROGRAM WHICH COULD BE USED TO
; SORT RECORDS.
;
;

.IF	EQ,-1					; KILL MACRO ASSEMBLY
;
PROGRAM	SORT DEMO
C
C THIS PROGRAM WILL INPUT 'N' RECORDS OF LENGHT 'S', WHERE 'N' AND
C 'S' ARE SPECIFIED BY THE OPERATOR, SORT THEN IN 'SORTER.DAT',
C AND INVOKE 'SORTER.SAV' TO SORT THEM. IT WILL THEN PRINT OUT
C THE SORTED RECORDS.
C
C
	LOGICAL*1	BUF(50)		! RECORD BUFFER
C
C
	CALL	SRTFIN				! INITIALIZE
	CALL	SRTFOR(2,II,NUMREC,ISTATUS,ICHAIN)! GET PARMS
	ISIZE	= II-2				! WORK OUT DATA LENGHT
	IF	(ICHAIN.NE.0)	GOTO	5000	! IF CHAINED TO -> 5000
C
	WRITE	(7,10)				! PROMPT
10	FORMAT	(1X,'INPUT SIZE,NUM RECORDS > ',$)
	READ	(5,20)	ISIZE,NUMREC		! GET PARAMETERS
20	FORMAT	(3I10)
	II	= ISIZE+2			! WORK OUT RECORD SIZE	
	CALL	SRTFOR(3,II,NUMREC,3,'T.SAV')	! SET UP PARAMETERS
C
	DO	100	I=1,NUMREC		! SET UP LOOP
		WRITE	(7,110) I		! PROMPT
110	FORMAT	(1X,I3,'>',$)
		READ	(5,120) BUF		! GET RECORD
120	FORMAT	(100A1)
	CALL	SRTFOR(5,I,2)			! PUT RECORD NUMBER
	CALL	SRTFOR(5,BUF,ISIZE)		! PUT DATA
100	CONTINUE
	CALL	SRTFOR(6)			! CHAIN TO SORTER
	STOP	'CHAIN ERROR'
C
C
C
C
5000	CONTINUE				! GET HERE AFTER SORTED
	DO	5100	I=1,NUMREC		! SET UP LOOP
	CALL	SRTFOR(4,N,2)			! GET REC NUMBER
	CALL	SRTFOR(4,BUF,ISIZE)		! GET DATA
	WRITE	(7,5040) N,(BUF(J),J=1,ISIZE)	! DISPLAY
5040	FORMAT	(2X,I5,' ',100A1)
5100	CONTINUE
C
	STOP	'DONE'
	END
;
.ENDC						; RESTORE MACRO ASSEMBLY

.SBTTL	DECLARATIONS
;
;
.MCALL	.PUSH,.POP				; STACKING
.MCALL	.PRINT,.EXIT,.CHAIN			; RT11
.MCALL	FILSPT					; FILE SUPPORT
;
	FILSPT					; INVOKE FILE SUPPORT
;
;
.GLOBL	SRTFIN					; INITILAIZATION
.GLOBL	SRTFOR					; NORMAL ENTRY
;
.GLOBL	IGETC					; 'SYSLIB'
;
;
;
JSW	= 44					; JSW ADDRESS
 JSW.CH	= 400					; CHAIN BIT
;
;
	.PSECT	CODE				; OPEN CODE SECTION
;	------	----
;

;
;
;
.SBTTL	MACRO	DEFINITIONS
;
;
.MACRO	ERRON	CND,MES,?X,?Y,?Z
	B'CND	X
	BR	Y
X:	.PRINT	#'Z
	.EXIT
Z:	.ASCII	/"SRTFOR" -FATAL-/
	.ASCII	/MES/<200>
.EVEN
Y:
.ENDM	ERRON
;
;

;
;
;
.SBTTL	ROUTINE	- 'SRTFIN'	...	INITIALIZE ROUTINE
;
;
;
SRTFIN:	TST	INITFL			; TEST INITIALIZE FLAG
	ERRON	<NE>,<'SRTFIN' ALREADY INITIALIZED!>
	INC	INITFL			; SET FLAG
	CALL	IGETC			; GET A CHANNEL
	MOV	R0,FDB+FDB.CH		; SET UP CHANNEL
	ERRON	MI,<'SRTFIN' CANNOT GET A CHANNEL!>
	CALL	SFLINT			; OPEN DATA FILE
	RETURN				; EXIT
;
;
INITFL:	.WORD	0			; INITIALIZATION FLAG
;

;
;
;
.SBTTL	ENTRY - 'SRTFOR'	...	DISPATCHER
;
;
;
SRTFOR:	TST	INITFL			; INITIALIZED?
	ERRON	<EQ>,<NOT INITIALIZED - CALL 'SRTFIN' FIRST!>
	CMP	(R5),#1			; RIGHT NUMBER OF ARGUMENTS?
	ERRON	LO,<NOT ENOUGH ARGUMENTS>
;
	MOV	@2(R5),R0		; FUNCTION NUMBER -> R0
	CMP	R0,1000$-2		; LEGAL FUNCTION?
	ERRON	HIS,<ILLEGAL FUNCTION CODE>
	ASL	R0			; FUNCTION CODE*2 -> R0
	JMP	@1000$(R0)		; PROCEED TO FUNCTION
;
;
	.WORD	<1100$-1000$>/2		; NUMBER FUNCTIONS
1000$:	.WORD	SFLRST			; RESET/REWIND FILE
	.WORD	SFLEND			; CLOSE OUT FILE
	.WORD	GETPAR			; GET PARAMETERS
	.WORD	PUTPAR			; PUT PARAMERS
	.WORD	GETDAT			; GET DATA
	.WORD	PUTDAT			; PUT DATA
	.WORD	CHAIN			; CHAIN TIME
1100$:	.WORD	0			; EOT
;
;

;
;
;
.SBTTL	PRIMITIVE - 'GETPAR' ... RETURN PARAMETERS TO CALLER
;
;
;
GETPAR:	CMP	(R5),#5			; NEED 5 ARGUMENTS
	ERRON	NE,<'GETPAR' NEED 5 PARAMETES!>
	MOV	SFLPAR,@4(R5)		; RETURN SIZE
	MOV	SFLPAR+2,@6(R5)		; RETURN NUMRECS
	MOV	SFLPAR+200,@10(R5)	; RETURN STATUS
	MOV	@#JSW,@12(R5)		; RETURN JSW
	BIC	#^CJSW.CH,@12(R5)	; LEAVE CHAIN FLAG ONLY
	RETURN				; EXIT
;
;
;

;
;
;
.SBTTL PRIMITIVE - 'PUTPAR' ... PUT PARAMETERS IN DATA FILE
;
;
;
PUTPAR:	CMP	(R5),#5			; NEED 5 ARGUMENTS
	ERRON	NE,<'PUTPAR' NEED 5 PARAMETES!>
	MOV	@4(R5),SFLPAR		; SET UP SIZE
	MOV	@6(R5),SFLPAR+2		; NUM REC
	MOV	@10(R5),SFLPAR+200	; STATUS
	MOV	12(R5),R0		; R0 -> FILE NAME
	MOV	#SFLPAR+4,R1		; R1 --> NAME AREA
10$:	MOVB	(R0)+,(R1)+		; COPY
	BNE	10$			; LOOP
	RETURN				; EXIT
;
;
;

;
;
;
.SBTTL PRIMITIVE - 'GETDAT' ... GET DATA FROM FILE
;
;
;
GETDAT:	CMP	(R5),#3			; NEED THREE ARGS
	ERRON	NE,<'GETDAT' NEED THREE ARGUMENTS!>
	MOV	4(R5),R0		; R0 --> BUFFER
	MOV	@6(R5),R1		; R1 === NUM BYTES
	CALL	SFLINP			; GET DATA
	RETURN				; EXIT
;
;
;

;
;
;
.SBTTL PRIMITIVE - 'PUTDAT' ... PUT DATA IN FILE
;
;
;
PUTDAT:	CMP	(R5),#3			; NEED THREE ARGS
	ERRON	NE,<'PUTDAT' NEED THREE ARGUMENTS!>
	MOV	4(R5),R0		; R0 --> BUFFER
	MOV	@6(R5),R1		; R1 === NUM BYTES
	CALL	SFLOUT			; PUT DATA
	RETURN				; EXIT
;
;
;

;
;
;
.SBTTL PRIMITIVE - 'CHAIN' ... INVOKE SORTER.SAV
;
;
;
CHAIN:	CALL	SFLEND			; ENSURE CLOSED
	MOV	#1000$,R0		; R0 -> NAME
	MOV	#500,R1			; R1 -> AREA
	.REPT	4			; --- 4 WORD NAME ---
		MOV	(R0)+,(R1)+	;    SET UP NAME
	.ENDR				; -------------------
	.CHAIN				; CHAIN TIME
;
;
1000$:	.RAD50	/DK SORTERSAV/
;

.SBTTL	*******************
.SBTTL	*                 *
.SBTTL	* MODULE 'SFLEIO' *
.SBTTL	*                 *
.SBTTL	*******************
;
;
;	THE FUNCTION OF THIS MODULE IS TO ALLOW DATA TO BE TRANSFERED
;	TO AND FROM THE SORT DATA FILE. THE ROUTINES PROVIDED ARE
;
;		SFLINT	...	MUST BE CALLED FIRST TO INITIALIZE
;				THE BUFFERS, CHANNELS ETC. DESTROYS
;				ALL REGISTERS.
;
;		SFLRST	...	MUST BE CALLED AT THE END OF A PASS
;				TO RESET THE INPUT/OUTPUT BUFFERS.
;				EFFECTIVELY REWINDS THE FILE.
;
;		SFLEND	...	CALLED AT END OF I/O TO CLOSE OUT
;				THE FILE.
;
;		SFLINP	...	INPUT 'R1' CHARACTERS INTO BUFFER @ 'R0'
;
;		SFLOUT	...	OUTPUTS 'R1' CHARS FROM BUFFER @ 'R0'
;
;		SFLINC	...	RETURNS NEXT INPUT CHARACTER IN R0.
;
;		SFLOUC	...	OUTPUTS CHARACTER IN R0 TO FILE.
;
;
;	THE FOLLOWING VARIABLES/BUFFERS ARE USED BY THE PROGRAM. THE
;	USER MAY ACCESS ONLY THOSE MARKED WITH AN ** SYMBOL-
;
;	SFLPAR **	HOLDS BLOCK '0' OF THE SORT DATA FILE. THE
;			CONTENTS THIS 400 (OCTAL) WORD BUFFER CAN BE
;			MODIFIED TO SEND INFORMATION TO THE NEXT
;			PROGRAM TO USE THE FILE AS IT WILL BE WRITTEN
;			OUT WHEN THE FILE IS CLOSED.
;
;	INPBUF/OUTBUF	400 (OCTAL) WORDS FILE BUFFERS
;
;	INPCNT/OUTCNT	COUNT OF CHARACTERS IN INPUT/OUTPUT BUFFERS
;
;	FDB		FILE DESCRIPTOR BLOCK FOR DATA FILE.
;
;

;
;
;
.SBTTL	DOCUMENTATION	- 'SFLEIO' FILE STRUCTURE
;
;
;	'SFLEIO' IS USED TO ACCESS THE DATA IN THE "DK:SORTER.DAT"
;	DATA FILE. THIS FILE IS TREATED AS COMPOSED OF TWO SECTIONS.
;	THESE SECTIONS ARE -
;
;	BLOCK 0		A 400 (OCTAL) WORD PARAMETER BLOCK WHICH
;			WILL BE READ INTO 'SFLPAR' AT INITIALIZATION
;			TIME, AND REWRITTEN OUT AT EXIT TIME. THIS
;			PARAMETER BLOCK CAN BE USED BY THE CALLER
;			TO PASS ARGUMENTS BETWEEN PROGRAMS.
;
;	BLOCK 1-N	A STREAM OF 8 BIT BYTES, WHICH CAN BE READ/WRITTEN
;			SEQUENTIALLY. THE FILE CAN BE REWOUND AT ANY
;			TIME BY CALLING 'SFLRST'
;
;
;
.SBTTL	DOCUMENTATION	- 'SFLEIO' FILE I/O
;
;
;	ALL FILE I/O IS DONE USING THE 'FILSPT' MODULE. ANY FILE I/O
;	ERRORS WILL CAUSE A FATAL ERROR MESSAGE TO BE PRINTED.
;
;
;
;
;
.SBTTL	DECLARATIONS FOR 'SFLEIO'
;
;
.MCALL	.PUSH,.POP			; STACKING
.MCALL	.PRINT,.EXIT			; ERROR HANDLING
.MCALL	FILSPT				; FILE SUPPORT
;
	FILSPT				; INVOKE FILE SUPPORT
;
;
.GLOBL	FIL.ST				; "FILSPT"
;
;
	.PSECT	CODE			; OPEN CODE AREA
;	------	----
;

;
;
;
.SBTTL	INITIALIZATION SECTION
;
;
;	OPEN SORT DATA FILE FOR INPUT, AND READ IN THE PARAMETER
;	BLOCK. RESET THE INPUT/OUTPUT BLOCK/CHARACTER COUNTER.
;
SFLINT:	NAME	#FDB,#100$,ERROR=40$	; INSERT FILE NAME
	PURGE				; ENSURE CHANNEL OK
	LOOKUP	ERROR=60$		; LOOKUP FILE
	READ	BLOCK=#0,SIZE=#400,BUFFER=#SFLPAR,ERROR=60$
	MOV	#1,INPBLK		; START INPUTING BLOCK 1
	MOV	#1,OUTBLK		; START OUTPUTING BLOCK 1
	MOV	#1000,INPCNT		; 1000 BYTES IN INPUT BUFFER
	MOV	#0,OUTCNT		; NO BYTES IN OUTPUT BUFFER
	READ	FDB=#FDB,BLOCK=#1,BUFFER=#INPBUF,ERROR=60$
	RETURN				; ALL DONE
;
40$:	.PRINT	110$			; TELL FAILED
60$:	CALL	FIL.ST			; DISPLAY FILE STATUS
	.EXIT				; QUIT
;
.NLIST	BIN
.ENABL	LC
100$:	.ASCIZ	/DK:SORTER.DAT/		; FILE SPEC
	.BLKB	120-.+100$		; PAD OUT (***FIL.NM***)
110$:	.ASCIZ	/"SFLEIO" -??????- HOW CAN NAMING FAIL?/
.EVEN
.DSABL	LC
.LIST	BIN
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'SFLRST'	...	RESET SORT FILE
;
;
;	THIS ROUTINE IS CALLED TO RESET THE SORT FILE PRIOR TO A NEW
;	PASS. IT EMPTIES OUT THE CONTENTS OF THE OUTPUT BUFFER AND
;	RESETS THE INPUT/OUTPUT BLOCK/CHARACTER COUNTERS.
;
;
;
SFLRST:	.PUSH	R5			; SAVE R5
	TST	OUTCNT			; ANY CHARACTERS TO OUTPUT?
	BEQ	100$			; NO -> 100$
	WRITE	FDB=#FDB,BLOCK=OUTBLK,BUFFER=#OUTBUF,ERROR=1000$
100$:	MOV	#1,INPBLK		; START INPUTING BLOCK 1
	MOV	#1,OUTBLK		; START OUTPUTING BLOCK 1
	MOV	#1000,INPCNT		; 1000 BYTES IN INPUT BUFFER
	MOV	#0,OUTCNT		; NO BYTES IN OUTPUT BUFFER
	READ	FDB=#FDB,BLOCK=#1,BUFFER=#INPBUF,ERROR=1000$
	.POP	R5			; RESTORE
	RETURN				; EXIT
;
1000$:	CALL	FIL.ST			; SHOW CAUSE OF ERROR
	.EXIT				; QUIT
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'SFLEND'	...	CLOSE SORT FILE
;
;
;	THIS ROUTINE IS CALLED WHEN THE USER IS FINISHED WITH THE
;	SORT FILE. IT EMPTIES OUT THE CONTENTS OF THE OUTPUT BUFFER
;	AND CLOSES OUT THE FILE.
;
;
;
SFLEND:	.PUSH	R5			; SAVE R5
	TST	OUTCNT			; ANY CHARACTERS TO OUTPUT?
	BEQ	100$			; NO -> 100$
	WRITE	FDB=#FDB,BLOCK=OUTBLK,BUFFER=#OUTBUF,ERROR=1000$
100$:	WRITE	FDB=#FDB,BLOCK=#0,BUFFER=#SFLPAR,ERROR=1000$
	PURGE	#FDB			; PURGE CHANNEL
	.POP	R5			; RESTORE
	RETURN				; BYE
;
;
1000$:	CALL	FIL.ST			; SHOW ERROR CAUSE
	.EXIT				; EXIT TIME
;
;

;
;
;
.SBTTL	ROUTINES	- 'SLFINP'/'SLFOUT'	...	INPUT/OUTPUT
;
;
;	THESE ROUTINES TRANSFER 'R1' CHARACTERS TO/FROM THE USER
;	BUFFER TO/FROM THE SORT DATA FILE. THE ADDRESS OF THE USER
;	BUFFER MUST BE IN 'R0'.
;
;
;
SFLINP:	.PUSH	<R0,R1,R2>		; SAVE
	MOV	R0,R2			; BUFFER ADDRESS -> R2
10$:	CALL	SFLINC			; GET ONE CHARACTER
	MOVB	R0,(R2)+		; SAVE IT
	SOB	R1,10$			; LOOP
	.POP	<R2,R1,R0>		; RESTORE
	RETURN				; EXIT
;
;
;
SFLOUT:	.PUSH	<R0,R1,R2>		; SAVE
	MOV	R0,R2			; BUFFER ADDRESS -> R2
10$:	MOVB	(R2)+,R0		; CHAR -> R0
	CALL	SFLOUC			; OUTPUT A CHARACTER
	SOB	R1,10$			; LOOP
	.POP	<R2,R1,R0>		; SAVE
	RETURN				; EXIT
;
;

;
;
;
.SBTTL	PRIMITIVE	- 'SFLINC'	...	INPUT A CHARACTER
;
;
;	THIS ROUTINE READS IN THE NEXT CHARACTER FROM THE SORT
;	DATA FILE IN RETURNS IT IN R0.
;
;
;
SFLINC:	.PUSH	<R4,R5>			; SAVE
	MOV	INPCNT,R4		; INPUT COUNT -> R4
	BNE	100$			; YES -> SKIP
	INC	INPBLK			; UP BLOCK COUNT
	READ	FDB=#FDB,BLOCK=INPBLK,BUFFER=#INPBUF,ERROR=1000$
	MOV	#1000,R4		; RESET COUNTER
;
100$:	MOV	#INPBUF+1000,R5		; R5 -> END OF BUFFER
	SUB	R4,R5			; R5 -> CHARACTER TO GET
	MOVB	(R5),R0			; GET CHAR
	DEC	R4			; ONE LESS CHAR IN BUFFER
	MOV	R4,INPCNT		; SAVE
	.POP	<R5,R4>			; RESTORE
	RETURN				; EXIT
;
1000$:	CALL	FIL.ST			; DISPLAY ERROR CAUSE
	.EXIT				; QUIT
;
;

;
;
;
.SBTTL	PRIMITIVE	- 'SFLOUC'	...	OUTPUT A CHARACTER
;
;
;	THIS ROUTINE WILL PLACE THE CHARACTER IN R0 INTO THE DATA
;	FILE.
;
;
;
SFLOUC:	.PUSH	<R4,R5>			; SAVE
	MOV	OUTCNT,R4		; NUM CHRS IN BUF -> R4
	MOVB	R0,OUTBUF(R4)		; SAVE CHAR
	INC	R4			; UP COUNT
	CMP	R4,#1000		; FULL BUFFER?
	BGT	1000$			; > -> HARD ERROR
	BNE	100$			; < -> OK
	WRITE	FDB=#FDB,BLOCK=OUTBLK,BUFFER=#OUTBUF,ERROR=1100$
	INC	OUTBLK			; BUMP UP BLOCK NUMBER
	CLR	R4			; RESET CHAR COUNTER
;
100$:	MOV	R4,OUTCNT		; SAVE COUNT
	.POP	<R5,R4>			; SAVE
	RETURN				; EXIT
;
;
1000$:	.PRINT	#1777$			; TELL FAILED
1100$:	CALL	FIL.ST			; SHOW ERROR CAUSE
	.EXIT				; QUIT
;
.NLIST	BIN
1777$:	.ASCIZ	/"SFLEIO" -?????- WE IS STUFFED!/
.EVEN
.LIST	BIN
;
;

;
;
;
.SBTTL	DATA AREA FOR 'SFLEIO'
;
;
;
	.PSECT	$$$FDB			; OPEN FILES AREA
;	------	------
;
;
	FDB	NAME=FDB,CHANNEL=15.,BUFFER=SFLPAR,SIZE=400
;
;
INPCNT:	.WORD	0			; NUMBER WORDS IN BUFFER
OUTCNT:	.WORD	0
INPBLK:	.WORD	-1			; BLOCK NUMBER
OUTBLK:	.WORD	-1			; BLOCK NUMBER
SFLPAR:	.BLKW	400			; PARAMETER BLOCK
INPBUF:	.BLKW	400			; INPUT BUFFER
OUTBUF:	.BLKW	400			; OUTPUT BUFFER
;
;

.END
