; THIS FILE CONTAINS THE CODE FOR THE 'SORTER' PROGRAM, WHICH IS USED
; TO SORT RECORDS. THE MODULE IS USED AS FOLLOWS-
;
;	THE CALLER PLACES RECORDS TO BE SORTED INTO THE 'SORTER.DAT'
;	FILE. RECORDS ARE PLACED IN THE FILE STARTING AT BLOCK 1, WORD
;	ZERO. THE FIRST WORD OF EACH RECORD IS ASSUMED TO BE A RECORD
;	NUMBER, AND WHEN SORTING THE RECORDS IS IGNORED. THE CALLER THEN
;	PLACES THE SIZE OF EACH RECORD (IN BYTE) IN BLOCK 0, WORD 0 OF
;	THE FILE, AND THE NUMBER OF RECORDS IN BLOCK 0, WORD 1. THE
;	CALLER THEN CHAINS THE TO PROGRAM. ONCE THE FILE HAS BEEN SORTED,
;	THE PROGRAM WILL CHAIN BACK THE THE PROGRAM WHOSE NAME IS IN
;	.ASCIZ FORMAT IN THE STRING IN BLOCK 0, WORD 4 ONWARDS OF THE
;	DATA FILE. IF THE PROGRAM IS INVOKED VIA THE 'RUN' COMMAND INSTEAD
;	OF BEING CHAINED TO, THE PROGRAM ISSUES A '.EXIT' AT THE END
;	INSTEAD OF CHAINING.
;
;
;

;
;
;
.SBTTL	DOCUMENTATION	- DATA FILE FORMAT
;
;
;	THE DATA FILE CONSISTS OF TWO SECTIONS. THE TWO SECTIONS ARE
;	FORMATED AS FOLLOWS-
;
;	BLOCK 0		THE PARAMETER BLOCK. THE SIZE OF EACH RECORD
;			IS STORED IN WORD 0, AND THE NUMBER OF RECORDS
;			IN WORD 1. THE NAME OF THE PROGRAM TO CHAIN BACK
;			TO IS STORED IN THE ASCIZ STRIN STARTING AT
;			BYTE OFFSET 4 FROM THE START OF THE BLOCK.
;
;	BLOCK 1-N	A SERIES OF RECORDS. THE FORMAT OF EACH RECORD
;			IS-
;
;			START-	.WORD	NUM	; RECORD NUMBER
;				.BLKB	N	; DATA TO BE SORTED
;
;			THE SIZE OF THE RECORD IS THEREFOR N+2. NOTE
;			THAT THE FIRST WORD OF THE RECORD WILL BE IGNORED
;			IN THE SORT PHASE.
;
;

;
;
;
.SBTTL	DOCUMENTATION	- SORT ALGORITHM
;
;
;	AT START UP, THE PROGRAM WILL ASK THE MONITOR FOR EXTRA MEMORY.
;	THIS MEMORY IS THEN USED TO MAKE QUEUE ELEMENTS, WHICH ARE LINKED
;	TO A FREE QUEUE.
;
;	THE PROGRAM THEN INPUTS A RECORD, AND STORES ITS DATA IN A QUEUE
;	ELEMENT. IT THEN PLACES THE ELEMENT ON THE ACTIVE QUEUE. THE
;	ELEMENT IS LINKED AFTER ALL ELEMENTS WHICH ARE LARGER THAN IT.
;	(IE THE QUEUE IS ORDERED IN DECENDING ORDER). IF THE PROGRAM
;	RUNS OUT OF FREE QUEUE ELEMENTS, IT STARTS WRITTING OUT THE
;	CONTENTS OF THE LAST ELEMENTS ON THE ACTIVE QUEUE. (IE IT
;	WRITES OUT THE SMALLEST RECORD.) IT THEN DELINKS THE ELEMENT,
;	AND USES IT TO HOLD THE DATA OF THE NEXT RECORD. IF AT ANY
;	STAGE A RECORD IS NOT LINKED AT THE START OF THE ACTIVE QUEUE
;	IT SETS A FLAG TO REQUEST ANOTHER SORT PASS. THIS PROCESS
;	CONTINUES TILL ALL RECORDS HAVE BEEN PLACED IN MEMORY.
;
;	ONCE ALL RECORDS ARE IN MEMORY, THE ACTIVE QUEUE IS SCANNED FROM
;	THE BACK. THE LAST ELEMENT IS REMOVED, ITS DATA WRITTEN OUT,
;	AND THE ELEMENT RELINKED TO THE FREE QUEUE. THE DATA FILE IS
;	THEN REWOUND. THE RESORT FLAG IS THEN CHECKED. IF THE RECORDS
;	WERE NOT SORTED THIS TIME, THE SORT PASS IS RESTARTED.
;
;	IF NO SCHUFFLING WAS DONE ON THE LAST PASS, THE DATA FILE IS
;	CLOSED OUT, AND THE PROGRAM EXITS.
;
;

;
;
;
.SBTTL	NOTES
;
;
;	THE PROGRAM WILL ABORT IF FILE I/O OR QUEUE ERRORS OCCUR. THE
;	PROGRAM WILL PRINT OUT AN APPROPRIATE ERROR MESSAGE PRIOR TO
;	EXITING.
;
;	ALL FILE I/O OCCURES ON CHANNEL 15 (DECIMAL). THIS CHANNEL
;	IS PURGED AT ENTRY, AND THEN CLOSED OUT! THE DATA FILE IS
;	'SORTER.DAT'
;
;
;

;
;

;
;

;
	FILSPT				; INVOKE FILE SUPPORT
;
;
.GLOBL	CNAFD5				; 'CNAF'
.GLOBL	FIL.ST				; FILE STATUS
;
	.PSECT	CODE			; OPEN CODE SECTION

;
;
JSW	= 44				; ADDRESS JSW
 JSW.CH	= 400				; CHAIN BIT
;
;
.MACRO	ERRON	CND,MES,?X,?Y,?Z
	B'CND	X
	BR	Y
X:	.PRINT	#'Z
	.EXIT
Z:	.ASCII	/"SORTER" -FATAL-/

.EVEN

.ENDM	ERRON

;
;
.SBTTL	ENTRY POINT	- SORTER
;
;
;
;
SORTER:	.PRINT	#HEADER			; IDENTIFY SELF
	CALL	SFLINT			; INITIALIZE FILE I/O
	CALL	QUEINT			; INITILIZE QUEUES
;
;
;

;
;
;
.SBTTL	MAIN LINE CODE
;
;
;
;	WRITE OUT THE NUMBER OF THIS PASS.
;
	CLR	R5			; PASS COUNTER
1000$:	INC	R5			; UP PASS COUNTER
	MOV	R5,R2			; PASS COUNT -> R5
	MOV	#MESP.N,R1		; R1 --> NUMERIC BUFFER
	CLR	R0			; NO FORMAT CONTROL
	CALL	CNAFD5			; CHANGE PASS COUNT TO ASCII
	.PRINT	#MESP			; SHOW PASS NUMBER
;
;
;	SET UP PASS VARIABLES. R1 HOLDS SIZE OF RECORDS IN BYTES,
;	WHILE R2 HOLDS NUMBER OF RECORDS TO SORT. RESET 'DONFLG'.
;
	MOV	SFLPAR,R1		; R1 --> ENTRY SIZE
	MOV	SFLPAR+2,R2		; R2 --> NUMBER RECORDS
	CLR	DONFLG			; RESET DONE FLAG
;
;
;	MAIN LOOP. GET A QUEUE ELEMENT, INPUT NEXT RECORD INTO ITS
;	DATA AREA, INSERT THE ELEMENT INTO THE ACTIVE QUEUE. LOOP
;	TILL ALL RECORDS PROCESSED.
;
1100$:	CALL	GETELM			; GET AN ELEMENT
	CALL	INPUT			; INPUT THE DATA
	CALL	INSERT			; INSERT ELEMENT IN QUEUE
	SOB	R2,1100$		; LOOP
;
;
;	WRITE OUT ALL ELEMENTS IN MEMORY AND RESET DATA FILE (REWIND).
;	IF 'DONFLG' IS NON-ZERO (==> NOT FINISHED MOVING DATA ARROUND)
;	THEN RESTART SORT PHASE, ELSE EXIT.
;
	CALL	EMPTY			; EMPTY CONTENTS OF QUEUE
	CALL	SFLRST			; RESET SORT FILE
	TST	DONFLG			; ALL DONE?
	BNE	1000$			; NO -> LOOP
	CALL	SFLEND			; CLOSE OUT FILE
;
;
;

;
;
;
;	WE HAVE COMPLETED SORT. SEE IF WE WHERE CHAINED TO. IF WE WERE
;	CHAINED TO THEN PASS CONTROL BACK TO CALLER, ELSE EXIT.
;
;
	BIT	#JSW.CH,@#JSW			; CHAINED?
	BNE	6000$				; YES -> 6000$
	.PRINT	#7000$				; PRINT MESSAGE
	.EXIT					; BYE
;
;
6000$:	.PRINT	#7010$				; TELL ATTEMPTING TO CHAIN
	NAME	FDB=#DMYFDB,STRING=#<SFLPAR+4>,ERROR=6400$
	LOOKUP	ERROR=6500$			; ENSURE FILE EXISTS
	MOV	DMYFDB,R1			; R1 -> FILE NAME
	PURGE					; PURGE CHANNEL
	MOV	#500,R0				; R0 -> DEST
	.REPT	4				; --- 4 WORDS RAD50 ---
		MOV	(R1)+,(R0)+		;  SET UP FILE NAME
	.ENDR					; ---------------------
	.CHAIN					; PASS CONTROL
;
;
6400$:	.PRINT	#7020$				; TELL FAILED
	.EXIT					; QUIT
;
6500$:	.PRINT	#7030$				; TELL FAILED
	.PRINT	DMYFDB+2			; PRINT FILE NAME
	.EXIT
;
;
;
;
.NLIST	BIN
7000$:	.ASCIZ	/"SORTER" - SORT COMPLETE!/<12><12><12>
7010$:	.ASCII	/"SORTER" - SORT COMPLETE/<12><15>
	.ASCIZ	/          - WILL ATTEMPT TO CHAIN BACK TO CALLER!/
7020$:	.ASCIZ	/"SORTER" - CALLING PROGRAM NAME NOT IN PARBLK?/<12><12>
7030$:	.ASCII	/"SORTER" - CALLING PROGRAM CANNOT BE FOUND - /<200>
	FDB	DMYFDB,CHANNEL=15.		; DUMMY FILE DES BLOCK
.LIST	BIN
;
;

;
;
;
.SBTTL	ROUTINE	- 'GETELM'	...	GET AN ELEMENT
;
;
;	THIS ROUTINE RETURNS THE ADDRESS OF A QUEUE ELEMENT IN R0.
;	IT WILL GET ONE OFF THE FREE QUEUE. IF NONE ARE AVAILABLE
;	ON THE FREE QUEUE IT WILL REMOVE THE LAST ELEMENT OF THE
;	ACTIVE QUEUE, WRITE OUT ITS CONTENTS AND RETURN ITS ADDRESS.
;
;
;
;
;
GETELM:	CALL	QUEOFF			; TRY AND GET A FREE ELEMENT
	BCC	100$			; OK -> SKIP
	CALL	QUEOF			; GET A BUSY ELEMENT
	ERRON	CS,<'GETELM' CANNOT GET AN ELEMENT!>
	.PUSH	R0			; SAVE ELEMENT ADDRESS
	ADD	#4,R0			; POINT TO DATA
	CALL	SFLOUT			; WRITE IT OUT
	.POP	R0			; RESTORE ADDRESS
100$:	RETURN				; EXIT
;
;

;
;
;
.SBTTL	ROUTINE	- 'INPUT'	...	INPUT NEXT RECORD
;
;
;	THIS ROUTINE WILL INPUT THE THE RECORD FROM THE DATA FILE, AND
;	STORE IT IN THE DATA SECTION OF ELEMENT R0. THE SIZE OF THE
;	RECORD IS PASSED OVER IN R1 BY THE MAIN LINE CODE.;
;
;
INPUT:	.PUSH	R0			; SAVE ELEMENT ADDRESS
	ADD	#4,R0			; POINT TO DATA AREA
	CALL	SFLINP			; GET DATA
	.POP	R0			; RESTORE
	RETURN				; EXIT
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'INSERT'	...	INSERT ELEMENT R0 INTO ACTIVE QUEUE
;
;
;	THIS ROUTINE WILL INSERT ELEMENT R0 INTO THE ACTIVE QUEUE JUST AFTER
;	THE SMALLEST ELEMENT WHICH IS LARGER THAN IT. THE '10000$' ROUTINE
;	IS CALLED TO COMPARE THE NEW ELEMENT WITH ONE ALREADY ON THE QUEUE.
;	 IF THE NEW ELEMENT IS NOT INSERTED AT HEAD OF QUEUE SET 'DONFLG'.
;
;
INSERT:	.PUSH	<R1,R2,R3,R4,R5,R0>	; SAVE
	ADD	#6,R0			; POINT TO DATA TO BE SORTED
	MOV	R0,10010$		; SAVE FOR '10000$' LOOP
	SUB	#2,R1			; FORGET RECORD NUMBER
	MOV	R1,10020$		; SET UP FOR LOOP
;
	MOV	#QUEHDR,R5		; R5 --> PARENT
	MOV	(R5),R4			; R4 --> CHILD
	BEQ	2000$			; AT EOQ -> EXIT
	CALL	10000$			; SEE IF AT RIGHT PLACE
	BHIS	2000$			; YES -> EXIT
	MOV	#1,DONFLG		; WILL HAVE TO RESORT
1000$:	MOV	R4,R5			; CHILD IS NEW PARENT
	MOV	(R5),R4			; R4 -> CHILD
	BEQ	2000$			; AT EOQ -> EXIT
	CALL	10000$			; AT RIGHT PLACE
	BLO	1000$			; LOOP IF NOT THERE
;
2000$:	MOV	R5,R1			; R1 --> PARENT
	.POP	R0			; RESTORE ELMT ADD
	CALL	QUEJN			; ADD TO QUEUE
	.POP	<R5,R4,R3,R2,R1>	; RESTORE
	RETURN				; EXIT
;
;
;
;	THIS ROUTINE COMPARES THE DATA SECTIONS OF THE NEW ELEMENT AND
;	ELEMENT R4 BYTE BY BYTE, START AT RECORD BYTE 2 (BYTES 0,1 ARE
;	THE RECORD NUMBER). EFFECTIVE 'CMP ELEMNEW,ELEMR4'
;
10000$:	MOV	(PC)+,R0		; RESET DATA POINTER
10010$:	.WORD	0			; START SOURCE DATA
	MOV	(PC)+,R3		; NUMBER BYTES TO CHECK
10020$:	.WORD	0			; HOLDS NUMBER OF BYTES
	MOV	R4,R2			; ELEM ADD -> R2
	ADD	#6,R2			; POINT TO DATA
10100$:	CMPB	(R0)+,(R2)+		; HOW COMPARE?
	BNE	10200$			; <> -> EXIT
	SOB	R3,10100$		; LOOP
10200$:	RETURN				; EXIT
;

;
;
;
.SBTTL	ROUTINE	- 'EMPTY'	...	EMPTY OUT ACTIVE QUEUE
;
;
;	THIS ROUTINE WILL EMPTY OUT THE CONTENTS OF THE ACTIVE QUEUE.
;	IT REMOVES AN ELEMENT, OUTPUTS IT DATA, AND RETURNS IT TO
;	THE FREE QUEUE. IT DOES THIS TILL ALL ELEMENTS HAVE BEEN REMOVED
;	OFF THE ACTIVE QUEUE.
;
;
;
EMPTY:	CALL	QUEOF			; GET AN ELEMENT
	BCS	100$			; DONE -> 100$
	.PUSH	R0			; SAVE ADDRESS
	ADD	#4,R0			; POINT TO DATA
	CALL	SFLOUT			; OUTPUT IT
	.POP	R0			; RESTORE
	CALL	QUEJNF			; RETURN ELEMENT TO FREE LIST
	BR	EMPTY			; LOOP
100$:	RETURN				; EXIT
;
;
;

;
;
;
.SBTTL	VARIABLES/MESSAGES
;
;
DONFLG:	.WORD	0			; 1 -> RESORT
;
.NLIST	BIN
HEADER:	.ASCII	<12><12><12><12><12><12><12><12><12><12><12><12>
	.ASCII	<12><12><12><12><12><12><12><12><12><12><12><12>
	.ASCIZ	<12><15>/       SORTER (RDM310181.1)/<12><12>
;
MESP:	.ASCII	/ PASS /
MESP.N:	.ASCIZ	/XXXXX/
.EVEN
.LIST	BIN
;

.SBTTL	*                 *
.SBTTL	* MODULE 'QUEMNG' *
.SBTTL	*                 *
.SBTTL	*******************
;
;
;	THE FUNCTION OF THIS MODULE IS TO PROVIDE SUPPORT FOR THE
;	MANIPULATION OF QUEUE ELEMENTS. THE ROUTINES IN THE MODULE
;	ARE-
;
;	QUEINT	INITIALIZE THE QUEUES. THIS ROUTINE MUST BE CALLED AT
;		INITIALIZATION TIME. IT ISSUES A '.SETTOP' TO OBTAIN
;		ALL AVAILABLE MEMORY ABOVE THE PROGRAM HIGH LIMIT. IT
;		THEN USES THIS MEMORY TO FORM AS MANY QUEUE ELEMENTS
;		AS POSSIBLE, WHICH IT LINKS THE THE 'FREHDR' WORD.
;
;	QUEOFF	RETURNS THE ADDRESS OF A QUEUE ELEMENT WHICH IS CURRENTLY
;		FREE TO THE USER IN R0. THE 'C' FLAG IS SET IFF NO
;		ELEMENT IS CURRENTLY FREE.
;
;	QUEJNF	RETURNS ELEMENT R0 THE THE FREE QUEUE
;
;	QUEOF	RETURNS THE ADDRESS OF THE LAST ELEMENT IN THE ACTIVE
;		QUEUE IN R0, AFTER FREEING THE ELEMENT FROM THE QUEUE.
;		THE 'C' FLAG IS SET ON EXIT IFF NO ACTIVE ELEMENT WAS
;		FOUND.
;
;	QUEJN	INSERTS ELEMENT R0 IN THE ACTIVE QUEUE AFTER PARENT R1.
;
;
;	THE FOLLOWING VARIABLES ARE USED BY THE MODULE. THIS USER
;	SHOULD ACCESS ONLY THOSE MARKED WITH A '**' SYMBOL-
;
;	LIMIT		HOLDS PROGRAM LIMITS. SET UP BY 'LINKER.SAV'
;
;	MEMTOP		HOLDS HIGHEST ADDRESS OBTAINED WITH '.SETTOP'
;
;	NUMELM		HOLDS NUMBER QUEUE ELEMENTS CREATED
;
;	FREHDR		HEAD OF FREE QUEUE
;
;	QUEHDR	**	HEAD OF ACTIVE QUEUE
;
;	QUEEND		POINTS TO LAST ELEMENT ON ACTIVE QUEUE
;
;

;
;
;
.SBTTL	DOCUMENTATION	- 'QUEMNG' ELEMENT STRUCTURE
;
;
;	THE SIZE OF THE DATA TO BE STORED IN THE QUEUE ELEMENT IS
;	STORED IN THE FIRST WORD OF THE 'SFLPAR' BLOCK. THE INIT-
;	IALIZATION ROUTINE ADDS 4 (THERE ARE TWO LINK/WORDS PER
;	ELEMENT) TO THIS NUMBER, AND THEN ROUNDS IT UP TO MAKE IT
;	AN EVEN NUMBER OF BYTES. THE FORMAT OF AN ELEMENT IS-
;
;		ELEM N -	.WORD	FRWPNT
;				.WORD	BCKPNT
;				.BLKW	N
;
;	WHERE	FRWPNT	HOLDS THE ADRESS OF THE NEXT ELEMENT
;		ON THE QUEUE
;
;	BCKPNT	HOLDS THE ADRESS OF THE PREVIOUS ELEMENT ON THE
;		QUEUE
;
;	N	IS THE CONTENTS OF 'SFLPAR' WORD PLUS 1 DIVIDED BY
;		2.
;
;
.SBTTL	DOCUMENTATION	- 'QUEMNG' FREE QUEUE
;
;
;	THE 'FREHDR' WORD POINTS TO THE FIRST ELEMENT ON THE
;	FREE QUEUE. THE EOQ IS MARKED BY A LINK POINTER OF ZERO.
;	THE BACK POINTER AND DATA AREAS OF THE ELEMENTS ARE IGNORED.
;
;
;
.SBTTL	DOCUMENTATION	- 'QUEMNG' ACTIVE QUEUE
;
;
;	THE 'QUEHDR' WORD POINTS TO THE FIRST ELEMENT ON THE QUEUE.
;	THE 'QUEEND' WORD POINTS TO THE LAST ELEMENT ON THE QUEUE.
;	BOTH THESE WORDS ARE ZERO WHEN THERE ARE NO ELEMENTS ON THE
;	QUEUE. THE QUEUE CAN BE TRAVERSED IN BOTH THE FORWARD AND
;	BACKWARD DIRRECTION. THE LAST ELEMENT IN EITHER THE FORWARD
;	OR BACK DIRRECTION HAS ITS FORWARD/BACKWORD POINT SET TO
;	ZERO
;
;

;
;
;
.SBTTL	DECLARTIONS FOR 'QUEMNG'
;
;
.MCALL	.EXIT,.SETTOP,.PRINT		; RT11
.MCALL	.PUSH,.POP			; STACKING
;
;
	.PSECT	CODE			; OPEN CODE SECTION

;
;
;
.MACRO	ERRON	CND,MES,?X,?Y,?Z
	B'CND	X
	BR	Y
X:	.PRINT	#'Z
	.EXIT
Z:	.ASCII	/"QUEMNG" -FATAL-/
	.ASCII	/MES/<200>
.EVEN
Y:
.ENDM	ERRON
;

;
;
.SBTTL	ROUTINE	- 'QUEINT'	...	QUE INITIALIZATION
;
;
QUEINT:	.SETTOP	#-2			; GET ALL OF MEMORY
	MOV	R0,MEMTOP		; SAVE ALL OF MEMORY
	MOV	SFLPAR,R1		; RECORD SIZE -> R1
	CMP	R1,#3			; MUST BE 3 BYTES AT LEAST
	ERRON	EQ,<'QUEINT' RECORD TO SMALL>
	ADD	#5,R1			; ALLOW FOR LINK POINTERS
	BIC	#1,R1			; ENSURE EVEN NUM BYTES
;
	CLR	QUEHDR			; CLEAR FRONT HEADER
	CLR	QUEEND			; CLEAR BACK HEADER
	CLR	R2			; COUNT OF ELEMENTS
;
	MOV	#FREHDR,R5		; R5 --> LAST ELEMENT
	MOV	LIMIT+2,R4		; R4 --> NEXT ELEMENT
1000$:	MOV	R4,R3			; POINT R3 ...
	ADD	R1,R3			; ... TO NEXT
	CMP	R3,MEMTOP		; IN MEMORY RANGE
	BHIS	2000$			; NO -> EXIT
	MOV	R4,(R5)			; LINK THIS ELEMENT TO FREE QUEUE
	MOV	R4,R5			; MAKE THIS ELEMENT NEW HEAD
	MOV	R3,R4			; MAKE NEXT ELEMENT NEW NEXT
	INC	R2			; INCREMENT COUNT
	BR	1000$			; LOOP
;
2000$:	CLR	(R5)			; MARK EOQ
	CMP	R2,#4			; NEED AT LEAST 4 ELEMENTS
	ERRON	LOS,<'QUEINT' NOT ENOUGH MEMORY>
	MOV	R2,NUMELM		; SAVE NUMBER OF ELEMENTS
	RETURN				; EXIT
;
;

;
;
;
.SBTTL	ROUTINES	- 'QUEOFF'/'QUEJNF'	...	FREE QUEUE SUPPORT
;
;
;
QUEOFF:	MOV	FREHDR,R0		; GET ADDRESS NEXT ELEMENT
	BEQ	100$			; NONE -> SKIP
	MOV	(R0),FREHDR		; DELINK
	CLR	(R0)			; CLEAR POINTER
	TST	(PC)+			; CLEAR FAIL FLAG (SKIP NEXT)
100$:	SEC				; SET FAILED
	RETURN				; EXIT
;
;
QUEJNF:	MOV	FREHDR,(R0)		; LINK ...
	MOV	R0,FREHDR		; ... TO QUEUE
	RETURN				; EXIT
;
;

;
;
;
.SBTTL	ROUTINE	- 'QUEJN'	... JOIN ACTIVE QUEUE
;
;
;
QUEJN:	MOV	(R1),(R0)		; POINT ELEMENT TO NEXT
	BEQ	100$			; NO MORE -> 100$
	.PUSH	R1			; SAVE PARENT ADDRESS
	MOV	R0,(R1)			; POINT PARENT TO ELEMENT
	MOV	(R0),R1			; R1 --> NEXT
	MOV	2(R1),2(R0)		; POINT ELEMENT TO PARENT
	MOV	R0,2(R1)		; POINT NEXT BACK TO ELEMENT
	.POP	R1			; RESTORE
	RETURN				; EXIT
;
100$:	MOV	R1,2(R0)		; POINT BACK TO PARENT
	CMP	R1,#QUEHDR		; IS PARENT HEAD OF QUEUE
	BNE	110$			; NO SKIP
	CLR	2(R0)			; NO BACK POINTER
110$:	MOV	R0,(R1)			; SET UP HEADER
	MOV	R0,QUEEND		; SET UP BACK POINTER HEADER
	RETURN				; EXIT
;
;

;
;
;
.SBTTL	ROUTINE	- 'QUEOF'	... TAKE ELEM OFF ACTIVE QUEUE
;
;
QUEOF:	MOV	QUEEND,R0		; GET ADDRESS OF ELEMENT
	BEQ	1000$			; NONE -> EXIT
	MOV	2(R0),QUEEND		; DELINK ELEMENT
	BNE	100$			; NOT AT EOQ -> SKIP
	CMP	R0,QUEHDR		; ONLY ONE ON QUEUE!
	ERRON	NE,<'QUEOF' - QUEUE CORRUPTED!>
	CLR	QUEHDR			; CLEAR FRONT POINTER
100$:	CLC				; CLEAR FAIL
	RETURN				; EXIT
;
1000$:	TST	QUEHDR			; SHOULD BE ZERO
	ERRON	NE,<'QUEOF' - QUEHDR CORRUPTED!>
	SEC				; SET FAILED
	RETURN				; EXIT
;
;

;
;
;
.SBTTL	DATA AREA FOR 'QUEMNG'
;
;
;
	.PSECT	DATA			; OPEN DATA AREA
;	------	----
;
;
LIMIT:	.LIMIT				; LIMITS
MEMTOP:	.WORD	0			; TOP OF MEMORY
NUMELM:	.WORD	0			; NUMBER ELEMENTS
FREHDR:	.WORD	0			; HEAD FREE QUEUE
QUEHDR:	.WORD	0,0			; HEAD FORWARD QUEUE
QUEEND:	.WORD	0,0			; HEAD BACK QUEUE
;

.SBTTL	*******************
.SBTTL	*                 *
.SBTTL	* MODULE 'SFLEIO' *
.SBTTL	*                 *
.SBTTL	*******************
;
;
;	THE FUNCTION OF THIS MODULE IS TO ALLOW DATA TO BE TRANSFERED
;	TO AND FROM THE SORT DATA FILE. THE ROUTINES PROVIDED ARE
;
;		SFLINT	...	MUST BE CALLED FIRST TO INITIALIZE
;				THE BUFFERS, CHANNELS ETC. DESTROYS
;				ALL REGISTERS.
;
;		SFLRST	...	MUST BE CALLED AT THE END OF A PASS
;				TO RESET THE INPUT/OUTPUT BUFFERS.
;				EFFECTIVELY REWINDS THE FILE.
;
;		SFLEND	...	CALLED AT END OF I/O TO CLOSE OUT
;				THE FILE.
;
;		SFLINP	...	INPUT 'R1' CHARACTERS INTO BUFFER @ 'R0'
;
;		SFLOUT	...	OUTPUTS 'R1' CHARS FROM BUFFER @ 'R0'
;
;		SFLINC	...	RETURNS NEXT INPUT CHARACTER IN R0.
;
;		SFLOUC	...	OUTPUTS CHARACTER IN R0 TO FILE.
;
;
;	THE FOLLOWING VARIABLES/BUFFERS ARE USED BY THE PROGRAM. THE
;	USER MAY ACCESS ONLY THOSE MARKED WITH AN ** SYMBOL-
;
;	SFLPAR **	HOLDS BLOCK '0' OF THE SORT DATA FILE. THE
;			CONTENTS THIS 400 (OCTAL) WORD BUFFER CAN BE
;			MODIFIED TO SEND INFORMATION TO THE NEXT
;			PROGRAM TO USE THE FILE AS IT WILL BE WRITTEN
;			OUT WHEN THE FILE IS CLOSED.
;
;	INPBUF/OUTBUF	400 (OCTAL) WORDS FILE BUFFERS
;
;	INPCNT/OUTCNT	COUNT OF CHARACTERS IN INPUT/OUTPUT BUFFERS
;
;	FDB		FILE DESCRIPTOR BLOCK FOR DATA FILE.
;
;

;
;
;
.SBTTL	DOCUMENTATION	- 'SFLEIO' FILE STRUCTURE
;
;
;	'SFLEIO' IS USED TO ACCESS THE DATA IN THE "DK:SORTER.DAT"
;	DATA FILE. THIS FILE IS TREATED AS COMPOSED OF TWO SECTIONS.
;	THESE SECTIONS ARE -
;
;	BLOCK 0		A 400 (OCTAL) WORD PARAMETER BLOCK WHICH
;			WILL BE READ INTO 'SFLPAR' AT INITIALIZATION
;			TIME, AND REWRITTEN OUT AT EXIT TIME. THIS
;			PARAMETER BLOCK CAN BE USED BY THE CALLER
;			TO PASS ARGUMENTS BETWEEN PROGRAMS.
;
;	BLOCK 1-N	A STREAM OF 8 BIT BYTES, WHICH CAN BE READ/WRITTEN
;			SEQUENTIALLY. THE FILE CAN BE REWOUND AT ANY
;			TIME BY CALLING 'SFLRST'
;
;
;
.SBTTL	DOCUMENTATION	- 'SFLEIO' FILE I/O
;
;
;	ALL FILE I/O IS DONE USING THE 'FILSPT' MODULE. ANY FILE I/O
;	ERRORS WILL CAUSE A FATAL ERROR MESSAGE TO BE PRINTED.
;
;
;
;
;
.SBTTL	DECLARATIONS FOR 'SFLEIO'
;
;
.MCALL	.PUSH,.POP			; STACKING
.MCALL	.PRINT,.EXIT			; ERROR HANDLING
.MCALL	FILSPT				; FILE SUPPORT
;
	FILSPT				; INVOKE FILE SUPPORT
;
;
.GLOBL	FIL.ST				; "FILSPT"
;
;
	.PSECT	CODE			; OPEN CODE AREA
;	------	----
;

;
;
;
.SBTTL	INITIALIZATION SECTION
;
;
;	OPEN SORT DATA FILE FOR INPUT, AND READ IN THE PARAMETER
;	BLOCK. RESET THE INPUT/OUTPUT BLOCK/CHARACTER COUNTER.
;
SFLINT:	NAME	#FDB,#100$,ERROR=40$	; INSERT FILE NAME
	PURGE				; ENSURE CHANNEL OK
	LOOKUP	ERROR=60$		; LOOKUP FILE
	READ	BLOCK=#0,SIZE=#400,BUFFER=#SFLPAR,ERROR=60$
	MOV	#1,INPBLK		; START INPUTING BLOCK 1
	MOV	#1,OUTBLK		; START OUTPUTING BLOCK 1
	MOV	#1000,INPCNT		; 1000 BYTES IN INPUT BUFFER
	MOV	#0,OUTCNT		; NO BYTES IN OUTPUT BUFFER
	READ	FDB=#FDB,BLOCK=#1,BUFFER=#INPBUF,ERROR=60$
	RETURN				; ALL DONE
;
40$:	.PRINT	110$			; TELL FAILED
60$:	CALL	FIL.ST			; DISPLAY FILE STATUS
	.EXIT				; QUIT
;
.NLIST	BIN
.ENABL	LC
100$:	.ASCIZ	/DK:SORTER.DAT/		; FILE SPEC
	.BLKB	120-.+100$		; PAD OUT (***FIL.NM***)
110$:	.ASCIZ	/"SFLEIO" -??????- HOW CAN NAMING FAIL?/
.EVEN
.DSABL	LC
.LIST	BIN
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'SFLRST'	...	RESET SORT FILE
;
;
;	THIS ROUTINE IS CALLED TO RESET THE SORT FILE PRIOR TO A NEW
;	PASS. IT EMPTIES OUT THE CONTENTS OF THE OUTPUT BUFFER AND
;	RESETS THE INPUT/OUTPUT BLOCK/CHARACTER COUNTERS.
;
;
;
SFLRST:	.PUSH	R5			; SAVE R5
	TST	OUTCNT			; ANY CHARACTERS TO OUTPUT?
	BEQ	100$			; NO -> 100$
	WRITE	FDB=#FDB,BLOCK=OUTBLK,BUFFER=#OUTBUF,ERROR=1000$
100$:	MOV	#1,INPBLK		; START INPUTING BLOCK 1
	MOV	#1,OUTBLK		; START OUTPUTING BLOCK 1
	MOV	#1000,INPCNT		; 1000 BYTES IN INPUT BUFFER
	MOV	#0,OUTCNT		; NO BYTES IN OUTPUT BUFFER
	READ	FDB=#FDB,BLOCK=#1,BUFFER=#INPBUF,ERROR=1000$
	.POP	R5			; RESTORE
	RETURN				; EXIT
;
1000$:	CALL	FIL.ST			; SHOW CAUSE OF ERROR
	.EXIT				; QUIT
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'SFLEND'	...	CLOSE SORT FILE
;
;
;	THIS ROUTINE IS CALLED WHEN THE USER IS FINISHED WITH THE
;	SORT FILE. IT EMPTIES OUT THE CONTENTS OF THE OUTPUT BUFFER
;	AND CLOSES OUT THE FILE.
;
;
;
SFLEND:	.PUSH	R5			; SAVE R5
	TST	OUTCNT			; ANY CHARACTERS TO OUTPUT?
	BEQ	100$			; NO -> 100$
	WRITE	FDB=#FDB,BLOCK=OUTBLK,BUFFER=#OUTBUF,ERROR=1000$
100$:	WRITE	FDB=#FDB,BLOCK=#0,BUFFER=#SFLPAR,ERROR=1000$
	PURGE	#FDB			; PURGE CHANNEL
	.POP	R5			; RESTORE
	RETURN				; BYE
;
;
1000$:	CALL	FIL.ST			; SHOW ERROR CAUSE
	.EXIT				; EXIT TIME
;
;

;
;
;
.SBTTL	ROUTINES	- 'SLFINP'/'SLFOUT'	...	INPUT/OUTPUT
;
;
;	THESE ROUTINES TRANSFER 'R1' CHARACTERS TO/FROM THE USER
;	BUFFER TO/FROM THE SORT DATA FILE. THE ADDRESS OF THE USER
;	BUFFER MUST BE IN 'R0'.
;
;
;
SFLINP:	.PUSH	<R0,R1,R2>		; SAVE
	MOV	R0,R2			; BUFFER ADDRESS -> R2
10$:	CALL	SFLINC			; GET ONE CHARACTER
	MOVB	R0,(R2)+		; SAVE IT
	SOB	R1,10$			; LOOP
	.POP	<R2,R1,R0>		; RESTORE
	RETURN				; EXIT
;
;
;
SFLOUT:	.PUSH	<R0,R1,R2>		; SAVE
	MOV	R0,R2			; BUFFER ADDRESS -> R2
10$:	MOVB	(R2)+,R0		; CHAR -> R0
	CALL	SFLOUC			; OUTPUT A CHARACTER
	SOB	R1,10$			; LOOP
	.POP	<R2,R1,R0>		; SAVE
	RETURN				; EXIT
;
;

;
;
;
.SBTTL	PRIMITIVE	- 'SFLINC'	...	INPUT A CHARACTER
;
;
;	THIS ROUTINE READS IN THE NEXT CHARACTER FROM THE SORT
;	DATA FILE IN RETURNS IT IN R0.
;
;
;
SFLINC:	.PUSH	<R4,R5>			; SAVE
	MOV	INPCNT,R4		; INPUT COUNT -> R4
	BNE	100$			; YES -> SKIP
	INC	INPBLK			; UP BLOCK COUNT
	READ	FDB=#FDB,BLOCK=INPBLK,BUFFER=#INPBUF,ERROR=1000$
	MOV	#1000,R4		; RESET COUNTER
;
100$:	MOV	#INPBUF+1000,R5		; R5 -> END OF BUFFER
	SUB	R4,R5			; R5 -> CHARACTER TO GET
	MOVB	(R5),R0			; GET CHAR
	DEC	R4			; ONE LESS CHAR IN BUFFER
	MOV	R4,INPCNT		; SAVE
	.POP	<R5,R4>			; RESTORE
	RETURN				; EXIT
;
1000$:	CALL	FIL.ST			; DISPLAY ERROR CAUSE
	.EXIT				; QUIT
;
;

;
;
;
.SBTTL	PRIMITIVE	- 'SFLOUC'	...	OUTPUT A CHARACTER
;
;
;	THIS ROUTINE WILL PLACE THE CHARACTER IN R0 INTO THE DATA
;	FILE.
;
;
;
SFLOUC:	.PUSH	<R4,R5>			; SAVE
	MOV	OUTCNT,R4		; NUM CHRS IN BUF -> R4
	MOVB	R0,OUTBUF(R4)		; SAVE CHAR
	INC	R4			; UP COUNT
	CMP	R4,#1000		; FULL BUFFER?
	BGT	1000$			; > -> HARD ERROR
	BNE	100$			; < -> OK
	WRITE	FDB=#FDB,BLOCK=OUTBLK,BUFFER=#OUTBUF,ERROR=1100$
	INC	OUTBLK			; BUMP UP BLOCK NUMBER
	CLR	R4			; RESET CHAR COUNTER
;
100$:	MOV	R4,OUTCNT		; SAVE COUNT
	.POP	<R5,R4>			; SAVE
	RETURN				; EXIT
;
;
1000$:	.PRINT	#1777$			; TELL FAILED
1100$:	CALL	FIL.ST			; SHOW ERROR CAUSE
	.EXIT				; QUIT
;
.NLIST	BIN
1777$:	.ASCIZ	/"SFLEIO" -?????- WE IS STUFFED!/
.EVEN
.LIST	BIN
;
;

;
;
;
.SBTTL	DATA AREA FOR 'SFLEIO'
;
;
;
	.PSECT	$$$FDB			; OPEN FILES AREA
;	------	------
;
;
	FDB	NAME=FDB,CHANNEL=15.,BUFFER=SFLPAR,SIZE=400
;
;
INPCNT:	.WORD	0			; NUMBER WORDS IN BUFFER
OUTCNT:	.WORD	0
INPBLK:	.WORD	-1			; BLOCK NUMBER
OUTBLK:	.WORD	-1			; BLOCK NUMBER
SFLPAR:	.BLKW	400			; PARAMETER BLOCK
INPBUF:	.BLKW	400			; INPUT BUFFER
OUTBUF:	.BLKW	400			; OUTPUT BUFFER
;
;
;
.END	SORTER
