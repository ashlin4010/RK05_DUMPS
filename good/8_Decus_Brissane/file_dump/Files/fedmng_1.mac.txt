;
; THIS MODULE CONTAINS THE ROUTINES NEEDED TO CREATE AND EDIT A FORM. A
; FORM CONSISTS OF A DATA RECORD AND A RECORD DESCTIPTOR BLOCK, WHICH
; SPECIFIES WHICH FIELDS ARE IN THE FORM AND HOW THEY ARE TO BE
; DISPLAYED ETC. THERE ARE TWO ENTRIES WITHIN THE MODULE. THEY ARE-
;
;	FED.NW	CREATE A NEW FORM/RECORD. THIS ENTRY POINT CLEARS
;		THE DATA IN THE RECORD AND PLACES THE EDITOR INTO
;		ENTRY MODE.
;
;	FED.OD	EDIT AN OLD RECORD. THIS ENTRY DISPLAYS THE CURRENT
;		DATA IN THE RECORD AND PLACES THE EDITOR IN COMMAND
;		MODE.
;
;
;
.SBTTL	MODIFICATIONS
;
;
; 14-JAN-81	SUPPORT 'WNDIOF' RESETTING.
; 28-APR-81	'ENTER' - SUPPORT 'BACKSP' KEY
; 30-APR-81	MAJOR CHANGES TO SUPPORT MULTIPLE PAGES
; 21-MAY-81	CHANGE EXIT CODE (WAS USING MTPS/MFPS-->BOMBED ON 11/40)
;
;

;
;
;
.SBTTL	DOCOMENTATION	- ARGUMENT PASSING AND REGISTER USAGE
;
;
;	AT ENTRY THE ADDRESS OF THE RDT IS PASSED OVER IN R5. ALL REGISTERS
; ARE PRESERVED THROUGH THE ROUTINE. AT EXIT THE 'C' FLAG IS SET IFF THE
; ABORT COMMAND WAS ENTERED.
;
;
;
.SBTTL	DOCUMENATION	- 'FED.MD' WORD
;
;
; THE USER CAN SET BIT 3 IN THIS WORD IF WISHS CONTROL TO PASS BACK TO THE 
; CALLER AS SOON AS ENTRY MODE IS EXITTED AFTER EXECUTING A 'FED.NW' CALL. THIS
; EFFECTIVELY INHIBITS CONTROL PASSING BACK TO THE MODULE COMMAND LEVEL.
;
;
;
;
.SBTTL	DOCUMENTATION	- COMMAND MODE
;
;
; THE EDITOR COMMAND INTERPRETOR RECOGNIZES THE FOLLOWING COMMANDS-
;
;		FIRST/TOP	START EDITING AT FIRST FIELD
;		EDIT/ENTER	ENTER DATA ENTRY MODE
;		LOCATE		START EDITING LOCATED FIELD
;		REFRESH		REFRESH SCREEN
;		CLEAR		CLEAR/ZERO RECORD
;		ABORT		ERROR EXIT
;		QUIT		EXIT WITHOUT ERROR
;
;

;
;
;
.SBTTL	DECLARATIONS
;
;
.MCALL	.PUSH,.POP				; STACKING
;
;
.GLOBL	FED.OD,FED.NW				; ENTRIES
.GLOBL	FED.MD					; MODE CONTROL FLAG
.GLOBL	PAGE$$					;; PAGE NUMBER
;
.GLOBL	TBLKUP					; "TBLKUP" TABLE LOOKUP
.GLOBL	FRM.EF,FRM.ZE,FRM.BG,FRM.FG		; "FRMIO"
.GLOBL	FLD.OU,FLD.IN,FLD.CK			; "FLDIO"
.GLOBL	WND.IN					; "WNDIO"
.GLOBL	WND.FL					; "WNDIO"
.GLOBL	ASCNUM					; "ASCNUM"
.GLOBL	CNAFD2					; 'CNAFD'
;
;
;
	.PSECT	CODE				; OPEN CODE AREA
;	------	----
;
;
ENTSIZ	= 40					; FDB SIZE IN BYTES
;

;
;
;
.SBTTL	MACRO	- 'CMDTAB'	...	CREATE A COMMAND TABLE
;
;
;
;	THIS MACRO IS USED TO CREATE A COMMAND TABLE WHICH CAN BE USED
;	BY 'TBLKUP' FOR COMMAND DECODING. TO USE IT DO THE FOLLOWING-
;
;		CMDTAB	NAME		; CREATE AND LABLE TABLE
;		CMDENT	COMMAND,ROUTINE	; COMMAND --> ROUTINE
;		.......
;		CMDEND			; TERMINATE TABLE
;
;	'TBLKUP' IS CALLED WITH THE COMMAND STRING IN 'R0' AND THE TABLE
;	ADDRESS (NAME) IN 'R1'. IT RETURNS WITH THE ROUTINE ADDRESS IN
;	2(R1) IF IT FINDS THE COMMAND IN THE TABLE.
;
;
;
;
.MACRO	CMDTAB	TABLE
TABLE:	.MACRO	CMDENT	NAME,VAL1,VAL2=0,VAL3=0
		.SAVE
		.PSECT	MESSAGE
		ZZZZZ1	= .
		.ASCIZ	/NAME/
		.RESTORE
		.WORD	ZZZZZ1,VAL1,VAL2,VAL3
	.ENDM	CMDENT
;
	.MACRO	CMDEND
		.WORD	0,0,0,0
		.MACRO	CMDTAB
		.ENDM	CMDTAB
		.MACRO	CMDENT
		.ENDM	CMDENT
	.ENDM	CMDEND
;
.ENDM	CMDTAB
;

;
;
;
.SBTTL	MACRO	- 'INPUT'	...	INPUT A COMMAND
;
;
;	THIS MACRO IS USED TO INPUT A COMMAND LINE. THE ADDRESS OF THE
;	INPUT BUFFER IS RETURNED IN R0, AND IT IS TERMINATED BY A NULL.
;	OPTIONAL PROMPT, STATUS AND ERROR MESSAGES CAN BE SPECIFIED.
;
;
;
;
.MACRO	INPUT	BUFFER=IOM.BF,PROMPT,STATUS,ERROR
	.IF	NB,ERROR
		.PUSH	R0
		MOV	ERROR,R0
		.GLOBL	IOM.ER
		CALL	IOM.ER
		.POP	R0
	.ENDC
;
	.IF	NB,STATUS
		.PUSH	R0
		MOV	STATUS,R0
		.GLOBL	IOM.ST
		CALL	IOM.ST
		.POP	R0
	.ENDC
;
	.IF	NB,PROMPT
		.PUSH	R0
		MOV	PROMPT,R0
		.GLOBL	IOM.PT
		CALL	IOM.PT
		.POP	R0
	.ENDC
;
	MOV	BUFFER,R0
	.IIF	IDN,BUFFER,IOM.BF,.GLOBL	IOM.BF
	.GLOBL	IOM.IN
	CALL	IOM.IN
;
.ENDM	INPUT
;
;

;
;
;
.SBTTL	MACRO	- 'DISPLAY'	... DISPLAY PROMPT, STATUS AND ERROR MESSAGES
;
;
;	THIS MACRO IS USED TO DISPLAY THE PROMPT, STATUS AND ERROR MESSAGES.
;
;
;
.MACRO	DISPLAY	PROMPT,STATUS,ERROR
	.IF	NB,ERROR
		.PUSH	R0
		MOV	ERROR,R0
		.GLOBL	IOM.ER
		CALL	IOM.ER
		.POP	R0
	.ENDC
;
	.IF	NB,STATUS
		.PUSH	R0
		MOV	STATUS,R0
		.GLOBL	IOM.ST
		CALL	IOM.ST
		.POP	R0
	.ENDC
;
	.IF	NB,PROMPT
		.PUSH	R0
		MOV	PROMPT,R0
		.GLOBL	IOM.PT
		CALL	IOM.PT
		.POP	R0
	.ENDC
;
.ENDM	DISPLAY
;
;

;
;
;
.SBTTL	MACRO	- 'PROMPT'	...	OUTPUT PROMPT
;
;
;
.MACRO	PROMPT	MESSAGE
	.IF	NB,MESSAGE
		.PUSH	R0
		MOV	MESSAGE,R0
	.ENDC
	.GLOBL	IOM.PT
	CALL	IOM.PT
	.IIF	NB,MESSAGE,	.POP	R0
.ENDM	PROMPT
;
;
;

;
;
;
.SBTTL	MACRO	- 'STATUS'	...	DISPLAY STATUS MESSAGE
;
;
;
.MACRO	STATUS	MESSAGE
	.IF	NB,MESSAGE
		.PUSH	R0
		MOV	MESSAGE,R0
	.ENDC
	.GLOBL	IOM.ST
	CALL	IOM.ST
	.IIF	NB,MESSAGE,	.POP	R0
.ENDM	STATUS
;
;

;
;
;
;
.SBTTL	MACRO	- 'ERROR'	...	DISPLAY ERROR MESSAGE
;
;
;
.MACRO	ERROR	MESSAGE
	.IF	NB,MESSAGE
		.PUSH	R0
		MOV	MESSAGE,R0
	.ENDC
	.GLOBL	IOM.ER
	CALL	IOM.ER
	.IIF	NB,MESSAGE,	.POP	R0
.ENDM	ERROR
;
;
;

;
;
;
.SBTTL	ENTRY	- 'FED.NW'	...	CREATE A NEW FORM
;
;
;	SET BITS TO CAUSE ENTRY MODE TO BE ENTERED AND RECORD TO BE ERASED.
;
;
;
FED.NW:	BIS	#3,FED.MD		; ZERO RECORD - ENTERY MODE
	MOV	#1,PAGE$$		;; START ENTERING DATA FOR PAGE 0/1
	JMP	CMDMNG			; GOTO MAIN LOOP
;
;
;
.SBTTL	ENTRY	- 'FED.OD'	...	EDIT AN OLD FORM
;
;
;	SET BIT TO ENTER 'ENTRY' MODE. ENSURE THAT 'CLEAR' BIT IS DISABLED.
;
;
FED.OD:	BIC	#1,FED.MD		; NO ZERO
	BIS	#2,FED.MD		; ENTERY MODE
	MOV	PAGE$$+2,PAGE$$		;; RESELECT LAST EDITTED PAGE
	JMP	CMDMNG			; GOTO MAIN LOOP
;
;
;

;
.SBTTL	ROUTINE	- 'CMDMNG'	...	COMMAND MANAGER
;
;
;	SAVE REGISTERS AND ADDRESS OF RDT AND RECORD BUFFER. SET UP NUMBER
;	OF PAGES IN FORM IN THE STATUS MESSAGE FOR LATER USE.
;
;
CMDMNG:	.PUSH	<R0,R1,R2,R3,R4,R5>			; SAVE
	MOV	R5,RECPNT				; SAVE RDT ADDRESS
	MOV	(R5),RECADD				; SAVE REC ADDRESS
	CLR	R0					;; FORMAT = 0
	MOV	#9011$,R1				;; R1 ---> BUFFER
	MOV	10(R5),R2				;; R2 = PAGES
	CALL	CNAFD2					;; GET NUMBER PAGES
	MOV	#9012$,R1				;; R1 --> BUFFER
	MOV	PAGE$$,R2				;; R2 == PAGE NUMBER
	CALL	CNAFD2					;; CHANGE TO ASCII
;
;
;	CLEAR ERROR LINE AND ZERO OUT RECORD IF CALLED IN 'NEW' MODE.
;
;
	ERROR	#0					; NO ERRORS AS YET
	MOV	R5,R0					; RDT -> R5
	MOV	(R5),R5					; BUFFER ADDRESS -> R5
	BIT	#1,FED.MD				; ZERO RECORD?
	BEQ	10$					; NO -> SKIP
	CALL	FRM.ZE					; ZERO RECORD
;
;
;
;	REFRESH SCREEN AND ACTIVATE FIRST FIELD ON SCREEN.
;
;
10$:	CALL	FRM.EF					; ERASE FORM AREA
	CALL	FRM.BG					; DISPLAY BACKGROUND
	CALL	FRM.FG					; DISPLAY FG
	CALL	MOVTOP					;; MOVE TO TOP OF FORM
;
;
;	SEE IF WE ARE TO DROP INTO ENTRY MODE IMMEDIATELY AND IF SO THEN
;	CHECK WHETHER WE ARE TO EXIT IMMEDIATELY OR ENTER COMMAND MODE.
;
;
	BIT	#2,FED.MD				; ENTERY MODE?
	BEQ	20$					; NO -> 20$
	CALL	.EDIT					;; ENTER DATA
	BIT	#4,FED.MD				; EXIT IMMEDIATELY?
	BNE	774$					; YES -> 774$
;

;
;
;
;	COMMAND MODE. SWITCH TO NORMAL EDIT MODE AND ENSURE THAT STATUS MESSAGE
;	REFLECTS CURRENTLY SELECTED PAGE.
;
;
20$:	MOVB	#1,WND.FL+1				;; NORMAL EDIT MODE
700$:	CLR	R0					;; FORMAT = 0
	MOV	#9012$,R1				;; R1 ---> BUFFER
	MOV	PAGE$$,R2				;; R2 = PAGE NUMBER
	CALL	CNAFD2					;; GET PAGE NUMBER
;
;
;	GET A COMMAND AND CLEAR ERROR MESSAGE LINE.
;
;
	INPUT	,#900$,#901$				; INPUT COMMAND
	ERROR	#0					; CLEAR ERRORS
;
;
;	IF <NULL> LINE INPUT THEN TERMINATE IMMEDIATELY ELSE TRY AND DECODE
;	COMMAND.
;
;
	TSTB	(R0)					; NULL INPUT?
	BEQ	774$					; YES -> EXIT
	MOV	#1000$,R1				; COMMAND TAB ADD -> R1
	CALL	TBLKUP					; LOOKUP
	BCC	770$					; OK -> 770$
	ERROR	#902$					; TELL NO GO
	BR	700$					; LOOP
;
;
;

;
;
;
;	DECODED COMMAND SUCESSFULLY. PASS CONTROL TO COMMAND ROUTINE
;	WITH REGISTERS SET UP AS FOLLOWS-
;
;
;		R0	--> RDT
;		R1	--> DECODED COMMAND ARG BLOCK
;		R2	--> REST OF INPUT STRING
;
;
770$:	MOV	R0,R2					;; SAVE REST OF INPUT
	MOV	RECPNT,R0				; RESET POINTER
	CALL	@2(R1)					; EXECUTE
;
;
;
;	IF COMMAND ROUTINE RETURNS WITH 'Z' CLEAR THEN RELOOP AND GET ANOTHER
;	COMMAND ELSE EXIT. WHEN EXIT SAVE CURRENT PAGE NUMBER IN 'PAGE$$+2'
;	IF NOT STARTED IN 'NEW' MODE AND RESTORE 'PAGE$$'.
;
;
	BNE	700$					; OK -> CONTINUE
774$:	MOV	#0,R0					;; 0 -> R0
	BCC	776$					;; NO CARRY -> SKIP
	MOV	#100000,R0				;; SET UP TIME
776$:	BIT	#1,FED.MD				;; WAS NEW MODE USED?
	BNE	777$					; YES -> SKIP
	MOV	PAGE$$,PAGE$$+2				; SAVE SELECTED PAGE
777$:	MOV	PAGE$$+2,PAGE$$				; RESTORE PAGE NUMBER
	MOVB	#1,WND.FL+1				;; NORMAL EDIT MODE
	CLR	FED.MD					; RESET MODE FLAGS
	ASL	R0					;; SET FLAGS (C IF NEEDED)
	.POP	<R5,R4,R3,R2,R1,R0>			;; RESTORE
	RETURN						; HOME
;
;

;
;
.NLIST	BIN
.ENABL	LC
900$:	.ASCII	/COMMANDS ARE - <CR>==EXIT First Top Edit LOcate CLEAR ABort/
	.ASCII	/ Refresh Page/<200>
;
901$:	.ASCII	/"FEDMNG" - Command Mode! (Pages=/
9011$:	.ASCII	/XX Page=/
9012$:	.ASCII	/XX)/<200>
902$:	.ASCII	<7>/SORRY - Command not recognized?/<200>
.DSABL	LC
.LIST	BIN
.EVEN

;
;	THIS IS THE COMMAND DISPATCH TABLE. IF THE INPUT COMMAND
;	MATCHES THE KEYWORD WE ARE SUPPOSED TO DISPATCH TO INDICATED
;	ROUTINE. ROUTINES RETURN WITH 'Z' SET IFF WE ARE TO EXIT
;	FROM COMMAND MODE.
;
	CMDTAB	1000$					; DECLARE TABLE
	CMDENT	<AB-ORT>,.ABORT				; ABORT
	CMDENT	<QU-IT>,.QUIT				; QUIT
	CMDENT	<EX-IT>,.QUIT				; QUIT
	CMDENT	<F-IRST>,.FIRST				; FIRST FIELD
	CMDENT	<T-OP>,.TOP				; TOP OF FORM
	CMDENT	<E-DIT>,.EDIT				; EDIT
	CMDENT	<EN-TER>,.EDIT				; ENTER DATA
	CMDENT	<CLEAR->,.CLEAR				; CLEAR
	CMDENT	<R-EFRESH>,.REFRE			; REFRESH
	CMDENT	<LO-CATE>,.LOCATE			; LOCATE
	CMDENT	<P-AGE>,.PAGE				; PAGE NUMBER
	CMDEND						; END OF TABLE
;
;
;
;

;
;
;
.SBTTL	COMMAND	- '.ABORT'	...	ABORT EDITTING OF RECORD
;
;
;
;	WE ARE TO ABORT EDITTING. SET THE 'Z' FLAG SO THAT THE COMMAND
;	DISPATCHER WILL PASS CONTROL BACK TO THE CALLER, AND SET THE
;	'C' FLAG TO SIGNAL THAT THE ABORT OPTION WAS USED TO EXIT.
;
.ABORT:	SEZ						; SET END
	SEC						; SET ABORT
	RETURN						; HOME
;
;

;
;
;
.SBTTL	COMMAND	- '.QUIT'	...	QUIT EDIITING (SAVE RECORD)
;
;
;
;	THE OPERATOR HAS FINISHED EDITTING THE RECORD AND NOW WISHS TO
;	TERMINATE THE OPERATION. SET THE 'Z' FLAG SO THAT THE COMMAND
;	DISPATCHER WILL PASS CONTROL BACK TO THE CALLER, AND CLEAR THE
;	'C' FLAG TO INDICATE THAT THE 'QUIT' (RATHER THAN THE 'ABORT')
;	EXIT WAS USED.
;
;
.QUIT:	SEZ						; SET END
	CLC						; CLEAR ABORT
	RETURN						; HOME
;
;
;

;
;
;
.SBTTL	COMMAND	- '.FIRST'	...	ACTIVATE FIRST FIELD FOR EDITTING
;
;
;
;	THE OPERATOR WISHES TO ACTIVATE THE FIRST FIELD FOR EDITTING.
;	FIRSTLY SET UP 'ENTDAT' VARIABLE SO THAT FIRST FIELD IN RDT
;	IS ACTIVATED.
;
;
.FIRST:	MOV	2(R0),ENTDAT				; POINT FIRST FIELD
	MOV	4(R0),ENTDAT+2				; RESET COUNTER
;
;
;
;	NOW ENSURE THAT THE FIRST FIELD IS EITHER OF PAGE TYPE 0 OR THAT
;	IT IS ON THE PAGE CURRENTLY SELECTED. IF THIS IS NOT THE CASE
;	WE MUST THEN SWITCH PAGES AND REFRESH THE SCREEN.
;
;
	MOV	2(R0),R1				;; R1 --> RDT 1
	MOV	14(R1),R1				;; GET PAGE NUMBER
	BEQ	100$					;; PAGE 0 -> SKIP
	CMP	R1,PAGE$$				;; ON THIS PAGE
	BEQ	100$					;; ON THIS PAGE -> SKIP
	MOV	R1,PAGE$$				;; SWITCH PAGES
	CALL	.REFRESH				;; REFRESH SCREEN
;
;
;
;	ENTER DATA INTO FORM STARTING AT SELECTED FIELD.
;
;
100$:	CALL	ENTER					; ENTER
	CLZ						; NOT DONE
	RETURN						; HOME
;
;
;

;
;
;
.SBTTL	COMMAND	- '.TOP'	...	ACTIVATE FIELD AT TOP OF FORM
;
;
;
;	THE OPERATOR WISHES TO START EDITTING THE FIELD AT THE TOP
;	OF THE CURRENTLY SELECTED PAGE. ACTIVATE THE TOPMOST FIELD
;	ON THE CURRENTLY SELECTED PAGE AND START ENTERING DATA.
;
;
.TOP:	CALL	MOVTOP					;; TOP OF FORM
	CALL	ENTER					;; ENTER DATA
	CLZ						;; NO EXIT
	RETURN						;; HOME
;
;
;
;

;
;
;
.SBTTL	COMMAND	- '.EDIT'	...	RESUME EDITTING
;
;
;
;	START ENTERING DATA AT CURRENTLY ACTIVATED FIELD. EFFECTIVELY
;	RESUMES EDITTING WHERE PREVIOUSLY STOPPED.
;
;
;
.EDIT:	CALL	ENTER					; ENTER DATA
	CLZ						; NOT DONE
	RETURN						; HOME
;
;
;

;
;
;
.SBTTL	COMMAND	- '.LOCAT'	...	LOCATE AND ACTIVATE FIELD
;
;
;
;	LOCATE FDB FOR SPECIFIED FIELD. IF CANNOT LOCATE FIELD THEN EXIT.
;
;
.LOCAT:	CALL	LOCATE					; LOCATE DATA
	BCS	100$					; FAIL -> 100$
;
;
;	ACTIVATE THE FIELD, ENSURING THAT A PAGE CHANGE IS DONE IF THE
;	FIELD IS NOT ON THE CURRENTLY ACTIVATED PAGE.
;
;
	MOV	ENTDAT,R1				;; R1 -> RDT FOR FIELD
	MOV	14(R1),R1				;; GET PAGE NUMBER
	BEQ	60$					;; PAGE 0 FLD -> OK
	CMP	R1,PAGE$$				;; ON THIS PAGE?
	BEQ	60$					;; YES -> SKIP
	MOV	R1,PAGE$$				;; CHANGE PAGES
	CALL	.REFRE					;; REFRESH PAGE
;
;
;
;	ENTER DATA STARTING AT SELECTED FIELD. WHEN EXIT ENSURE 'Z' FLAG
;	IS NOT SET OR THE COMMAND DISPATCHER WILL TERMINATE OPERATION.
;
;
60$:	CALL	ENTER					; ENTER DATA
100$:	CLZ						; NOT DONE
	RETURN						; HOME
;
;
;

;
;
;
.SBTTL	COMMAND	- '.CLEAR'	...	CLEAR/ZERO ALL DATA IN RECORD
;
;
;
;	CLEAR OUT ALL DATA IN RECORD AND REPAINT FOREGROUND.
;
;
.CLEAR:	CALL	FRM.ZE					; ZERO DATA
	CALL	FRM.FG					; REDISPLAY
	CLZ						; NOT DONE
	RETURN						; HOME
;
;

;
;
;
.SBTTL	COMMAND	- '.REFRE'	...	REFRESH SCREEN
;
;
;
;	CLEAR SCREEN AND REPAINT BACKGROUND AND FOREGROUND PARTS OF RECORD.
;
;
.REFRE:	.PUSH	R0			; SAVE
	MOV	RECPNT,R0		; R0 --> RECORD DESCRIPTION
	CALL	FRM.EF			; ERASE FORM PART OF SCREE
	CALL	FRM.BG			; DISPLAY BACKGROUND PART OF FORM
	CALL	FRM.FG			; DISPLAY FOREGROUND PART OF FORM
	.POP	R0			; RESTORE
	CLZ				; CLEAR EXIT FLAG
	RETURN				; HOME
;
;
;

;
;
;
.SBTTL	COMMAND	- '.PAGE'	...	SELECT PAGE
;
;
;
;	GET NUMBER OF SELECTED PAGE AND SEE IF A PAGE FLIP IS NEEDED.
;
.PAGE:	MOV	R2,R1					; R1 --> INPUT
	CALL	ASCNUM					; GET PAGE NUMBER
	CMP	R0,PAGE$$				; PAGE CHANGE?
	BEQ	1000$					; NO -> SKIP
;
;
;	PAGE FLIP NEEDED. UPDATE SELECTED PAGE VARAIABLE AND REFRESH
;	THE SCREEN.
;
;
	MOV	R0,PAGE$$				; SET UP PAGE NUMBER
	CALL	.REFRESH				; REFRESH PAGE
;
;
;	STARTING FROM CURRENTLY ACTIVATED FIELD, SCAN THROUGH THE RDT
;	TO SEE IF WE CAN FIND FIELD WHICH IS ON THE NEWLY SELECTED PAGE
;	OR ON PAGE 0. MAKE FIRST SUITABLE FIELD THE ACTIVE ONE.
;
;
	MOV	ENTDAT,R1				; R1 -> ACT FDB
	MOV	ENTDAT+2,R2				; R2 -> NUM FDB LEFT
	BEQ	400$					; NONE -> 400$
100$:	TST	14(R1)					; PAGE 0 FIELD?
	BEQ	700$					; YES -> OK
	CMP	14(R1),PAGE$$				; ON THIS PAGE?
	BEQ	700$					; YES -> OK
	ADD	#40,R1					; TRY NEXT ONE
	SOB	R2,100$					; LOOP
;
;
;	NO SUITABLE FIELD WAS FOUND, THEREFORE MAKE THE ONE AT THE TOP
;	OF THE SELECTED PAGE THE ACTIVE ONE.
;
;
400$:	MOV	RECPNT,R0				; R0 --> RDT
	CALL	MOVTOP					; ENSURE THAT AT TOP
	BR	1000$					; ALL DONE
;
;
;
700$:	MOV	R1,ENTDAT				; SET UP ...
	MOV	R2,ENTDAT+2				; ... ACT FIELD PARAM
1000$:	CLZ						; CLEAR 'END' FLAG
	RETURN						; HOME
;

;
;
;
.SBTTL	PRIMITIVE - 'MOVTOP'	...	ACTIVATE FIELD AT TOP OF PAGE
;
;
;	STARTING AT FIRST FIELD IN RDT, SCAN THROUGH UNTILL FIND EITHER A
;	PAGE 0 FIELD OR ONE WHICH RESIDES ON SELECTED PAGE AND ACTIVATE IT.
;
;
MOVTOP:	.PUSH	<R0,R1,R2>				;; SAVE
	MOV	2(R0),R1				;; R1 -> RDB 1
	MOV	4(R0),R2				;; R2 = COUNT OF RDBS
	BEQ	1000$					;; NONE -> SKIP
;
100$:	TST	14(R1)					;; PAGE 0 FIELD?
	BEQ	1000$					;; YES -> SKIP
	CMP	14(R1),PAGE$$				;; ON THIS PAGE
	BEQ	1000$					;; YES -> ACCEPT
	ADD	#40,R1					;; TRY NEXT FIELD
	SOB	R2,100$					;; LOOP
;
	MOV	2(R0),R1				;; NO SUITABLE FIELD ...
	MOV	4(R0),R2				;; ... SO USE FIRST
;
1000$:	MOV	R1,ENTDAT				;; TOP FIELD FDB ADDRESS
	MOV	R2,ENTDAT+2				;; NUMBER FIELDS LEFT
	MOV	4(R0),ENTDAT+4				;; TOTAL NUMBER OF FIELDS
;
	.POP	<R2,R1,R0>				;; RESTORE
	CLZ						;; NO END
	RETURN						;; ALL DONE
;
;
;

;
;
;
.SBTTL	PRIMITIVE - 'ENTER'	...	ENTER DATA IN FIELD
;
;
;	THIS ROUTINE ENTERS INPUTTING OF DATA INTO FIELDS. THE 'ENTDAT'
;	BLOCK IS USED FOR ARGUMENT PASSING. ALL REGISTERS ARE PRESERVED.
;
;
;	SAVE REGISTERS, CLEAR ERROR LINE AND DISPLAY STATUS AND PROMPT.
;
ENTER:	.PUSH	<R0,R1,R2>		; SAVE
	ERROR	#0			; NO ERRORS AS YET
	BIT	#4,FED.MD		; CHECK MODE FLAG
	BNE	5$			; CONTROLING MODULE EXTERNAL -> SKIP
	STATUS	#1000$			; STATUS MESSAGE
5$:	PROMPT	#1001$			; PROMPT MESSAGE
;
;
;	SET UP RDB IN R1 AND COUNT IN R2, CLEAR ERRORS AND DISPLAY FIELD DATA.
;
	MOV	ENTDAT,R1		; ADDRESS OF ACTIVE RDB
	MOV	ENTDAT+2,R2		; NUMBER OF FOLLOWING RECORDS
	BEQ	700$			; NONE -> ERROR
10$:	ERROR	#0			; CLEAR ERROR MESSAGE
	MOV	R1,R0			; RDB ADDRESS -> R0
	CALL	FLD.OU			; ACTIVATE FIELD
;
;	INPUT DATA INTO FIELD (LOOP TILL RECIEVE LEGAL DATA).
;
20$:	CALL	WND.IN			; UP DATE THE FIELD
	CALL	FLD.CK			; ANY ERRORS?
	BCC	24$			; NO -> 24$
	ERROR	#TYPCHK			; WRONG TYPE OF DATA
	BR	20$			; LOOP
24$:	CALL	FLD.IN			; ACCEPT DATA
;
;
;

;
;
;	IF TERMINATED WITH A 'TAB' ADVANCE TO NEXT UNPROTECTED FIELD
;	ON CURRENTLY SELECTED PAGE.

;
	CMPB	R0,#<'I-'@>		; TAB USED TO TERMINATE?
	BNE	200$			; NO -> 200$
	.PUSH	<R1,R2>			; SAVE IN CASE HAVE TROUBLE
110$:	DEC	R2			; MOVING TO NEXT FIELD
	BEQ	170$			; AT EOT --> TROUBLE
	ADD	#40,R1			; POINT NEXT ENTRY
	TST	(R1)			; PROTECTES FIELD?
	.IIF	NDF,NOPROT,BMI	110$	; ??? SKIP IF PROTECTED FIELD ???
	TST	14(R1)			;; PAGE 0 FIELD?
	BEQ	140$			;; YES -> ACCEPT
	CMP	14(R1),PAGE$$		;; THIS PAGE FIELD?
	BNE	110$			;; NO -> NO GO
140$:	ADD	#4,SP			; DISCARD OLD R1,R2
	BR	10$			; RE-EDIT
170$:	.POP	<R2,R1>			; RESTORE
	BR	700$			;; TREAT AS A <CR>
;

;
;
;
;	IF TERMINATED WITH A <BACKSPACE> MOVE BACK 1 FIELD
;
;
;
200$:	CMPB	R0,#<'H-'@>		;; BACKSPACE?
	BNE	300$			;; NO -> SKIP
	.PUSH	<R1,R2>			;; SAVE IN CASE NEED
210$:	INC	R2			;; BACKING UP A FIELD
	CMP	R2,ENTDAT+4		;; STILL WITHIN LEGAL LIMITS?
	BHI	270$			;; NO -> SKIP
	SUB	#40,R1			;; POINT PREVIOUS ENTRY
	TST	(R1)			;; PROTECTED?
	.IIF	NDF,NOPROT,BMI	210$	;; ??? SKIP IF PROTECTED FIELD ???
	TST	14(R1)			;; PAGE 0 FIELD?
	BEQ	240$			;; YES -> ACCEPT
	CMP	14(R1),PAGE$$		;; THIS PAGE FIELD?
	BNE	210$			;; NO -> NO GO
240$:	ADD	#4,SP			;; DISCARD OLD R1,R2
	BR	10$			;; EDIT TIME
270$:	.POP	<R2,R1>			;; RESTORE OLD R1,R2
	BR	700$			;; TREAT AS A <CR>
;
;
;

300$:
;
;	SAVE PARAMETERS AND QUIT. RESET MESSAGE LINES TO NULL.
;
700$:	MOV	R1,ENTDAT		; SAVE RDT ADDRESS
	MOV	R2,ENTDAT+2		; SAVE COUNT
	ERROR	#0			; NO ERRORS
	STATUS	#0			; DELETE STATUS MESSAGE
	PROMPT	#0			; CLEAR PROMPT
	.POP	<R2,R1,R0>		; RESTORE
	RETURN				; HOME
;
;
.NLIST	BIN
.ENABL	LC
1000$:	.ASCII	/"FEDMNG" - Form Entry Mode!/<200>
1001$:	.ASCII	/<CR>==>exit entry mode    <TAB>==>next field/<200>
.DSABL	LC
.LIST	BIN
.EVEN
;
;

;
.SBTTL	ROUTINE	- 'LOCATE'	...	LOCATE AN ENTRY
;
;
;	AT ENTRY THE RECORD DESCRIPTOR TABLE ADDRESS IS IN R0. WE OBTAIN A
;	FIELD NAME FROM THE CONSOLE AND LOCATE THE FIELD'S ENTRY IN THE
;	TABLE.
;
;		SAVE REGISTERS AND GET INPUT. RDT ADDRESS GOES INTO R1.
;		MOVE FIRST FDE INTO R2 AND USE R1 AS A COUNTER.
;
;
LOCATE:	.PUSH	<R0,R1,R2,R3,R4>			; SAVE
	MOV	R0,R1					; RDT ADDRESS -> R1
	MOV	R0,R4					; SAVE FOR LATER
	TSTB	(R2)					; AT EOS?
	BEQ	60$					; YES -> SKIP
;
	MOV	R2,R0					; STRING ADD -> R0
	MOV	2(R1),R2				; R2 --> START RDT
	MOV	4(R1),R3				; R3 == NUMBER FIELDS
40$:	MOV	22(R2),R1				; R1 = NAME FIELD
	CALL	1000$					; COMPARE STRINGS
	BEQ	100$					; MATCH -> 100$
	ADD	#ENTSIZ,R2				; NEXT ENTRY
	SOB	R3,40$					; LOOP
	MOV	R4,R1					; R1 --> RDT
;
;
60$:	INPUT	,PROMPT=#9000$,STATUS=#9001$,ERROR=#0	; GET FIELD NAME
	MOV	2(R1),R2				; START RDT --> R2
	MOV	4(R1),R3				; NUMBER RECORDS
;
;
;		SEE IF THIS IS THE RIGHT FIELD. IF IT IS -> 100$.
;
70$:	MOV	22(R2),R1				; NAME STRING ADD -> R1
	CALL	1000$					; COMPARE STRINGS
	BEQ	100$					; OK -> 100$
	ADD	#ENTSIZ,R2				; NEXT ENTRY
	SOB	R3,70$					; LOOP
;
;
;		NO GO - CANNOT LOCATE FIELD.
;
	ERROR	#9002$					; FAIL MESSAGE
	BR	400$					; EXIT
;
;

;		IF FIELD NOT PROTECTED ACCEPT RESULT.
;
100$:	TST	(R2)					; PROTECTED?
	.IIF	NDF,NOPROT,	BMI	200$		; PROTECTED -> 200$
	MOV	R2,ENTDAT				; SAVE ENTRY DESCRIPT
	MOV	R3,ENTDAT+2				; SAVE NUMBER FOLLOWING
	PROMPT	#0					; CLEAR PROMPT
	STATUS	#0					; CLEAR STATUS
	.POP	<R4,R3,R2,R1,R0>			; RESTORE
	CLC						; CLEAR FAIL
	RETURN						; HOME
;
;
;		EXIT CODE =400$. PROTECTED FIELD -> 200$.
;
200$:	ERROR	#9003$					; FAILED
400$:	PROMPT	#0					; CLEAR PROMPT
	STATUS	#0					; CLEAR STATUS
	.POP	<R4,R3,R2,R1,R0>			; RESTORE
	SEC						; SET FAIL
	RETURN						; HOME


;
;
;	THIS ROUTINE IS CALLED TO SEE IF STRING (R0) = STRING (R1). THE
;	STRINGS ARE TERMINATED WITH A NULL AND 200(OCTAL) BYTE RESPECTIVELY.
;
;		WE MUST PRESERVE R0 FOR LATER USAGE. SKIP LEADING SPACES.
;
1000$:	.PUSH	R0					; SAVE
1010$:	CMPB	(R0)+,#40				; SPACE?
	BEQ	1010$					; DISCARD
	TSTB	-(R0)					; HIT NULL?
	BEQ	1100$					; YES -> FAIL
1020$:	CMPB	(R0)+,(R1)+				; MATCH?
	BEQ	1020$					; YES -> LOOP
	TSTB	-(R0)					; AT EOS?
	BNE	1100$					; NO -> FAIL
	.POP	R0					; RESTORE
	SEZ						; SET =
	RETURN						; HOME
;
1100$:	.POP	R0					; RESTORE
	CLZ						; CLEAR =
	RETURN						; HOME
;
;
;
.NLIST	BIN
.ENABL	LC
9000$:	.ASCII	/Enter the NAME of the FIELD to be located./<200>
9001$:	.ASCII	/"FEDMNG" - Field Locate Mode!/<200>
9002$:	.ASCII	<7>/ERROR - Cannot locate specified field?/<200>
9003$:	.ASCII	<7>/SORRY - That field is PROTECTED!/<200>
.DSABL	LC
.LIST	BIN
.EVEN

	.WORD	1			;; PAGE NUMBER

ENTDAT:	.WORD	0			; ADDRESS OF RDB FOR EDIT FIELD
	.WORD	0			; NUMBER OF FIELDS LEFT
	.WORD	0			;; NUMBER OF FIELDS IN RDT
	.WORD	0			;; SPARE
RECPNT:	.WORD	0			; RECORD DESCRIPTOR
RECADD:	.WORD	0			; RECORD BUFFER
