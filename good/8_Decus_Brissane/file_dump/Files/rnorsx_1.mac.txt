	.MCALL	GCMLB$,FDBDF$,FDAT$A,FDOP$A,FDRC$A,NMBLK$,FSRSZ$
	.MCALL	CSI$SW,CSI$ND,CSI$SV

 

;	LUN	FOR
;
;	1	COMMAND INPUT
;	2	??
;	3	TERMINAL I/O (ERROR MESSAGES, WAIT REQUEST)
;	4	DATA INPUT
;	5	DATA OUTPUT
;

	FSRSZ$	4,<512.*4+80.>							;SIG01-2
TTBLK::	FDBDF$									;**-11
	FDAT$A	R.VAR,FD.CR
	FDOP$A	3,,TTFIL	;LUN=3
	FDRC$A	,TTBUF,TTLIN
TTFIL:	NMBLK$	CMO,CMD
TTIBLK::GCMLB$	2,RNO,TTIBUF	;LUN=1
TTLDMY::FDBDF$
	FDAT$A	R.VAR
	FDOP$A	4,,TTLFIL	;LUN=4
TTLFIL:	NMBLK$	RUNOFF,RNO,,SY,0
LSTBLK::FDBDF$
	.IF	DF	RSTS							;SIG02
	FDAT$A	R.STRM								;SIG02
	.IFF									;SIG02
	FDAT$A	R.VAR								;SIG02
	.ENDC									;SIG02
	FDOP$A	5,,LSTFIL							;SIG02
LSTFIL:	NMBLK$	RUNOFF,DOC,,SY,0						;**-11
 
	.PSECT	SWTCH,GBL
SWTBL::	CSI$SW	SP,SPLSW,$SWTCH,SET,NEG	;SPOOLING SWITCH
	CSI$SW	UC,UPCSW,$SWTCH,SET	;FORCE UPPER CASE
	CSI$SW	HY,HYPSW,$SWTCH,SET,NEG	;HYPHENATION SWITCH
	CSI$SW	UL,,,,,ULCVAL		;UNDERLINE CONTROL
	CSI$SW	FF,FFDSW,$SWTCH,SET,NEG,FFVAL ;KICK PAGES WITH FORMFEEDS
	CSI$SW	WA,PAUSW,$SWTCH,SET	;WAIT BETWEEN PAGES
	CSI$SW	PA,,,,,PAGUAL		;LOWER/UPPER LIMIT TO PAGES
	CSI$SW	PS,,,,,PSVAL		;PAGE SIZE
	CSI$ND

PAGUAL:	CSI$SV	DECIMAL,LOWPAG,2
	CSI$SV	DECIMAL,HGHPAG,2
	CSI$ND
ULCVAL:	CSI$SV	ASCII,ULSWT,2
	CSI$ND
FFVAL:	CSI$SV	DECIMAL,LPPG,2		;HARDWARE FORM SIZE
	CSI$ND
PSVAL:	CSI$SV	DECIMAL,PNLPG,2		;PAGE SIZE
	CSI$SV	DECIMAL,PRMRG,2		;RIGHT MARGIN
	CSI$ND

ULSWT::.BLKB	2	;BUFFER FOR UNDERLINE SWITCH
	.PSECT

SPSAV::	.BLKW	1
$SWTCH::.BLKW	1	;SWITCH WORD
 
	.END
