; THE FUNCTION OF THIS MODULE IS TO DISPATCH CONSOL I/O REQUESTS TO THE
; CORRECT CONSOL DRIVER ROUTINE. AT START UP, THE ROUTINE CHECKS LOCATION
; 540 (OCTAL) IN THE CHAIN DATA PASSING AREA. IF THIS LOCATION CONTAINS
; THE RADIX 50 WORD 'CRT', LOCATION 542 IS ASSUMED TO HOLD THE CONSOL
; DEVICE TYPE NAME IN RAD50. IF LOCATION '540' IS NOT SET UP, THE
; ROUTINE REQUESTS THE OPERATOR TO SELECT A CRT TYPE. THE ENTRY POINTS
; IN THE MODULE ARE -
;
;
; CON.CI	...	INPUTS A CHARACTER (= .TTYIN)
; CON.CO	...	OUTPUTS A CHARACTER (= .TTYOUT)
; CON.LI	...	INPUTS A LINE (= .GTLINE R0)
; CON.LO	...	OUTPUTS A LINE (= .PRINT R0)
; CON.ES	...	ERASES ALL SCREEN
; CON.EL	...	ERASES SPECIFIED LINE
; CON.XY	...	POSITIONS CUSOR AT X=LOB Y=HOB OF R0.
;
;
; DEVICE DRIVER MODULES MUST CONTAIN ENTRY POINTS TO CARRY OUT THESE
; FUNCTIONS; THESE ENTRY POINTS MUST START WITH 'XXX.' INSTEAD OF 'CON.',
; WHERE 'XXX' IS THE DEVICE NAME.
;
;
.SBTTL	MODIFICATIONS

.SBTTL	MACRO	- 'CMDTAB'	...	CREATE A COMMAND TABLE
;
;
;
;	THIS MACRO IS USED TO CREATE A COMMAND TABLE WHICH CAN BE USED
;	BY 'TBLKUP' FOR COMMAND DECODING. TO USE IT DO THE FOLLOWING-
;
;		CMDTAB	NAME		; CREATE AND LABLE TABLE
;		CMDENT	COMMAND,ROUTINE	; COMMAND --> ROUTINE
;		.......
;		CMDEND			; TERMINATE TABLE
;
;	'TBLKUP' IS CALLED WITH THE COMMAND STRING IN 'R0' AND THE TABLE
;	ADDRESS (NAME) IN 'R1'. IT RETURNS WITH THE ROUTINE ADDRESS IN
;	2(R1) IF IT FINDS THE COMMAND IN THE TABLE.
;
;
;
;
.MACRO	CMDTAB	TABLE
TABLE:	.MACRO	CMDENT	NAME,VAL1,VAL2=0,VAL3=0
		.SAVE
		.PSECT	MESSAGE
		ZZZZZ1	= .
		.ASCIZ	/NAME/
		.RESTORE
		.WORD	ZZZZZ1,VAL1,VAL2,VAL3
	.ENDM	CMDENT
;
	.MACRO	CMDEND
		.WORD	0,0,0,0
		.MACRO	CMDTAB
		.ENDM	CMDTAB
		.MACRO	CMDENT
		.ENDM	CMDENT
	.ENDM	CMDEND
;
.ENDM	CMDTAB
;

;
;
;
.SBTTL	ROUTINE	- 'CON.ST'	...	SET UP I/O VECTORS
;
;
;
CON.ST:	CMP	@#540,#^RCRT		; IS CRT TYPE SET UP?
	BEQ	1000$			; YES -> 1000$
	.PRINT	#LOTSLF			; CLEAR SCREEN WITH <LF>
100$:	.GTLIN	#BUF,#PRMT		; REQUEST THAT CRT TYPE BE ENTERED
	MOV	#BUF,R0			; GET BUFFER READY
	MOV	#TABLE,R1		; COMMAND TAB ADD -> R1
	CALL	TBLKUP			; LOOKUP
	BCC	200$			; OK -> 200$
	.PRINT	#ERRMES			; TELL FAILED
	BR	100$			; LOOP
;
;
200$:	CALL	@2(R1)			; CALL ROUTINE
	MOV	#^RCRT,@#540		; SET UP ...
	MOV	R0,@#542		; ... CRT TYPE
;
;
1000$:	MOV	@#542,R0		; CRT TIME -> R0
	MOV	#DVCTAB,R1		; R1 --> TABLE
1100$:	TST	(R1)			; AT EOT?
	BEQ	7000$			; YES -> EXIT
	CMP	R0,(R1)			; FOUND ENTRY
	BEQ	2000$			; YES -> SKIP
	ADD	#20,R1			; POINT NEXT ENTRY
	BR	1100$			; LOOP
;
2000$:	MOV	#DSPTAB,R0		; R0 --> DSPTAB
	MOV	#10,R2			; R2 = COUNTER
2100$:	MOV	(R1)+,(R0)+		; COP
	SOB	R2,2100$		; LOOP
;
	CALL	CON.ES			; CLEAR SCREEN
	RETURN				; EXIT
;
;
7000$:	.PRINT	#ABRMES			; TELL QUITING
	.EXIT				; BYE
;
;
;
;

;
;
;
.SBTTL	TABLE - 'TABLE'	...	LIST OF VALID CRT TYPES
;
;
;
	CMDTAB	TABLE			; OPEN TABLE
	CMDENT	<VT5-2>,.VT52		; VT52
	CMDENT	<VT1-00>,.VT100		; VT100 
	CMDENT	<I-NTECOLOR>,.ISC	; INTECOLOR
	CMDENT	<T-VI912>,.TVI		; TVI-912
	CMDENT				; END TABLE
;
;
.VT52:	CALL	V52.ES			; ERASE SCREEN
	MOV	#^RV52,R0		; RETURN ID
	RETURN
;
;
.ISC:	CALL	ISC.ES			; ERASE SCREEN
	MOV	#^RISC,R0		; RETURN ID
	RETURN
;
.TVI:	.GTLIN	#BUF,#TVIPMT		;; ASK FOR DELAY
	MOV	#BUF,R1			;; R1 -> ANSWER
	CALL	ASCNUM			;; GET ANSWER
	MOV	R0,@#544		;; SAVE DELAY
	BNE	10$			;; NON ZERO -> 10$
	INC	@#544			;; MAKE NON-ZERO
10$:	CALL	TVI.ES			; ERASE SCREEN
	MOV	#^RTVI,R0		; RETURN ID
	RETURN				; EXIT
;
;
.VT100:	CALL	V10.ES			; ERASE SCREEN
	MOV	#^RV10,R0		; RETURN OUR NAME
	RETURN				; ALL DONE

.SBTTL	ENTRIES	- 'CON.XX'	...	DISPATCH VIA VECTORS
;
;
.MACRO	ENTRY	NAME
	.GLOBL	CON.'NAME
CON.'NAME:
	JMP	@TAB.'NAME
.ENDM	ENTRY
;
;
	ENTRY	CI	; INPUT ONE CHRACTER FROM THE CONSOLE
	ENTRY	CO	; OUTPUT ONE CHRACTER TO THE CONSOLE
	ENTRY	LO	; OUTPUT A LINE TO THE CONSOLE
	ENTRY	ES	; ERASE THE CONSOLE SCREEN
	ENTRY	EL	; ERASE A LINE ON THE CONSOLE SCREEN
	ENTRY	XY	; POSITION THE CUSOR ON THE SCREEN
;
;
;
CON.LI::CALL	@TAB.LI		; GET A LINE
	MOV	R0,-(SP)	; SAVE ADDRESS
10$:	TSTB	(R0)		; AT EOL?
	BEQ	200$		; YES -> SKIP
	CMPB	(R0)+,#140	; LC?
	BLO	10$		; NO -> SKIP
	BICB	#40,-1(R0)	; CHANGE TO UC
	BR	10$		; LOOP
;
200$:	MOV	(SP)+,R0	; RESTORE
	RETURN			; HOME
;

.SBTTL	FUNCTION DISPATCH TABLE
;
;
;	THE FOLLOWING TABLE IS USED TO PASS CONTROL TO THE SPECIFIED
;	CONSOLE DEVICE DRIVER ROUTINES. THE ENTRIES IN THIS TABLE ARE
;	PSEUDO-VECTORS.
;
;
;
;
;
.MACRO	ENTRY	NAME
	.GLOBL	V52.'NAME
TAB.'NAME:
	.WORD	V52.'NAME
.ENDM	ENTRY
;
;
DSPTAB:	.RAD50	/V52/	; DEFAULT DEVICE IS A VT52
	ENTRY	CI	; INPUT ONE CHRACTER FROM THE CONSOLE
	ENTRY	CO	; OUTPUT ONE CHRACTER TO THE CONSOLE
	ENTRY	LI	; INPUT A LINE FROM THE CONSOLE
	ENTRY	LO	; OUTPUT A LINE TO THE CONSOLE
	ENTRY	ES	; ERASE THE CONSOLE SCREEN
	ENTRY	EL	; ERASE A LINE ON THE CONSOLE SCREEN
	ENTRY	XY	; POSITION THE CUSOR ON THE SCREEN
;
;
;

;
;
;
.SBTTL	TABLE	- DVCTAB	...	LEGAL DEVICE DRIVERS
;
;
;	THIS TABLE CONTAINS A LIST OF ENTRY POINTS FOR EACH LEGAL
;	DEVICE DRIVER.
;
;
.MACRO	ENTRY	NAME
NAME:	.RAD50	/NAME/
	.GLOBL	NAME'.CI	; INPUT ONE CHRACTER FROM THE CONSOLE
	.WORD	NAME'.CI	; INPUT ONE CHRACTER FROM THE CONSOLE
	.GLOBL	NAME'.CO	; OUTPUT ONE CHRACTER TO THE CONSOLE
	.WORD	NAME'.CO	; OUTPUT ONE CHRACTER TO THE CONSOLE
	.GLOBL	NAME'.LI	; INPUT A LINE FROM THE CONSOLE
	.WORD	NAME'.LI	; INPUT A LINE FROM THE CONSOLE
	.GLOBL	NAME'.LO	; OUTPUT A LINE TO THE CONSOLE
	.WORD	NAME'.LO	; OUTPUT A LINE TO THE CONSOLE
	.GLOBL	NAME'.ES	; ERASE THE CONSOLE SCREEN
	.WORD	NAME'.ES	; ERASE THE CONSOLE SCREEN
	.GLOBL	NAME'.EL	; ERASE A LINE ON THE CONSOLE SCREEN
	.WORD	NAME'.EL	; ERASE A LINE ON THE CONSOLE SCREEN
	.GLOBL	NAME'.XY	; POSITION THE CUSOR ON THE SCREEN
	.WORD	NAME'.XY	; POSITION THE CUSOR ON THE SCREEN
.ENDM	ENTRY
;
;
;
DVCTAB:	ENTRY	V52		; VT52 ENTRY
	ENTRY	V10		; VT100
	ENTRY	ISC		; ISC  ENTRY
	ENTRY	TVI		; TVI-912
	.WORD	0,0
