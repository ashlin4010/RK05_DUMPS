! INDIRECT COMMAND FILE FOR ASSEMBLING AND TASK BUILDING RUNOFF
! FOR IAS SYSTEMS
!
! FIRST ASSEMBLE THE SOURCES
MACRO RNPRE+COMND
MACRO RNPRE+CMTAB
MACRO RNPRE+ERMSG
MACRO RNPRE+FMTCM
MACRO RNPRE+HYPHEN
MACRO RNPRE+INDEX
MACRO RNPRE+PINDX
MACRO RNPRE+RNCMD
MACRO RNPRE+RNFIO
MACRO RNPRE+RNORSX
MACRO RNPRE+RUNOFF
MACRO RNPRE+START
! NOW CREATE OBJECT LIBRARY
COPY *.OBJ RNO.OBS
LIBRARIAN CREATE RNO RNO.OBS
! NOW BUILD THE TASK.
LINK/TASK:RNO/MULTIUSER/NOFLOAT/OVERLAY:RNOIAS/OPTIONS
;
;	TASK BUILD COMMAND FILE FOR RUNOFF (MAPPED)
;
TASK=$$$RNO
PAR=GEN
ASG=TI:1:2:3:4
STACK=64
EXTTSK=2048
;
; DEFAULT HARDWARE LINES PER PAGE (OCTAL) (/PS:N)
;
GBLPAT=RNO:HWLPP:102
;
; SET DEFAULT SWITCHES
;
; /FF	1	FORM FEEDS (1=YES;  0=SIMULATE WITH LINE FEEDS)
; /HY	2	HYPHENATE OUTPUT (1=YES)
; /WA	4	WAIT BETWEEN PAGES (1=YES)
; /SP	10	SPOOL OUTPUT FILE (1=YES)
; /UC	20	FORCE ALL UPPER CASE (1=YES)
;
GBLPAT=RNO:INI$SW:3
;
; SET DEFAULT UNDERLINE MODE
;	102 =	/UL:B
;	114 =	/UL:L
;	116 =	/UL:N
;	123 =	/UL:S
GBLPAT=RNO:IUL:102
/
