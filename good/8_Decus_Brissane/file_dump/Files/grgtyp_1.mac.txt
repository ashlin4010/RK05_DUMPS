	.TITLE	TYP	GENERAL TYPE/DUMP

	.ENABL	LC
	.NLIST	BEX

	.IDENT	/GG81B /

; General type/dump utility program
; Author:  G. Grinton

	.MCALL	PRINT,EXIT$S
	.MCALL	QIOW$,QIOW$C,DIR$
	.MCALL	GCML$,GCMLB$,CSI$,CSI$1,CSI$2
	.MCALL	CSI$SW,CSI$ND

	.MCALL	FCSMC$
	FCSMC$

	FSRSZ$	2

; COMMAND LINE STUFF

; LUN ASSIGNMENT FOR GCML
; **** NOTE THAT THIS MUST BE ASSIGNED AT TKB TIME ****
; **** BY USE OF THE OPTION:
;		ASG=TI:4

GLUN = 4

GCML:	GCMLB$	2,TYP,,GLUN	;BUFFER FOR GCML

	CSI$
	.EVEN
CSIBLK:	.BLKB	C.SIZE		;CSI BUFFER

; SWITCH TABLE STUFF

OBIT = 1
ABIT = 2
RBIT = 4
LBIT = 10

SWTBL:	CSI$SW	O,OBIT,,SET,,,LONG
	CSI$SW	A,ABIT,,SET,NEG,,LONG
	CSI$SW	R,RBIT,,SET,,,LONG
	CSI$SW	L,LBIT,,SET,,,LONG
	CSI$ND

; OUTPUT STUFF

; VERSION ID

VID:	.ASCII	/TYP> V01.1/
VIDL=.-VID
	.EVEN

; A BLANK LINE FOR FIDDLING WITH

BLANK:	.ASCII	/ /
	.EVEN

; QIO FOR WRITING A LINE OF THE FILE

OP:	QIOW$	IO.WVB,5,24.,,,,<OBUFF,,40>

OLONG = 512.
OBUFF:	.BLKB	OLONG
	.EVEN

; FDB FOR THE OUTPUT FILE

IFL:	FDBDF$

	FDOP$A	1,CSIBLK+C.DSDS,IDFNM,FO.RD

IDFNM:	NMBLK$	,MAC,,SY,0

; ISB FOR THE INPUT FILE

IOSB:	.BLKW	2


; BLOCK NUMBER FOR DUMPS

BKNUM:	.BLKW
BKMSG:	.ASCIZ	/%FBlock number %P%N/
	.EVEN
	.PAGE
TYP::

; ATTACH THE TERMINAL

	QIOW$C	IO.ATT,5

; CANCEL CTRL/O BY WRITING A BLANK LINE

	QIOW$C	IO.CCO,5,24.,,,,<BLANK,1,40>


	.ENABL	LSB

; GET COMMAND LINE AND PARSE IT

	GCML$	#GCML
	BCC	10$		;OK

	CMPB	GCML+G.ERR,#GE.EOF	;EOF?
	BNE	11$
	EXIT$S
11$:	PRINT	^/TYP> Command line error/
	BR	TYP		;RESTART

10$:	CSI$1	#CSIBLK,GCML+G.CMLD+2,GCML+G.CMLD
	BCS	11$

NXTFIL:
	CLR	CSIBLK+C.MKW1
	CLR	CSIBLK+C.MKW2

	CSI$2	#CSIBLK,OUTPUT,#SWTBL	;GET FILE SPEC AND SWITCHES
	BCC	30$

25$:	PRINT	^/TYP> Invalid file-spec or switch/
	BR	TYP		;REGO

30$:	TST	CSIBLK+C.MKW1		;SWITCHES?
	BEQ	32$		;NO - ASSUME /L

	BIT	#LBIT,CSIBLK+C.MKW1	;/L?
	BEQ	35$		;NO - OTHERS

	BIT	#^C<LBIT>,CSIBLK+C.MKW1	;/L, SO NO OTHERS
	BNE	25$

; HERE FOR NORMAL RECORD I/O
; FORCE /L ON

32$:	MOV	#LBIT,CSIBLK+C.MKW1

	FDRC$R	#IFL,#0,#OBUFF,#OLONG	;INIT FDB FOR RECORD I/O

	BR	OPENIT

; HERE IF NOT /L, BUT SOME OTHERS

35$:	BIT	#OBIT!ABIT,CSIBLK+C.MKW1	;MUST BE /O AND-OR /A
	BEQ	25$		;/R ALONE IS BAD AT PRESENT

	BIT	#ABIT,CSIBLK+C.MKW1	;/A?
	BNE	40$		;NO - GO ON
	BIT	#ABIT,CSIBLK+C.MKW2	;YES - /-A?
	BNE	40$		;NO - GO ON
	BIT	#OBIT,CSIBLK+C.MKW1	;YES - ONLY LEGAL IF /O TOO
	BEQ	25$

; HERE IF /O OR /A SWITCHES OK
; BLOCK I/O

40$:	FDRC$R	#IFL,#FD.RWM	;INIT FDB FOR RECORD I/O
	FDBK$R	#IFL,#OBUFF,#OLONG,,#1,#IOSB
	.PAGE
; NOW READY TO OPEN THE FILE

OPENIT:	OPEN$R	#IFL		;OPEN
	BCC	OPENOK		;OK

	TSTB	F.ERR+1(R0)
	BMI	50$		;DSW ERROR
	CMPB	F.ERR(R0),#IE.NSF	;NON-EXISTENT FILE?
	BNE	50$		;NO
	QIOW$C	IO.WVB,5,24.,,,,<VID,VIDL,40>,,ERROR ;TELL VERSION NO.
	JMP	TYP

50$:	CALL	FCSERR		;OTHER ERROR


; OPEN WAS OK

OPENOK:	BIT	#LBIT,CSIBLK+C.MKW1	;/L MODE?
	BEQ	BLOCKS		;NO - GO TO DUMP MODE

	.DSABL	LSB
	.ENABL	LSB

TLOOP:	GET$	#IFL		;GET A RECORD
	BCC	1$		;OK?

	JMP	EOFCHK		;CHECK EOF OR ERROR

1$:	MOV	IFL+F.NRBD,OP+Q.IOPL+2	;SAVE COUNT
	BNE	3$		;NON-ZERO IS OK

	MOV	#40,OBUFF	;PRINT A BLANK LINE FOR NULL COUNT
	MOV	#1,OP+Q.IOPL+2

3$:	DIR$	#OP,ERROR

	BR	TLOOP
	.PAGE
	.DSABL	LSB
	.ENABL	LSB

; BLOCK I/O MODE

BLOCKS:	CLR	BKNUM		;START AT BLOCK 0

BLOOP:	READ$	#IFL		;READ A BLOCK
	BCC	1$
	JMP	EOFCHK		;EOF OR ERROR

1$:	WAIT$	#IFL,,,FCSERR	;WAIT TILL WE GOT IT

	MOV	#BKNUM,R2
	MOV	#BKMSG,R1
	CALL	OUTPUT		;HEADER

	MOV	#OBUFF,R0	;SET TO DUMP
	MOV	IOSB+2,R1	;LENGTH
	CALL	DMPBLK		;DO IT

	INC	BKNUM		;NEXT

	BR	BLOOP		;LOOP TILL EOF




	.DSABL	LSB
	.ENABL	LSB

; EOF CHECK ROUTINE

EOFCHK:	TSTB	F.ERR+1(R0)
	BMI	4$

	CMPB	F.ERR(R0),#IE.EOF	;EOF?
	BEQ	1$

4$:	CALL	FCSERR		;NO - BAD ERROR

1$:	CLOSE$	#IFL,FCSERR	;YES - CLOSE FILE
	PRINT	^/TYP> Eof/	;TELL HIM

	BITB	#CS.MOR,CSIBLK+C.STAT	;MORE FILES?
	BEQ	2$		;NO
	JMP	NXTFIL		;YES

2$:	JMP	TYP		;RESTART
	.PAGE
	.DSABL	LSB

; DUMP ROUTINE

; DESTROYS ALL REGS

DMPBLK:	MOV	R1,-(SP)	;SIZE
	MOV	R0,-(SP)	;ADDR

	CLR	SWRD		;LINE ADDRESS

	BIT	#OBIT,CSIBLK+C.MKW1	;/O?
	BEQ	DASCI		;NO - DUMP ASCII

OLOOP:	MOV	#10,R2		;10 WORDS
	MOV	#LBUF,R4
	MOV	@SP,R5

1$:	MOV	(R5)+,(R4)+
	SOB	R2,1$

	MOV	#20,R2		;20 BYTES
	MOV	@SP,R5
	MOV	#ASCBUF,R4
	CALL	CONVT

	MOV	#SWRD,R2
	MOV	#DPMSG,R1	;DEFAULT LONG MESSAGE

	BIT	#ABIT,CSIBLK+C.MKW1	;/A SPECIFIED?
	BEQ	2$		;NO - IT IS DEFAULT ANYWAY
	BIT	#ABIT,CSIBLK+C.MKW2	;YES - /-A?
	BNE	2$		;NO
	MOV	#SHMSG,R1	;YES

2$:	CALL	OUTPUT

	BIT	#RBIT,CSIBLK+C.MKW1	;/R?
	BEQ	3$		;NO

	MOV	#LBUF,R2
	MOV	#R5MSG,R1
	CALL	OUTPUT

3$:	ADD	#20,SWRD	;PREPARE FOR NEXT LINE
	ADD	#20,@SP
	SUB	#20,2(SP)	;DONE?
	BGT	OLOOP		;NO

	BR	DMPDUN		;YES


DASCI:	MOV	#100,R2		;DUMP 100 BYTES PER LINE
	MOV	#ASCBUF,R4
	MOV	@SP,R5
	CALL	CONVT

	MOV	#ASMSG,R1
	MOV	SWRD,ASWRD
	MOV	#ASWRD,R2
	CALL	OUTPUT		;WRITE IT

	ADD	#100,SWRD	;UPDATE
	ADD	#100,@SP
	SUB	#100,2(SP)	;DONE
	BGT	DASCI		;NO

DMPDUN:	ADD	#4,SP		;CLEAN UP

	RETURN



; CONVERT ASCII TO PRINTABLES.
; REPLACE NON-PRINTING CHARS BY '.'

CONVT:	MOVB	(R5)+,R0
	BIC	#-200,R0
	CMPB	R0,#40
	BLT	3$
	CMPB	R0,#177
	BNE	4$

3$:	MOVB	#'.,R0

4$:	MOVB	R0,(R4)+

	SOB	R2,CONVT

	RETURN



; STORAGE FOR DUMPER

DMPNUM:
SWRD:	.BLKW
LBUF:	.BLKW	10
	.WORD	ASCBUF

ASWRD:	.BLKW
	.WORD	ASCBUF

ASCBUF:	.BLKB	100

DPMSG:	.ASCIZ	/%P>/<11>/%8P/<11>/*%16E*/
SHMSG:	.ASCIZ	/%P>/<11>/%8P/
R5MSG:	.REPT	8.
	.ASCII	<11>/  /
	.ENDR
	.BYTE	0
ASMSG:	.ASCIZ	/%P> *%64E*/
	.EVEN



	.END	TYP
