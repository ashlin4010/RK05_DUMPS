	PROGRAM	SELECT
C	-------	------
C
C
C	WRITTEN BY	RAY DI MARCO
C			11-MAR-81
C
C
C	VERSION		110381/03.
C
C
C------------------------------------------------------------------------
C
C
C THIS MODULE CONTAINS THE 'SELECT' PROGRAM, WHICH WHEN LINKED WITH A
C DATABASE DEFINITION FILE, AND THE DATABASE SUPPORT LIBRARY (DBSLIB.OBJ)
C CAN BE USED TO SELECT SPECIFIC RECORDS FROM THE DATA BASE, AND PRODUCE
C A REPORT WITH THEM. ITS OPERATION IS AS FOLLOWS-
C
C
C	THE OPERATOR ENTERS THE SELECTION CRITERION
C
C	THE OPERATOR ENTERS THE NAME OF THE FIELD TO BE USED FOR SORTING
C
C	THE CONTENTS OF THE SORT FIELD (AND RECORD NUMBER) FOR SELECTED
C	RECORDS IS PLACED IN THE 'SORTER.DAT' FILE.
C
C	THE SORTER PROGRAM IS INITIATED TO SORT THE ENTRIES IN THE SORT
C	FILE.
C
C	UPON COMPLETION OF THE SORT CONTROL IS PASSED TO THE 'REPORT' PROGRAM
C	TO PRODUCE THE REPORT.
C
C
C
C MODIFICATIONS
C -------------
C
C 24-MAR-81	CHANGE 'SORT1' SO THAT NUMS/DATES PLACED IN REVERSE ORDER
C 23-APR-81	ADD IN 'RANGE' SELECTION SELECTION (FOR KP!)
C
C

C
C
C		---- DECLARATIONS ----
C
C
	LOGICAL*1	ANS,BUF(131),NAME(20)	! WORK VARIABLE
	LOGICAL*1	RLOW,RHIGH		! RANGE CHECKING VARIABLES
	INTEGER		IRLOW,IRHIGH		! RANGE CHECKING VARIABLES
	INTEGER		LOGREC(16)		! LOGGING BUFFER
	INTEGER		IKEYS(16)		! USED TO HOLD KEYS
C
C
	COMMON		/PARAM/		NUMREC	! USED BY 'IRECNX'
	COMMON		/SCRTH/	BUF		! SCRATCH AREA
	COMMON		/SORT/	NAME,IS,ITYPE	! SORT PARAMETERS
C
C

C
C
C		--- INITIALIZATION CODE ---
C
C
	WRITE	(7,40)					! IDENTIFY SELF
40	FORMAT	(//////////////////////////////////////
	1	1X,10X,'MODULE "SELECT" (RDM1103.03)'//)
	CALL	DBSFIN	()				! INIT DATA BASE
	CALL	SRTFIN	()				! INIT SORTER
	CALL	DBSFOR	(7,NUMREC)			! NUMBER OF ENTRIES
C
C
C
C
C
100	WRITE	(7,110)					! PROMPT
110	FORMAT	(1X,'ENTER SELECTION KEY (ALL RANGE) > ',$)
	READ	(5,120)	ANS				! GET KEY
120	FORMAT	(A1)
	IF	(ANS.EQ.'A')	GOTO	1000		! ALL -> 1000
	IF	(ANS.EQ.'R')	GOTO	2000		! RANGE -> 2000
	IF	(ANS.EQ.' ')	CALL IDATE (0,0,0,0,0)	! ' ' -> EXIT
	GOTO	100					! LOOP

C
C
C
C		--- WANT TO LIST ALL ACTIVE RECORDS ---
C
C
C
1000	CALL	SORT0	(ISIZE)				! SET UP SORT PARAM
	WRITE	(7,1020) NUMREC				! TELL WHATS HAPPENING
1020	FORMAT	(10X,'--- LISTING ALL ACTIVE RECORDS ---'
	1	/1X,'NUMBER OF RECORS PROCESSEDED',T40,I6)
	IRECS	= 0					! NO MATCHES AS YET
	I	= 1					! START A RECORD 1
1040	IF	(IRECNX(I).LE.0) GOTO 1400		! GET FIRST ASCTIVE REC
	IRECS	= IRECS+1				! GOT ANOTHER MATCH
	CALL	SORT1(I)				! STORE ASC SORT FIELD
	I=I+1						! UP REC REC NUM
	GOTO	1040					! LOOP
C
C
C
C		HAVE PLACED ALL ACTIVE RECORDS IN SORT FILE.
C		WE NOW DO SORTER. THIS IS DONE BY CHAINING TO
C		ANOTHER PROGRAM. ON SORT COMPLETION WE GO ARE
C		RESTARTED, AND AS THE 'ICHAIN' FLAG WILL BE
C		SET CONTROL PASSES TO THE LISTING SECTION.
C
C
1400	WRITE	(7,1410)	IRECS			! TELL NUMBER SELECTED
1410	FORMAT	(1X,'NUMBER OF RECORDS SELECTED',T40,I6)
	CALL	SRTFOR	(3,ISIZE,IRECS,'CL','REPORT.SAV')! SET UP SORT PARAM
	IF	(IRECS.EQ.0)	GOTO	1500		! NONE -> QUIT
C
	WRITE	(7,1420)				! TELL WILL SORT
1420	FORMAT	(/1X,'INITIATING SORT PHASE ..........')
	CALL	SRTFOR	(6)				! SORT TIME
C
1500	WRITE	(7,1520)				! TELL NONE TO SORT
1520	FORMAT	(/1X,'% NO RECORDS SELECTED - ABORTING OPERATION')
	GOTO	100					! LOOP
C
C

C
C
C
C		--- WANT TO LIST ALL ACTIVE RECORDS IN RANGE ---
C
C
C
2000	CALL	SORT0	(ISIZE)				! SET UP SORT PARAM
	IF	(ITYPE.GT.2)	GOTO	2100		! NOT ASCII -> SKIP
C
	WRITE	(7,2005)				! PROMPT
2005	FORMAT	(1X,'ENTER LOWER RANGE > ',$)
	READ	(5,2015)	RLOW			! GET RANGE
	WRITE	(7,2010)				! PROMPT
2010	FORMAT	(1X,'ENTER HIGH RANGE  > ',$)
	READ	(5,2015)	RHIGH
2015	FORMAT	(A1)
C
	WRITE	(7,2020) NUMREC				! TELL WHATS HAPPENING
2020	FORMAT	(10X,'--- LISTING ALL ACTIVE RECORDS IN RANGE ---'
	1	/1X,'NUMBER OF RECORS PROCESSEDED',T40,I6)
	IRECS	= 0					! NO MATCHES AS YET
	I	= 1					! START A RECORD 1
2040	IF	(IRECNX(I).LE.0) GOTO 2400		! GET FIRST ASCTIVE REC
	CALL	DBSFOR	(2,NAME,BUF,IS,ITYPE)	! GET DATA
	IF	(BUF(1).LT.RLOW)	GOTO	2060	! SKIP IF ...
	IF	(BUF(1).GT.RHIGH)	GOTO	2060	! ... NOT IN RANGE
	IRECS	= IRECS+1				! GOT ANOTHER MATCH
	CALL	SORT1(I)				! STORE ASC SORT FIELD
2060	I=I+1						! UP REC REC NUM
	GOTO	2040					! LOOP
C

C
C
2100	WRITE	(7,2105)				! PROMPT
2105	FORMAT	(1X,'ENTER LOWER RANGE > ',$)
	READ	(5,2115)	IRLOW			! GET RANGE
	WRITE	(7,2110)				! PROMPT
2110	FORMAT	(1X,'ENTER HIGH RANGE  > ',$)
	READ	(5,2115)	IRHIGH
2115	FORMAT	(I10)
C
	WRITE	(7,2121) NUMREC				! TELL WHATS HAPPENING
2121	FORMAT	(10X,'--- LISTING ALL ACTIVE RECORDS IN RANGE ---'
	1	/1X,'NUMBER OF RECORS PROCESSEDED',T40,I6)
	IRECS	= 0					! NO MATCHES AS YET
	I	= 1					! START A RECORD 1
2140	IF	(IRECNX(I).LE.0) GOTO 2400		! GET FIRST ASCTIVE REC
	CALL	DBSFOR	(2,NAME,ITEMP,IS,ITYPE)	! GET DATA
	IF	(ITEMP.LT.IRLOW)	GOTO	2160	! SKIP IF ...
	IF	(ITEMP.GT.IRHIGH)	GOTO	2160	! ... NOT IN RANGE
	IRECS	= IRECS+1				! GOT ANOTHER MATCH
	CALL	SORT1(I)				! STORE ASC SORT FIELD
2160	I=I+1						! UP REC REC NUM
	GOTO	2140					! LOOP
C

C
C
C		HAVE PLACED ALL ACTIVE RECORDS IN SORT FILE.
C		WE NOW DO SORTER. THIS IS DONE BY CHAINING TO
C		ANOTHER PROGRAM. ON SORT COMPLETION WE GO ARE
C		RESTARTED, AND AS THE 'ICHAIN' FLAG WILL BE
C		SET CONTROL PASSES TO THE LISTING SECTION.
C
C
2400	WRITE	(7,2410)	IRECS			! TELL NUMBER SELECTED
2410	FORMAT	(1X,'NUMBER OF RECORDS SELECTED',T40,I6)
	CALL	SRTFOR	(3,ISIZE,IRECS,'CL','REPORT.SAV')! SET UP SORT PARAM
	IF	(IRECS.EQ.0)	GOTO	2500		! NONE -> QUIT
C
	WRITE	(7,2420)				! TELL WILL SORT
2420	FORMAT	(/1X,'INITIATING SORT PHASE ..........')
	CALL	SRTFOR	(6)				! SORT TIME
C
2500	WRITE	(7,2520)				! TELL NONE TO SORT
2520	FORMAT	(/1X,'% NO RECORDS SELECTED - ABORTING OPERATION')
	GOTO	100					! LOOP
C
C

	END

	FUNCTION	IRECNX	(IREC)
C	--------	------
C
C	THIS FUNCTION WILL SCAN ALL ENTRIES IN THE DATA BASE STARTING AT
C	RECORD 'IREC', AND WILL RETURN THE NUMBER OF THE NEXT ACTIVE 
C	RECORD. ON EXIT 'IRECNX' IS SET TO -1 IFF NO MORE ACTIVE RECORDS
C	EXIST.
C
C		---- DECLARATIONS ----
C
C
	INTEGER		IKEYS(16)			! USED TO HOLD KEYS
	COMMON		/PARAM/		NUMREC
C
	DO	3500	I=IREC,NUMREC			! CHECK ALL RECORDS
		CALL	DBSFOR(4,I,IKEYS)		! GET KEYS
		IF	(IKEYS(1).EQ.0)	GOTO 3500	! IGNORE DEAD RECORDS
		CALL	DBSFOR (0,I)			! READ IN RECORD
		CALL	DBSFOR (2,'st',ISTS,ISTSS,ISTST)! GET STATUS
		IF	(ISTS.NE.'AC')	GOTO 3500	! IGNORE IF NOT ACTIVE
		IREC	= I				! MARK IT
		IRECNX	= I				! CLEAR FAIL
		RETURN					! EXIT
3500	CONTINUE
C
	IREC	= -1					! SET FAILED
	IRECNX	= -1					! SET FAILED
	RETURN						! EXIT
	END

	SUBROUTINE	SORT0	(ISIZE)
C	----------	-----
C
C
C	THIS ROUTINE IS CALLED PRIOR TO A CALL TO 'SORT1' OR 'SORT2'
C	TO INITIALIZE THE 'IS' VARIABLE. THE FUNCTION OF THE MODULE
C	IS AS FOLLOWS-
C
C		IT OBTAINS THE CURRENT RECORD SIZE SET IN THE SORT
C		FILE, AND WILL SET 'IS' TO TWO LESS THAN THIS IF
C		THIS IS THE SECOND PASS (IE ALREADY SORTED).
C
C		IF THIS IS PASS ONE (IE WE ARE SETTING UP FOR THE
C		SORT), IT WILL ASK THE OPERATOR TO ENTER THE NAME
C		OF THE SORT FIELD. IT WILL THEN LOOK UP THE DATA
C		BASE TO ENSURE THAT THE FIELD EXISTS. IF IT DOES
C		IT WILL SET 'IS' TO ITS SIZE, AND RETURN 'IS+2'
C		IN 'ISIZE' (THE RECORD SIZE).
C
C	THE VARIABLES USED BY THE SORT? ROUTINES ARE STORED IN THE
C	'SORT' NAMED COMMON BLOCK. THESE ARE-
C
C		NAME	THE ASCIZ NAME OF THE SORT FIELD
C		IS	THE SIZE OF THE SORT FIELD (2 LESS THAN REC SIZE)
C		IT	THE TYPE OF THE SORT FIELD
C
C
C

C
	LOGICAL*1	NAME(20),BUF(131)		! WORK VARIABLES
	INTEGER		DBSFOR				! EXTERNAL ROUTINE
C
	COMMON		/SCRATH/	BUF		! ONLY USED TEMPORILY
	COMMON		/SORT/		NAME,IS,IT	! SORT? COMMON VARS
C
C
C
50	WRITE	(7,100)					! PROMPT
100	FORMAT	(1X,'ENTER NAME OF SORT FIELD > ',$)
	READ	(5,200)	NAME				! GET SORT FIELD
200	FORMAT	(20A1)
C
	DO	300	I =20,1,-1			! DELETE TRAILING ...
300	IF	(NAME(I).NE.' ') GOTO 500		! ... BLANKS
500	NAME(I+1)	= 0				! MAKE ASCIZ
	CALL	FUDGE	(NAME)				! FUDGE NAME
	I	= DBSFOR (2,NAME,BUF,IS,IT)		! DOES FIELD EXIST?
	IF	(I.GE.0)	GOTO	1000		! YES -> OK
	WRITE	(7,510)					! TELL NO-NO
510	FORMAT	(10X,'% SORRY - ILLEGAL SORT FIELD%')
	GOTO	50					! LOOP
C
1000	ISIZE	= IS + 2				! RECORD NUM/SIZE
	RETURN						! EXIT
	END

	SUBROUTINE	FUDGE	(NAME)
C
	IF	(NAME.EQ.'CN')	NAME	= 'cn'
	IF	(NAME.EQ.'RN')	NAME	= 'rn'
	IF	(NAME.EQ.'ST')	NAME	= 'st'
	IF	(NAME.EQ.'CD')	NAME	= 'cd'
	IF	(NAME.EQ.'ED')	NAME	= 'ed'
	IF	(NAME.EQ.'DD')	NAME	= 'dd'
	RETURN
	END

	SUBROUTINE	SORT1	(IREC)
C	----------	-----
C
C	THIS ROUTINE IS CALLED TO PLACE THE CONTENTS OF THE FIELD TO
C	BE SORTED INTO THE SORT DATA FILE. THE 'IREC' VARIABLE IS
C	PASSED BY THE CALLER, AND IS THE RECORD NUMBER TO BE ASCOCIATED
C	WITH THE FIELD TO BE WRITTEN. THE ROUTINE PLACES THIS NUMBER
C	IN THE FIRST WORD OF THE RECORD, AND THEN OBTAINS THE CONTENTS
C	OF THE SORT FIELD FROM THE DATA BASE RECORD CURRENTLY IN MEMORY.
C	IT THEN WRITES OUT THIS FIELD CONTENTS TO THE SORT FILE.
C
C
C
	LOGICAL*1	BUF(131),NAME(20)
C
	COMMON		/SCRATCH/	BUF
	COMMON		/SORT/		NAME,IS,IT
C
C
	CALL	DBSFOR	(2,NAME,BUF,IS,IT)		! GET FIELD
	CALL	SRTFOR	(5,IREC,2)			! SAVE RECORD
C
	IF	(IT.GE.3)	GOTO	100		! IF NUM/DAT -> 100
	CALL	SRTFOR	(5,BUF,IS)			! SAVE DATA
	RETURN						! EXIT
C
100	CALL	SRTFOR	(5,BUF(2),1)			! HOB
	CALL	SRTFOR	(5,BUF(1),1)			! LOB
	RETURN						! ALL DONE
C
	END

	SUBROUTINE	SORT2	(IREC)
C	----------	-----
C
C
C	THIS ROUTINE IS CALLED IN PASS TWO, AFTER THE SORT HAS BEEN DONE.
C	IT READS OUT ONE SORTED RECORT AT A TIME, AND RETURNS TO THE CALLER
C	THE NUMBER ASCOCIATED WITH THAT RECORD.
C
C
	LOGICAL*1	BUF(131),NAME(20)
C
	COMMON		/SCRATCH/	BUF
	COMMON		/SORT/		NAME,IS,IT
C
	CALL	SRTFOR	(4,IREC,2)			! GET RECORD NUMBER
	CALL	SRTFOR	(4,BUF,IS)			! GET DATA
	RETURN						! EXIT
	END
