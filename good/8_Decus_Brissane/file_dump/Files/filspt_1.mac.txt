; THIS MODULE CONTAINS FILE I/O SUPPORT ROUTINES. THE ROUTINES
; AVAILABLE ARE-
;
; FIL.NM	ASSOCIATE A FILE NAME WITH A FILE DESCRIPTOR BLOCK
;
; FIL.LK	ASSOCIATE A FDB WITH AN EXISTING FILE.
;
; FIL.PG	PURGE CHANNEL AND DEALLOCATE FDB
;
; FIL.CL	CLOSE CHANNEL AND DEALLOCATE FDB
;
; FIL.RD	READ DATA FROM FILE
;
; FIL.WT	WRITE DATA TO FILE
;
; FIL.ST	DISPLAY STATUS OF FILE.
;
;
;
.SBTTL	MODIFICATION
;
; 11-FEB-81	ADD IN '.PRINT' MACRO DEFINITION SO USE 'CONSOL' I/O
;		CHANGE DEFAULT FILE EXTENSIONS.
;		ADD IN 'FIL.NW' ROUTINE
;

;
;
;
.SBTTL	DOCUMENTATION	- ARGUMENT PASSING AND REGISTER USAGE
;
;
; REGISTER R0 IS VOLITILE; ALL OTHER REGISTERS ARE PRESERVED. THE
; ADDRESS OF THE FDB MUST BE IN R5 AT ENTRY. AT EXIT 'C' FLAG IS
; SET IFF AN ERROR OCCURED.
;
; 'FIL.NM' ALSO REQUIRES THAT R4 HOLD THE ADDRESS OF AN ASCIZ STRING
; IN WHICH THE NAME OF THE FILE IS HELD. THIS STRING MUST BE IN A
; FORM RECONIZABLE BY THE RT11 CSI.
;
; 'FIL.RD' AND 'FIL.WT' ALSO USE R0 AND THE 'Z' FLAG TO INDICATE AN
; EOF ERROR. 'Z' IS SET AND R0 ZEROED ON EXIT IFF AN EOF ERROR OCCUED.
;
;
;
.SBTTL	DOCUMENTATION	- FILE DESCRIPTOR BLOCK FORMAT
;
;
; THE FDB CONSISTS ON A NUMBER OF WORDS. THE ADDRESS OF THE FDB MUST
; BE PASSED OVER IN R5 TO THE ROUTINES IN THIS MODULE. THE FDB FORMAT
; IS-
;
;	.WORD	FILSPC	; ADDRESS OF .RAD50 /DDDNNNNNNXXX/ STRING
;	.WORD	FILNME	; ADDRESS OF .ASCII /DDD:NNNNNN.XXX/<200> STRING
;	.WORD	CHANNEL	; NUMBER OF RT11 CHANNEL TO USE
;	.WORD	STATUS	; FLAGS BYTE AND ERROR BYTE
;	.WORD	BLOCK	; BLOCK NUMBER TO USE IN I/O
;	.WORD	RQSTD	; NUMBER OF WORDS REQUESTED TO TRANSFER
;	.WORD	SIZE	; FILE SIZE IN BLOCKS
;	.WORD	COPIED	; NUMBER OF WORDS ACTUALLY TRANSFERED
;	.WORD	BUFFER	; ADDRESS OF BUFFER TO BE USED.
;
;
;

;
;
;
.SBTTL	DOCUMENTATION	- FILE DESCRIPTOR BLOCK ELEMENTS
;
;
; FILSPC
;	THIS WORD CONTAINS THE ADDRESS OF THE RAD50 STRING TO BE USED
;	WHEN EXECUTING THE '.LOOKUP' TO ASSOCIATE THE FDB WITH AN
;	RT11 FILE.
;
; FILNME
;	THE ADDRESS OF THE ASCII STRING CONTAINING THE FILE NAME. THIS
;	STRING IS OUTPUT WHEN THE FILE STATUS IS DISPLAYED.
;
; NOTE-	THE FILSPC/FILNME STRING SPACE MUST BE ALLOCATED BY THE CALLER,
;	BUT THE CONTENTS OF THESE FIELDS CAN BE SET UP BY THE CALLER OR
;	BY 'FIL.NM'.
;
; CHANNEL
;	THIS WORD CONTAINS THE NUMBER OF THE RT-11 CHANNEL TO BE USED
;	FOR I/O OPERATIONS. THE CALLER MUST ALLOCATE A UNIQUE, LEGAL
;	CHANNEL NUMBER FOR EACH FDB SIMULTANEOUSLY OPEN.
;
; STATUS
;	THIS WORD IS DIVIDED INTO TWO BYTES. THE HOB CONTAINS THE
;	LAST ERROR RETURNED BY RT-11 VIA THE ERROR BYTE, AND THE LOB
;	CONSISTS OF FLAGS INDICATING THE STATE OF THE FDB.
;
; BLOCK
;	THE BLOCK NUMBER TO USE AS THE START FOR I/O OPERATIONS.
;
; RQSTD
;	NUMBER OF WORDS TO BE TRANSFERED IN I/O OPERATIONS.
;
; SIZE
;	SIZE OF FILE IN BLOCKS. SET UP BY 'FIL.LK'.
;
; COPIED
;	NUMBER OF WORDS ACTUALLY TRANSFERED (SET UP BY .RD/.WT)
;
; BUFFER
;	ADDRESS OF BUFFER TO BE USED IN I/O. MUST BE SET UP BY CALLER.
;
;

;
;
;
.SBTTL	DOCUMENTATION	- 'FILSPT' MACRO
;
;
; THIS MACRO SHOULD BE USED TO INVOKE THE ROUTINES IN THIS MODULE.
; THE CODE FOR THIS MACRO IS INCLUDED BELOW.
;
;

.MACRO	FILSPT
;
FDB.NR	= 0				; OFFSET TO RAD50 NAME
FDB.NA	= 2				; OFFSET TO ASCII NAME
FDB.CH	= 4				; OFFSET TO CHANNEL NUMBER
FDB.FL	= 6				; OFFSET TO FLAGS
FDB.BL	= 10				; OFFSET TO BLOCK NUMBER
FDB.RQ	= 12				; OFFSET TO TRANSFER SIZE (WORDS)
FDB.NB	= 14				; OFFSET TO DEVICE SIZE
FDB.CP	= 16				; WORDS TRANSFERED IN REALITY
FDB.BF	= 20				; BUFFER ADDRESS
;
;
.MACRO	STATUS	FDB
	.IF	NB,FDB
		MOV	R5,-(SP)
		MOV	FDB,R5
	.ENDC
	.GLOBL	FIL.ST
	CALL	FIL.ST
	.IIF	NB,FDB,	MOV	(SP)+,R5
.ENDM	STATUS
;
.MACRO	FDB	NAME,CHANNEL,BUFFER=0,SIZE=0,GLOBL
	.IIF	NB,GLOBL,	.GLOBL	NAME
NAME:	.WORD	10$				; RAD50 FILE NAME
	.WORD	20$				; ASCII FILE NAME
	.WORD	CHANNEL				; CHANNEL NUMBER
	.WORD	0				; STATUS
	.WORD	0				; START BLOCK
	.WORD	SIZE				; REQUEST SIZE
	.WORD	0				; FILE SIZE
	.WORD	0				; COPIED
	.WORD	BUFFER				; BUFFER ADDRESS
;
10$:	.WORD	0,0,0,0
20$:	.ASCII	/              /<200><0>
.ENDM	FDB
;
;
.MACRO	NAME	FDB,STRING,ERROR
	.IIF	NB,FDB,	MOV	FDB,R5
	.IIF	NB,STRING,	MOV	STRING,R4
	.GLOBL	FIL.NM
	CALL	FIL.NM
	.IIF	NB,ERROR,	BCS	ERROR
.ENDM	NAME
;
;
.MACRO	LOOKUP	FDB,ERROR
	.IIF	NB,FDB,	MOV	FDB,R5
	.GLOBL	FIL.LK
	CALL	FIL.LK
	.IIF	NB,ERROR,	BCS	ERROR
.ENDM	LOOKUP
;
;
.MACRO	ENTER	FDB,SIZE=#-1,ERROR
	.IIF	NB,FDB,	MOV	FDB,R5
	MOV	SIZE,FDB.BL(R5)
	.GLOBL	FIL.NW
	CALL	FIL.NW
	.IIF	NB,ERROR,	BCS	ERROR
.ENDM	ENTER
;
;
.MACRO	CLOSE	FDB
	.IIF	NB,FDB,	MOV	FDB,R5
	.GLOBL	FIL.CL
	CALL	FIL.CL
.ENDM	CLOSE
;
;
.MACRO	PURGE	FDB
	.IIF	NB,FDB,	MOV	FDB,R5
	.GLOBL	FIL.PG
	CALL	FIL.PG
.ENDM	PURGE
;
;
.MACRO	READ	FDB,BLOCK,SIZE,BUFFER,ERROR
	.IIF	NB,FDB,	MOV	FDB,R5
	.IIF	NB,BLOCK,	MOV	BLOCK,FDB.BL(R5)
	.IIF	NB,SIZE,	MOV	SIZE,FDB.RQ(R5)
	.IIF	NB,BUFFER,	MOV	BUFFER,FDB.BF(R5)
	.GLOBL	FIL.RD
	CALL	FIL.RD
	.IIF	NB,ERROR,	BCS	ERROR
.ENDM	READ
;
;
.MACRO	WRITE	FDB,BLOCK,SIZE,BUFFER,ERROR
	.IIF	NB,FDB,	MOV	FDB,R5
	.IIF	NB,BLOCK,	MOV	BLOCK,FDB.BL(R5)
	.IIF	NB,SIZE,	MOV	SIZE,FDB.RQ(R5)
	.IIF	NB,BUFFER,	MOV	BUFFER,FDB.BF(R5)
	.GLOBL	FIL.WT
	CALL	FIL.WT
	.IIF	NB,ERROR,	BCS	ERROR
.ENDM	WRITE
.ENDM	FILSPT

;
.SBTTL	ROUTINE	- 'FIL.NM'	...	SET UP FILE NAME
;
;
;
;		--- INITIAL SECTION ---
;
;
;	SAVE REGISTERS AND USE 'CSI' TO DECODE FILE STRING NAME. NOTE
;	THAT THE FILE NAME IS PASSED OVER IN STRING @R4, AND WE STORE
;	DECODED DATA IN SAME STRING.
;
;
FIL.NM:	.PUSH	<R0,R1,R2,R3,R4>	; SAVE ALL REGISTERS
	MOV	SP,R2			;; SAVE SP IN R2
;
	MOV	R4,R3			;; BUFFER ADDRESS -> R3
	INC	R3			;; ENSURE THAT ...
	BIC	#1,R3			;; ... IS EVEN
10$:	CMPB	(R4)+,#40		;; SPACE?
	BEQ	10$			;; YES -> IGNORE
	DEC	R4			;; WENT 1 TO FAR
	.CSISPC	R3,#6666,R4		;; TRY AND DECODE FILE SPEC
	BCS	700$			; FAILED -> 700$
	MOV	R2,SP			;; RESET STACK
	.POP	R4			; RESTORE
	MOV	R3,R4			;; POINT TO START FILE DATA
;
;
;
;		--- SET UP RAD50/ASCII NAME ---
;
;
;	SET UP RAD50 NAME OF FILE IN FILE DESCRIPTOR BLOCK.
;
;
	MOV	(R5),R3			; R3 --> FILE NAME IN RAD50
	.IRP	X,<30.,32.,34.,36.>	; --- 4 WORDS IN FILE SPEC ---
		MOV	X'(R4),(R3)+	; SET UP RAD50 FILE SPEC
	.ENDR				; ----------------------------
	TST	@(R5)			; WAS A FILE REALLY SPECIFIED?
	BEQ	740$			; NO -> ERROR
;
;
;

;
;
;	SET UP ASCII NAME OF FILE IN FILE DEXRIPTOR BLOCK.
;
;
	.PUSH	<R4,R5>			; SAVE ARG BLOCK/LINE BUFF ADDRESS
	MOV	#3,EMTBLK		; ... 3 ARGUMENTS ...
	MOV	#6000$,EMTBLK+2		; ... ADDRESS .WORD 12 ...
	MOV	(R5),EMTBLK+4		; ... SET UP RAD50 STRING ADDRESS ...
	MOV	R4,EMTBLK+6		; ... ADRESS OF ASCII STRING ...
	MOV	#EMTBLK,R5		; SET UP FORTRAN ARG BLOCK
	CALL	R50ASC			; GET FILE NAME IN ASCII
	.POP	<R5,R4>			; RESTORE REGISTERS
;
	MOV	R4,R0			; R0 --> SOURCE
	MOV	2(R5),R1		; R1 --> FILE NAME IN ASCII
	MOV	#3,R2			; DEVICE NAME = 3 LETTERS
	CALL	1000$			; COPY ACCROSS DEVICE NAME
	MOVB	#':,(R1)+		; DELIMIT
	MOV	#6,R2			; 6 LETTER FILE NAME
	CALL	1000$			; COPE FILE NAME PROPER
	CMPB	-1(R1),#':		; NULL FILE NAME?
	BEQ	100$			; YES -> SKIP
	MOVB	#'.,(R1)+		; DELIMIT
	MOV	#3,R2			; 3 LETTER EXTENSION
	CALL	1000$			; COPY FILE EXTENSION
100$:	MOVB	#200,(R1)		; EOS
;
	.POP	<R3,R2,R1,R0>		; RESTORE
	CLC				; CLEAR FAIL
	RETURN				; HOME
;
;

;
;
;		--- ERROR REPORTING ---
;
;
;	ILLEGAL FILE NAME - REPORT ERROR AND EXIT 
;
;
;
700$:	MOV	R2,SP			; RESET STACK
	.PRINT	#7000$			; IDENTIFY MODULE
	.PRINT	#7001$			; SHOW ERROR TYPE
	.POP	<R4,R3,R2,R1,R0>	; RESTORE
	SEC				; SET FAILED
	RETURN				; HOME
;
740$:	.PRINT	#7000$			; IDENTIFY MODULE
	.PRINT	#7002$			; SHOW ERROR TYPE
	.POP	<R3,R2,R1,R0>		; RESTORE
	SEC				; SET FAILED
	RETURN				; HOME
;
;
;

;
;
;		--- PRIMITIVE ---
;
;
;	COPY R2 NON-SPACE (IE # 40) CHARACTERS FROM (R0) TO (R1).
;
;
1000$:	CMPB	(R0)+,#40		; SPACE?
	BEQ	1100$			; YES -> IGNORE
	MOVB	-1(R0),(R1)+		; COPY
1100$:	SOB	R2,1000$		; LOOP
	RETURN				; HOME
;
;
;
;		--- MESSAGES/ARGUMENTS ---
;
;
.NLIST	BIN
6000$:	.WORD	12.			; ARG FOR 'R50ASC'
	.WORD	0,0,0			; REST ARGS FOR '.CSISPC'
6666$:	.RAD50	/MACMACMACMACMACMAC/	; DEFAULT FILE EXTENSIONS
7000$:	.ASCII	/"FILSPT" - ILLEGAL FILE NAME - /<200>
7001$:	.ASCIZ	/? FILE SPEC ERROR ?/
7002$:	.ASCIZ	/? NO FILE SPECIFIED ?/
.EVEN
.LIST	BIN
;
;

;
;
;
.SBTTL	ROUTINE	- 'FIL.NW'	...	ENTER A FILE
;
;
;		--- INITIALIZATION SECTION ---
;
;	ENSURE THAT 'FDB' IS NOT ALREADY ALLOCATED TO AN OPEN CHANNEL,
;	AND THAT DEVICE DRIVER IS LOADED INTO MEMORY. IF ANY ERROR
;	IS DETECTED ABORT VIA '1000$' WITH R4 HOLDING AN APPROPRIATE
;	ERROR MESSAGE. 
;
;
FIL.NW:	.PUSH	<R0,R4>			; SAVE REGISTERS
	MOV	#7000$,R4		; GET ERROR MESSAGE READY
	BIT	#1,O.FL(R5)		; CHANNEL OPEN?
	BNE	1000$			; YES -> ERROR TIME
	MOV	#7001$,R4		; GET MESSAGE READY
	.DSTAT	#EMTBLK,(R5)		; FIND DEVICE STATS
	BCS	1000$			; FAILED -> ERROR EXIT
	MOV	#7002$,R4		; GET MESSAGE READY
	TST	EMTBLK+4		; HANDLER LOADED
	BEQ	1000$			; NOT IN MEMORY -> ERROR EXIT
;
;
;
;		--- ENTER TIME ---
;
;
;	EXECUTE A ENTER FOR THE FILE SPECIFIED IN THE FDB. IF ENTER
;	SUCEEDS MARK FDB AS ALLOCATED TO A CHANNEL AND REWIND FILE.
;
;
	.ENTER	#EMTBLK,O.CH(R5),O.NR(R5),O.BL(R5)
	BCS	777$			; FAILED -> 777$
	MOV	#1,O.FL(R5)		; SET 'CHANNEL OPEN' FLAG
	MOV	O.BL(R5),O.MS(R5)	; SET UP FILE SIZE
	CLR	O.BL(R5)		; REWIND FILE
;
;
;
400$:	.POP	<R4,R0>			; RESTORE
	CLC				; CLEAR FAIL
	RETURN				; HOME
;
;
;

;
;
;
;		--- ERROR SECTION ---
;
;
;	IF WE GET HERE THE ENTER FAILED. PRINT OUT AN APPROPRIATE
;	ERROR MESSAGE AND RETURN WITH 'C' FLAG SET TO INDICATE
;	ERROR.
;
;
;
776$:	.PURGE	O.CH(R5)		; PURGE CHANNEL
	CLR	O.FL(R5)		; SHOW CHANNEL CLOSED
	MOV	#7005$,R4		; MESSAGE -> R4
	BR	1000$			; ERROR HANDLER TIME
777$:	MOV	#7003$,R4		; GET DEFAULT MESSAGE READY
	TSTB	@#RTERRB		; WAS IT MESSAGE 0
	BEQ	1000$			; YES -> GO PROCESS IT
	MOV	#7004$,R4		; FIX UP MISTAKE - USE RIGHT MESSAGE
;
1000$:	MOVB	@#RTERRB,O.FL+1(R5)	; SAVE ERROR CODE FOR LATER
	.PRINT	#7100$			; IDENTIFY OURSELVES
	.PRINT	R4			; TELL ERROR
	.POP	<R4,R0>			; RESTORE
	SEC				; SET FAILED
	RETURN				; HOME
;
;
;		--- ERROR MESSAGES ---
;
.NLIST	BIN
7000$:	.ASCIZ	/? CHANNEL ALREADY OPEN ?/
7001$:	.ASCIZ	/? ILLEGAL DEVICE ?/
7002$:	.ASCIZ	/? DEVICE HANDLER NOT LOADED ?/
7003$:	.ASCIZ	/? CHANNEL IN USE BY SOMEONE ELSE ?/
7004$:	.ASCIZ	/? NON-EXISTENT DEVICE\FILE ?/
7005$:	.ASCIZ	/? MULTIPLE VOLUMES NOT ALLOWED ON DEVICE ?/
7100$:	.ASCII	/"FILSPT" -ENTER ERROR- /<200>
.EVEN
.LIST	BIN
;
;

;
;
;
.SBTTL	ROUTINE	- 'FIL.LK'	...	LOOKUP A FILE
;
;
;		--- INITIALIZATION SECTION ---
;
;	ENSURE THAT 'FDB' IS NOT ALREADY ALLOCATED TO AN OPEN CHANNEL,
;	AND THAT DEVICE DRIVER IS LOADED INTO MEMORY. IF ANY ERROR
;	IS DETECTED ABORT VIA '1000$' WITH R4 HOLDING AN APPROPRIATE
;	ERROR MESSAGE. ALSO SET UP THE DEVICE SIZE IN THE 'O.MS' WORD.
;
;
FIL.LK:	.PUSH	<R0,R4>			; SAVE REGISTERS
	MOV	#7000$,R4		; GET ERROR MESSAGE READY
	BIT	#1,O.FL(R5)		; CHANNEL OPEN?
	BNE	1000$			; YES -> ERROR TIME
	MOV	#7001$,R4		; GET MESSAGE READY
	.DSTAT	#EMTBLK,(R5)		; FIND DEVICE STATS
	BCS	1000$			; FAILED -> ERROR EXIT
	MOV	#7002$,R4		; GET MESSAGE READY
	TST	EMTBLK+4		; HANDLER LOADED
	BEQ	1000$			; NOT IN MEMORY -> ERROR EXIT
	MOV	EMTBLK+6,O.MS(R5)	; SAVE DEVICE SIZE
	BNE	100$			; DISK TYPE DEVICE -> 100$
	DEC	O.MS(R5)		; FUDGE FOR SEQUENTIAL DEVICES ONLY
100$:
;
;
;
;		--- LOOKUP TIME ---
;
;
;	EXECUTE A LOOKUP FOR THE FILE SPECIFIED IN THE FDB. IF LOOKUP
;	SUCEEDS MARK FDB AS ALLOCATED TO A CHANNEL AND REWIND FILE.
;
;
	.LOOKUP	#EMTBLK,O.CH(R5),O.NR(R5); TRY AND LOOKUP FILE
	BCS	777$			; FAILED -> 777$
	MOV	#1,O.FL(R5)		; SET 'CHANNEL OPEN' FLAG
	CLR	O.BL(R5)		; REWIND FILE
;
;
;

;
;
;
;		--- SIZING FILE/DEVICE ---
;
;
;	WE PREVIOSLY FOUND THE DEVICE SIZE FROM THE DATA RETURNED IN THE
;	'DSTAT' EMT. CHECK TO SEE IF WE ARE OPENNING A DEVICE OR A FILE.
;	IF OPPENNING A DEVICE THEN ALREADY KNOW ITS 'SIZE', SO EXIT. IF
;	OPENNING A FILE THEN HAVE TO FIND ITS SIZE. DO THIS VIA A MESSY
;	TRICK. WE ISSUE A '.SAVES' EMT, WHICH SAVES THE CHANNEL STATUS
;	IN MEMORY. WE FIND THE FILE SIZE FROM THIS SAVED DATA, AND THEN
;	RESTORE THE CHANNEL.
;
;
	MOV	O.NR(R5),R4		; FILE SPEC -> R4
	TST	2(R4)			; FILE OR DEVICE LOOKUP?
	BEQ	400$			; DEVICE LOOKUP -> 400$
	.SAVES	#EMTBLK,O.CH(R5),#EMTBLK+4; MESSY WAY OF GETTING FILE SIZE
	BCS	776$			; FAILED -> ERROR
	MOV	EMTBLK+10,O.MS(R5)	; SAVE FILE SIZE
	.REOPEN	#EMTBLK,O.CH(R5),#EMTBLK+4; MUST REOPEN FILE
;
;
400$:	.POP	<R4,R0>			; RESTORE
	CLC				; CLEAR FAIL
	RETURN				; HOME
;
;
;

;
;
;
;		--- ERROR SECTION ---
;
;
;	IF WE GET HERE THE LOOKUP FAILED. PRINT OUT AN APPROPRIATE
;	ERROR MESSAGE AND RETURN WITH 'C' FLAG SET TO INDICATE
;	ERROR.
;
;
;
776$:	.PURGE	O.CH(R5)		; PURGE CHANNEL
	CLR	O.FL(R5)		; SHOW CHANNEL CLOSED
	MOV	#7005$,R4		; MESSAGE -> R4
	BR	1000$			; ERROR HANDLER TIME
777$:	MOV	#7003$,R4		; GET DEFAULT MESSAGE READY
	TSTB	@#RTERRB		; WAS IT MESSAGE 0
	BEQ	1000$			; YES -> GO PROCESS IT
	MOV	#7004$,R4		; FIX UP MISTAKE - USE RIGHT MESSAGE
;
1000$:	MOVB	@#RTERRB,O.FL+1(R5)	; SAVE ERROR CODE FOR LATER
	.PRINT	#7100$			; IDENTIFY OURSELVES
	.PRINT	R4			; TELL ERROR
	.POP	<R4,R0>			; RESTORE
	SEC				; SET FAILED
	RETURN				; HOME
;
;
;		--- ERROR MESSAGES ---
;
.NLIST	BIN
7000$:	.ASCIZ	/? CHANNEL ALREADY OPEN ?/
7001$:	.ASCIZ	/? ILLEGAL DEVICE ?/
7002$:	.ASCIZ	/? DEVICE HANDLER NOT LOADED ?/
7003$:	.ASCIZ	/? CHANNEL IN USE BY SOMEONE ELSE ?/
7004$:	.ASCIZ	/? NON-EXISTENT DEVICE\FILE ?/
7005$:	.ASCIZ	/? MULTIPLE VOLUMES NOT ALLOWED ON DEVICE ?/
7100$:	.ASCII	/"FILSPT" -LOOKUP ERROR- /<200>
.EVEN
.LIST	BIN
;

;
.SBTTL	ROUTINE	- 'FIL.CL'	...	CLOSE A FILE
;
;
;
FIL.CL:	.CLOSE	O.CH(R5)		; CLOSE CHANNEL
	CLR	O.FL(R5)		; CLEAR FLAGS
	RETURN				; HOME
;

.SBTTL	ROUTINE	- 'FIL.PG'	...	PURGE CHANNEL
;
;
FIL.PG:	.PURGE	O.CH(R5)		; PURGE CHANNEL
	CLR	O.FL(R5)		; CLEAR FLAGS
	RETURN				; HOME
;
;
;

;
;
.SBTTL	ROUTINE	- 'FIL.RD'	...	READ IN FILE
;
;
;
;		--- INITIALIZATION SECTION ---
;
;
;	CHECK TO SEE THAT 'FDB' IS ASSOCIATED WITH A FILE, AND THAT
;	WE ARE NOT TRYING TO READ PAST EOF. IF ATTEMPTING TO READ
;	TO MUCH DATA TRUNCATE READ REQUEST AUTOMATICALLY!
;
;
FIL.RD:	.PUSH	R4			; SAVE
	MOV	#7000$,R4		; SET UP MESSAGE ADDRESS
	BIT	#1,O.FL(R5)		; FILE OPEN?
	BEQ	1000$			; YES -> TELL HIM HE IS A FOOL
;
	CLR	O.TR(R5)		; ASSUME CANNOT COPY ANY
	MOV	O.MS(R5),R4		; MAX NUMBER BLOCKS -> R4
	SUB	O.BL(R5),R4		; MAX NUMBER BLOCKS CAN READ -> R4
	BLOS	710$			; TOO MUCH -> EOF
	MOV	O.TS(R5),R0		; NUMBER WORDS TO COPY -> R0
	MOV	R0,O.TR(R5)		; ASSUME WILL COPY ALL
	ADD	#377,R0			; ROUND UP
	SWAB	R0			; NUMBER BLOCKS TO READ -> R0
	BIC	#^C377,R0		; LEAVE BLOCK NUMBER ONLY
	CMP	R0,R4			; TRYING TO READ TOO MUCH?
	BLOS	100$			; NO -> OK
	SWAB	R4			; MAX NUMBER WORDS CAN COPY -> R4
	MOV	R4,O.TR(R5)		; THIS IS MOST CAN READ
100$:
;
;		--- READ SECTION ---
;
;
;	ATTEMPT READ. IF NO ERROR DETECTED THEN SAVE ACTUAL NUMBER OF
;	WORDS READ AND EXIT.
;
;
	MOV	#7002$,R4		; READY FOR A HARD ERROR
	.READW	#EMTBLK,O.CH(R5),O.BF(R5),O.TR(R5),O.BL(R5)
	BCS	700$			; FAILED -> 700$
	MOV	R0,O.TR(R5)		; SAVE NUMBER WORDS ACTUALLY TRANSFERED
	.POP	R4			; RESTORE
	CLC				; CLEAR ERROR FLAG
	RETURN				; HOME
;
;

;
;
;		--- EOF ERROR ---
;
;	ATTEMPT TO READ PAST EOF. PRINT WARNING MESSAGE AND EXIT WITH
;	'C' SET AND R0 ZEROED.
;
700$:	MOVB	@#RTERRB,O.FL+1(R5)	; SAVE ERROR CODE
	BNE	1000$			; NOT EOF ERROR -> SKIP
710$:	.PRINT	#7100$			; IDENTIFY MODULE
	.PRINT	#7001$			; TELL WENT PAST EOF
	.POP	R4			; RESTORE
	CLR	R0			; EOF MARKER
	SEC				; SET FAILED
	RETURN				; HOME
;
;
;		--- GENERAL ERRORS ---
;
;
;	PRINT ERROR MESSAGE AND EXIT WITH 'C' SET AND R0 NON-ZEROED.
;
;
1000$:	.PRINT	#7100$			; IDENTIFY MODULE
	.PRINT	R4			; PRINT ERROR MESSAGE
	.POP	R4			; RESTORE
	MOV	#1,R0			; NOT EOF ERROR
	SEC				; SET FAILED
	RETURN				; HOME
;
;
;
;		--- ERROR MESSAGES ---
;
;
.NLIST	BIN
7000$:	.ASCIZ	/? CHANNEL NOT OPEN ?/
7001$:	.ASCIZ	/? ATTEMPTED READ PAST EOF ?/
7002$:	.ASCIZ	/? HARD ERROR OCCURED ON CHANNEL ?/
7100$:	.ASCII	/"FILSPT" - READ ERROR - /<200>
.EVEN
.LIST	BIN
;

;
;
;
.SBTTL	ROUTINE	- 'FIL.WT'	...	WRITE TO FILE
;
;
;
;		--- INITIAL SECTION ---
;
;
;	ENSURE THAT CHANNEL IS OPEN AND THAT WE WILL NOT WRITE
;	PAST EOF!
;
;
FIL.WT:	.PUSH	R4			; SAVE
	MOV	#7000$,R4		; ERROR MESSAGE READY
	BIT	#1,O.FL(R5)		; CHANNEL OPEN?
	BEQ	1000$			; YES -> ABORT
;
	CLR	O.TR(R5)		; ASSUME CANNOT COPY ANY
	MOV	O.MS(R5),R4		; MAX NUMBER BLOCKS -> R4
	SUB	O.BL(R5),R4		; MAX NUMBER BLOCKS CAN READ -> R4
	BLOS	710$			; TOO MUCH -> EOF
	MOV	O.TS(R5),R0		; NUMBER WORDS TO COPY -> R0
	MOV	R0,O.TR(R5)		; ASSUME WILL COPY ALL
	ADD	#377,R0			; ROUND UP
	SWAB	R0			; NUMBER BLOCKS TO READ -> R0
	BIC	#^C377,R0		; LEAVE BLOCK NUMBER ONLY
	CMP	R0,R4			; TRYING TO READ TOO MUCH?
	BLOS	100$			; NO -> OK
	SWAB	R4			; MAX NUMBER WORDS CAN COPY -> R4
	MOV	R4,O.TR(R5)		; THIS IS MOST CAN READ
100$:
;
;
;		--- WRITE TO FILE ---
;
;
;	DO TRANSFER, AND IF OK EXIT AFTER SETTING UP NUMBER OF
;	WORDS ACTUALLY TRANSFERED.
;
;
	MOV	#7002$,R4		; ERROR MESSAGE READY
	.WRITW	#EMTBLK,O.CH(R5),O.BF(R5),O.TR(R5),O.BL(R5)
	BCS	700$			; FAILED -> 700$
	MOV	R0,O.TR(R5)		; SAVE AMOUNT TRANSFERED
	.POP	R4			; RESTORE
	CLC				; CLEAR FAILED FLAG
	RETURN				; EXIT
;
;

;
;
;		--- ERROR MESSAGE SECTION ---
;
;
;	TELL OF ERRORS AND EXIT. NOTE RETURN WITH R0=0 IFF ERROR
;	IS DUE TO EOF.
;
;
;
700$:	MOVB	@#RTERRB,O.FL(R5)	; SAVE ERROR CODE
	BNE	1000$			; HARD ERROR -> 1000$
710$:	.PRINT	#7100$			; IDENTIFY SELF
	.PRINT	#7001$			; TELL EOF OCCURED
	.POP	R4			; RESTORE
	CLR	R0			; SET EOF FLAG
	SEC				; SET FAILED
	RETURN				; EXIT
;
;
1000$:	.PRINT	#7100$			; IDENTIFY SELF
	.PRINT	R4			; IDENTIFY ERROR
	.POP	R4			; RESTORE
	MOV	#1,R0			; INDICATE HARD ERROR
	SEC				; SET FAILED
	RETURN				; HOME
;
;
;
;		--- MESSAGES ---
;
.NLIST	BIN
7000$:	.ASCIZ	/? CHANNEL NOT OPEN ?/
7001$:	.ASCIZ	/? ATTEMPTED WRITE PAST EOF ?/
7002$:	.ASCIZ	/? HARD ERROR ON CHANNEL ?/
7100$:	.ASCII	/"FILSPT" - WRITE ERROR - /<200>
.EVEN
.LIST	BIN
;

.SBTTL	ROUTINE	- 'FIL.ST'	...	PRINT FILE STATUS
;
;
FIL.ST:	.PUSH	<R0,R1,R2>		; SAVE
	.OUT	<#7000$,O.NA(R5),#7001$>; IDENTIFY MODULE AND FILE
;
	CNV	O.CH(R5),7110$		; SET UP CHANNEL NUMBER
	CNV	O.MS(R5),7120$		; SET UT DEVICE SIZE
	CNV	O.FL+1(R5),7130$,BYTE	; SET UP ERROR CODE
	CNV	O.FL(R5),7140$,BYTE	; SET UP STATUS FLAGS
;
	CNV	O.BF(R5),7210$		; BUFFER ADDRESS
	CNV	O.BL(R5),7220$		; SET UP BLOCK NUMBER
	CNV	O.TR(R5),7230$		; TRANSFERED WORDS IN REALITY
	CNV	O.TS(R5),7240$		; SET UP TRANSFER SIZE
;
	.PRINT	#7100$			; PRINT STATS
	.POP	<R2,R1,R0>		; RESTORE
	RETURN
;
;
.NLIST	BIN

	.ASCII	<15><12>
;

7220$:	.ASCII			/XXXXXX  COPIED=/

	.ASCIZ	//
