; THIS MODULE CONTAINS THE 'REPORT' PROGRAM, WHICH IS USED TO PRODUCE
; A REPORT WITH THE CONTENTS OF A DATA-BASE. THIS MODULE MUST BE LINKED
; WITH THE FORMAT DEFINITION FILE FOR THE DATABASE, AND THE DATABASE
; SUPPORT LIBRARY, 'DBSLIB.OBJ'.
;
;
; FEATURES OF THE REPORT GENERATOR ARE-
;
;		THE REPORT LAYOUT IS DEFINED IN A FORMAT FILE, WHICH IS
;		STRAIGHT ASCII
;
;		BOTH INPUT AND OUTPUT STREAMS CAN BE DIRRECTED TO FILES,
;		THE TERMINAL AND VARIABLES.
;
;		ITS JUST GREAT!
;
;
;

;
;
;
.SBTTL	DOCUMENTATION	- USING THE REPORT GENERATOR
;
;
; THIS PROGRAM IS DESIGNED TO BE CHAINED TO FROM THE USER SELECTOR
; PROGRAM. AN EXAMPLE SELECTOR PROGRAM IS INCLUDED IN THE DATA-BASE
; PACKAGE. THE SELECTOR PROGRAM WILL PLACE THE NUMBER OF THE RECORDS,
; AND THE CONTENTS OF THE FIELD BY WHICH THEY ARE TO BE SORTED INTO
; THE 'SORTER.DAT' FILE, AND THEN CHAIN TO THE 'SORTER.SAV' PROGRAM,
; OR DIRRECTLY TO THIS PROGRAM IF SORTING IS NOT NEEDED.
;
; UPON ACTIVATION OF THIS PROGRAM, THE FOLLOWING FILES ARE OPENED
;
;
;	'LST:XXXXXX.LST'	AS THE OUTPUT FILE
;
;	'DK:XXXXXX.FRM'	AS THE REPORT FORMAT DEFINITION FILE
;
;	'DK:SORTER.DAT'	AS THE RECORD NUMBER TABLE FILE
;
; IF THE 'XXXXXX.FRM' FILE IS NOT FOUND, THE USER IS PROMPTED FOR THE NAME
; OF THE FORMAT FILE.
;
; THE PROGRAM WILL TRANSFER CHARACTER FROM THE INPUT STREAM (INITIALLY THE
; FORMAT FILE) TO THE OUTPUT STREAM (INITIALLY THE LIST FILE). FACILITIES
; ARE AVAILABLE FOR SUBSTITION OF SPECIAL CHARACTERS, SETING UP LOOPS, ETC.
; OPERATION IS SIMILAR TO 'RUNOFF'.
;
;
;
; NOTE 'XXXXXX' IN THE ABOVE IS THE SIX LETTER NAME GIVEN TO THE DATABASE
;	IN THE 'FORM' DECLARATION WHEN THE DATA-BASE DESCRIPTOR WAS
;	GENERATED.

;
;
;
.SBTTL	DOCUMENTATION	- SPECIAL CHARACTERS IN FORMAT FILE
;
;
;	THE FORMAT FILE CAN CONTAIN SPECIAL CHARACTERS, WHICH CONTROL
; THE GENERATION OF THE REPORT. ANY NON-SPECIAL CHARACTER WILL BE
; SENT DIRRECTLY TO THE OUTPUT STREAM. A SPECIAL CHARACTER IS INDICATED
; BY PREFIXING WITH THE '$' (DOLLOR) SYMBOL. IN THE EVENT THAT THE
; '$' SYMBOL IS TO BE SENT TO THE OUTPUT STREAM, IT MUST BE PREFIXED
; WITH A '_' SYMBOL, WHICH MEANS 'PASS UNCONVERTED'. THE '_' CHARACTER
; CAN BE USED TO PREFIX ANY CHARACTER. IN SUMMARY
;
;	ANY SINGAL ASCII CHARACTER OTHER THAN '$' OR '_' WILL BE
;	PASSED DIRRECTLY TO THE OUTPUT STREAM.
;
;	THE CHARACTER '$' RESULTS IN THE NEXT CHARACTER BEING
;	USED TO SELECT A FUNCTION/OPERATION.
;
;	THE '_' CHARACTER WILL BE IGNORED, AND WILL CAUSE THE
;	NEXT CHARACTER TO BE PASSED UNCONVERTED TO THE OUTPUT
;	STEAM. (IE -$ --> PASS '$' TO OUTPUT STREAM. __ -> PASS 1 _)
;
;
;
;

;
;
;
.SBTTL	DOCUMENTATION	- SPECIAL CHARACTERS
;
;
; THE SPECIAL CHARACTERS ARE AS FOLLOWS
;
;
; $C	COMMENT. THE REST OF THE INPUT LINE (UPTO AND INCLUDING THE <LF>)
;	IGNORED.
;
; $D	DATE. THIS CHARACTER IS REPLACED WITH THE DATE.
;
; $E	TERMINATES PHASE 2 ON AN EOF CONDITION (ALL RECORDS PROCESSED)
;
; $F	OPENS NEW OUTPUT FILE. (ONLY IN PHASE 1, FIRST TIME ONLY)
;
; $P	PAGE. THIS CHARACTER IS REPLACED WITH THE PAGE NUMBER.
;
; $O	OUTPUT. CAUSES THE OUTPUT STREAM TO BE SWITCHED.
;
; $I	INPUT. CAUSES THE INPUT STREAM TO BE SWITCHED.
;
; $N	NEXT. CAUSES THE NEXT RECORD TO BE READ INTO MEMORY. THE ORDER
;	IN WHICH RECORDS ARE ACCESSED IS DETERMINED BY 'SORTER.DAT'.
;
; $<	DATA. STARTS A DATA SUBSTITUTION OPERATION.
;
; $=	TERMINATES PHASE 2 IFF TWO VARIABLES ARE EQUAL
;
; $#	TERMINATES PHASE 2 IFF TWO VARIABLES ARE NOT EQUAL
;
; $"	USED TO TOGGLE 'REPORT OUTPUT' SUPPRESSION
;
; $\	PHASE. INDICATES END OF A PHASE.
;
;

;
;
;
.SBTTL	DOCUMENTATION	- OUTPUT SWITCH OPERATION
;
;
; THE SEQUENCE $O? CAUSES ALL FURTHER OUTPUT TO BE DIRRECTED TO OUTPUT
; STREAM ?, WHERE ? CAN BE ANY OF THE FOLLOWING-
;
;	!	->	SEND TO REPORT FILE (REPORT.LST)
;
;	@	->	SEND TO TTY
;
;	A-Z	->	SEND TO VARIABLE. 26 VARIABLES, EQUIVALENT TO
;			THE BASIC A$ - Z$ VARIABLES ARE PROVIDED. EACH
;			VARIABLE CAN HOLD UP TO 254 CHARACTERS.
;
;
;
.SBTTL	DOCUMENTATION	- INPUT SWITCH OPERATION
;
;
; THE SEQUENCE $I? CAUSES ALL FURTHER INPUT TO BE OBTAINED FROM THE
; STREAM ?, WHERE ? CAN BE -
;
;	!	->	OBTAIN INPUT FROM (REPORT.FRM)
;
;	@	->	OBTAIN INPUT FROM TTY
;
;	A-Z	->	OBTAIN INPUT FROM VARIABLE A/Z
;
;
; IF TTY INPUT IS REQUIRED, THE '^' SYMBOL IS TYPED ON THE TERMINAL.
; THE INPUT OPERATION IS ENDED WHEN A '^Z' IS TYPED. IF INPUT IS
; REQUIRED FROM A VARIABLE, ALL THE CHARACTERS IN THE VARIABLE ARE
; SENT TO THE OUTPUT STEAM. INPUT IS AUTOMATICALLY REDIRECTED TO
; THE ! STREAM WHEN A 'Z' IS TYPED ON THE TERMINAL, OR WHEN ALL
; CHARACTERS IN A VARIABLE HAVE BEEN OUTPUT. ANY INPUT OBTAINED
; FROM A STREAM OTHER THAN THE '!' IS PASSED DIRRECTLY, WITHOUT
; CONVERSION TO THE OUTPUT STREAM.
;
;
;

;
;
;
.SBTTL	DOCUMENTATION	- NEXT RECORD OPERATION
;
;
; UPON ENCOUNTERING THE '$N' SEQUENCE, THE NUMBER OF THE NEXT RECORD
; CONTAINED IN THE 'SORTER.DAT' FILE IS USED TO READ INTO MEMORY A
; DATABASE RECORD. IF THE 'EOF' CONDITION OCCURES ON 'SORTER.DAT',
; THEN NO NEW RECORD IS READ.
;
;
;
.SBTTL	DOCUMENATION	- DATA SUBSTITUTION OPERATION
;
;
; THE SEQUENCE '$<' INDICATES THE START OF A DATA SUBSTITUTION. UPON
; OCCURANCE OF A $<, THE PROGRAM WILL REPLACE THE DATA SUBSTITUTION
; STRING WITH THE CONTENTS OF THE SPECIFIED FIELD FOR THE RECORD
; CURRENTLY IN MEMORY. THE DATA SUBSTITUTION STRING IS THE CHARACTER
; SEQUENCE-
;
;	[$<] [NAME] [DELIMITER]
;
; WHERE	NAME	IS THE NAME OF THE FIELD IN THE RECORD
;
;	AND	LEGAL DELIMITERS ARE '>' AND '<'. THE '>' DELIMITER
;		SUBSTUTES DATA DIRRECTLY, WHILE THE '<' DELIMITER
;		CUASES THE DATA TO BE TRIMMED PRIOR TO SUBSTITUTION.
;
; EXAMPLE	$<SURNAME>
;
;	THIS WILL CAUSE THE STRING '$<SURNAME>' TO BE REPLACED WITH THE
;	CONTENTS OF THE FIELD LABLED 'SURNAME' FOR THE RECORD CURRENTLY
;	IN MEMORY.
;
;
; NOTE	- IF THE RECORD DOES NOT CONTAIN A FIELD OF THE SPECIFIED NAME,
;	  THE CHARACTERS '???' APPEAR IN THE OUTPUT.
;
;
;

;
;
;
.SBTTL	DOCUMENTATION	- END OF PHASE OPERATION
;
;
;	THE PROGRAM OPERATES IN TWO PHASES. THE FIRST PHASE IS USED TO
; PROCESS HEADING, WHILE THE SECOND IS USED TO PROCESS DATABASE DATA.
;
; IN PHASE 1, THE PROGRAM WILL CAUSE THE OUTPUT FILE TO BE CLOSED
; AND THE PROGRAM TO TERMINATE WHEN ALL RECORDS HAVE BEEN PROCESSED. THE
; OCCURANCE OF A '$\' SEQUENCE CUASES PHASE 2 TO BE ENTERED.
;
; IN PHASE 2, THE PROGRAM WILL IGNORE THE EOF OCCURANCE. PHASE 1 IS
; RE-ENTERED WHEN ANOTHER '$\' SEQUENCE IS ENCOUNTERED. THIS ALSO
; CAUSES THE FORMAT DEFINITION FILE TO BE RE-WOUND.
;
;
;

;
;
;
.SBTTL	DOCUMENTATION	- CONDITIONAL TERMINATION OF PHASE TWO
;
;
; THE '$E' COMMAND WILL TERMINATE PHASE TWO IF ALL RECORDS HAVE BEEN
; PROCESSED.
;
; THE '$=??' COMMAND WILL TERMINATE PHASE TWO IF THE CONTENTS OF THE
; TWO VARIABLES ?? ARE IDENTICAL.
;
; THE '$#??' COMMAND WILL TERMINATE PHASE TWO IF THE CONTENTS OF THE
; TWO SPECIFIED VARIABLES ARE NOT IDENTICAL.
;
;
;
.SBTTL	DOCUMENTATION	- COMMENTS AND FORMATTING
;
;
; ALL CHARACTERS IN THE INPUT STREAM ARE PASSED TO THE OUTPUT, THEREFORE
; MAKING IT DIFFICULT TO COMMENT AND FORMAT THE COMMANDS IN THE FORMAT
; FILE. THE FOLLOWING COMMANDS SHOULD BE USED -
;
;	$C	COMMENT - ALL CHARACTERS FOLLOWING THE '$C' UPTO AND
;		INCLUDING THE NEXT <CR><LF> ARE DISCARDED. THIS IS
;		EQUIVALENT TO THE BASIC 'REMARK' STATEMENT.
;
;	$"	SUPPRESSION - ALL OUTPUT TO THE REPORT LISTING
;		FILE, (ONLY THE '!' STREAM), FOLLOWING THE '$"', UPTO
;		AND INCLUDING THE NEXT '$"' WILL BE DISCARDED. THIS
;		DOES NOT AFFECT OUTPUT TO ANY OTHER STREAM, OR INPUT.
;		IT ALLOWS COMPLEXE COMMANDS TO BE PROPERLY FORMATTED
;		AND COMMENTED.
;
;
;

;
;
;
.SBTTL	DOCUMENTATION	- USING FILE OTHER THAN 'LST:XXXXXX.LST' FOR REPORT
;
;
; THE PROGRAM SENDS THE REPORT TO THE FILE 'LST:XXXXXX.LST' BY DEFAULT.
; IN SOME CASES, IT MAY BE NECESSARY TO DIRRECT THE OUTPUT TO ANOTHER
; FILE. IN THIS CASE, THE '$F' COMMAND CAN BE USED. THE FORMAT OF THE
; COMMAND IS
;
;		$FDDD:NNNNNN.XXX<CR><LF>
;
;
;	WHERE	DDD:NNNNNN.XXX IS ANY LEGAL FILE SPEC. NOTE THAT THE
;		FILE SPEC MUST BE TERMINATED WITH A <CR><LF>.
;
; THIS COMMAND MUST BE ON A LINE BY ITSELF. THE COMMAND IS VALIDE ONLY
; DURING PHASE 1, AND WILL ONLY BE EXECUTED ON THE FIRST PASS. ON
; SEBSEQUENT PASSES IT IS TREATED AS A NON-OP.
;
;
;

.SBTTL	DEFINITION	- MACRO '.OUT'
;
;
;	THIS ROUTINE CAUSES A NUMBER OF CHARCATERS TO BE OUTPUT TO
;	THE CONSOLE.
;
;
.MACRO	.OUT	CHAR
	.IF	B,<CHAR>
		CALL	@OUTVEC
		.MEXIT
	.ENDC
	MOV	R0,-(SP)
	.IRP	X,<CHAR>
		MOVB	X,R0
		CALL	@OUTVEC
	.ENDR
	MOV	(SP)+,R0
.ENDM	.OUT
;
;
;
.SBTTL	DEFINITION	- MACRO '.IN'
;
;
.MACRO	.IN
		CALL	@INPVEC
.ENDM	.IN
;
;
;
.SBTTL	DEFINITION	- MACRO 'ERROR'
;
;
.MACRO	ERROR	MES,?A,?B,?C
	.PRINT	#'B
	.EXIT
B:	.ASCIZ	/MES/
.EVEN
C:
.ENDM	ERROR
;
;

;
;
;
.MACRO	COPY	FROM,TO
	.IIF	NB,FROM,	MOV	#FROM,R0
	.IIF	NB,TO,		MOV	#TO,R1
	CALL	COPY
.ENDM	COPY

;
.SBTTL	INITIALIZATION SECTION
;
;
;
;	OPEN FILES
;
;
REPORT:	.PRINT	#HELLO			; SAY HELLO
	CALL	SFLINT			; INITIALIZE SORT FILE
	CALL	RPTINT			; REPORT OUTPUT FILE
	CALL	FRMINT			; FORMAT DESCRIPTION FILE
	CALL	IOMINT			; INITIALIZE I/O STREAMS
;
;
;
;	OBTAIN ACCESS TO DATA BASE
;
;
	MOV	#FORM$$,(PC)+		; INITIALIZE FOR DBSLIB ROUTINES
$$FORM:	.WORD	0			; DATABASE FDT
	CALL	FILINT			; OPEN DATA BASE
	MOV	#1,R0			; READ IN ....
	CALL	FIL.DR			; .... RECORD #1
	BCC	100$			; IF OK -> 100$
	ERROR	<CANNOT ACCESS DATABASE!>
;
;
;
;	INITIALIZE VARIABLES
;
100$:	MOV	#1,PAGENO		; SET UP PAGE NUMBER
	MOV	SFLPAR+2,RECNUM		; NUMBER OF RECORDS
	JMP	DOER			; ENTER PHASE 1
;
;
.NLIST	BIN
HELLO:	.ASCII	<12><12><12><12><12><12><12><12><12><12><12><12>
	.ASCII	<12><12><12><12><12><12><12><12><12><12><12><12><15>
	.ASCIZ	/          "REPORT" (VERSION RDM110381.02)/<12>
.LIST	BIN
;
;

;
;
;
.SBTTL	ROUTINE - 'DOER'	...	MAIL LOOP
;
;
;	PHASE 1 ROUTINE. HERE WE ARE OUTPUTTING THE HEADER. USE COMMAND TABLE
;	7777$ TO DECODE SPECIAL CHARACTERS, ELSE SEND CHARACTER TO OUTPUT
;	STEAM. AS 'DOER1' RETURNS CONTROL HERE AFTER PROCESSING, ENSURE THAT
;	ALL I/O IS REDIRRECTED TO DEFAULT STREAMS, AND THAT THE REPORT FORMAT
;	FILE IS REWOUND.
;
;
DOER:	TST	RECNUM			; ALL DONE?
	BEQ	.BYE			; ALL DONE -> BYE
	BNE	10$			; NO -> SKIP
	CALL	RPTEND			; CLOSE OUTPUT FILE
	.EXIT				; BYE
;
10$:	CLR	RPTFLG			;; DISABLE REPORT SUPPRESSION
	CALL	IOMINT			; RESET I/O VECTORS
	CALL	FRMRST			; RESET FORM INPUT
;
	MOV	#7777$,R5		; R5 --> TABLE
100$:	.IN				; GET A CHAR
	BCS	500$			; NO MORE CHARS -> EXIT
	CALL	DECODE			; ATTEMPT TO DECODE
	BCC	1000$			; CONTROL CHAR -> 1000$
	.OUT				; OUTPUT
	BR	100$			; LOOP
;
500$:	ERROR	<<'DOER' - UNEXPECTEDLY HIT EOF?>>
;
;
1000$:	CALL	@R0			; CALL SERVANT
	BR	100$			; LOOP
;
;
;

;
;
;
.SBTTL	'DOER' COMMAND DISPATCH TABLE
;
;
;
.MACRO	ENTRY	A,B
	.WORD	A,B
.ENDM	ENTRY
;
;
;
7777$:	ENTRY	<<'C+200>>,..COMM	; $C -> COMMENT
	ENTRY	<<'D+200>>,..DATE	; $D -> DATE
	ENTRY	<<'F+200>>,..FILE	; $F -> NEW OUTPUT FILE
	ENTRY	<<'P+200>>,..PAGE	; $P -> PAGE
	ENTRY	<<'O+200>>,..OUT	; $O -> OUTPUT STREAM
	ENTRY	<<'I+200>>,..IN		; $I -> INPUT STREAM
	ENTRY	<<'N+200>>,..NEXT	; $N -> NEXT RECORD
	ENTRY	'<+200,..DATA		; $< -> DATA
	ENTRY	<<'"+200>>,..SUPP	; $" -> OUTPUT SUPPRESS
	ENTRY	<<'\+200>>,DOER1	; $\ -> START PROCESSING
	.WORD	0,0
;

;
;
;
;		THIS ROUTINE IS USED TO EXIT THE PROGRAM.
;		IT RETUENS CONTROL TO THE MENU PROGRAM.
;
;
.BYE:	CALL	RPTEND					; CLOSE OUTPUT FILE
	MOV	#15.,R1					; R1 = CHANNEL NUMBER
10$:	.PURGE	R1					; PURGE
	SOB	R1,10$					; LOOP
	MOV	#100$,R0				; SOURCE
	MOV	#500,R1					; DESTINATION
	.REPT	4					; --- 4 WORDS ---
		MOV	(R0)+,(R1)+			;   SET UP FILE SPEC
	.ENDR						; ----------------
	.CHAIN						; EXECUTE CHAIN
;
;
100$:	.RAD50	/DK RDM   SAV/				; CHAIN SPEC
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'DOER1'	...	BODY OF TEXT CONTROL CODE
;
;
;	PHASE 2 ROUTINE. SEND CHARACTERS TO OUTPUT STREAM, UNLESS IT
;	IS A SPECIAL, IN WHICH CASE USE TABLE 7777$ TO DECODE IT.
;
;
;
DOER1:	TST	(SP)+			; REMOVE RETURN ADDRESS
	MOV	#7777$,R5		; R5 --> TABLE
100$:	.IN				; GET A CHAR
	BCS	500$			; NO MORE CHARS -> EXIT
	CALL	DECODE			; ATTEMPT TO DECODE
	BCC	1000$			; CONTROL CHAR -> 1000$
	.OUT				; OUTPUT
	BR	100$			; LOOP
;
500$:	ERROR	<<'DOER' - UNEXPECTEDLY HIT EOF?>>
;
;
1000$:	CALL	@R0			; CALL SERVANT
	BR	100$			; LOOP
;
;

;
;
;
.SBTTL	'DOER1' COMMAND DISPATCH TABLE
;
;
;
7777$:	ENTRY	<<'C+200>>,..COMM	; $C -> COMMENT
	ENTRY	<<'D+200>>,..DATE	; $D -> DATE
	ENTRY	<<'E+200>>,..EOFT	; $E -> EOF TEST
	ENTRY	<<'P+200>>,..PAGE	; $P -> PAGE
	ENTRY	<<'O+200>>,..OUT	; $O -> OUTPUT STREAM
	ENTRY	<<'I+200>>,..IN		; $I -> INPUT STREAM
	ENTRY	<<'N+200>>,..NEXT	; $N -> NEXT RECORD
	ENTRY	'<+200,..DATA		; $< -> DATA
	ENTRY	<<'=+200>>,..EQ		; $= -> END PHASE ON MATCH
	ENTRY	<<'#+200>>,..NE		; $# -> END PHASE ON NO MATCH
	ENTRY	<<'"+200>>,..SUPP	; $" -> OUTPUT SUPPRESS
	ENTRY	<<'\+200>>,..PEND	; $\ -> END PROCESSING
	.WORD	0,0
;
;
..PEND:	INC	PAGENO			; NEXT PAGE
	TST	(SP)+			; CLEAN UP STACK
	JMP	DOER			; RESTART FORM
;
;
..EOFT:	TST	RECNUM			; ANY RECS LEFT?
	BEQ	..PEND			; NO -> EXIT
	RETURN				; YES-> CONTINUE
;
;

;
;
;
.SBTTL	PRIMITIVE - '..OUT'	...	REDIRECT OUTPUT STREAM
;
;
;	CHECK NEXT CHARACTER. IF IS A '!' REDIRRECT OUTPUT TO REPORT FILE.
;	ANY OTHER CHARACTER IMPLIES WHICH TO DIRRECT OUTPUT TO A VARIABLE.
;	IN THIS CASE CALL 'VAROUT' TO PROCESS.
;
;
;
;
..OUT:	.IN				; GET NEXT CHARACTER
	CMP	R0,#'!			; SWITCH BACK TO OUTDFILE
	BNE	100$			; NO -> 1000$
	MOV	#RPTOUC,OUTVEC		; REDIRRECT OUTPUT
	RETURN				; DONE
;
100$:	JMP	VAROUT			; LET 'VAROUT' TAKE CARE OF IT
;
;
;

;
;
;
.SBTTL	PRIMITIVE - '..IN'	...	REDIRECT INPUT STREAM
;
;
;
;	CHECK NEXT CHARACTER. A '!' IMPLIES REDIRRECT INPUT TO THE
;	FORMAT DEFINITION FILE. ANY OTHER CHARACTER IMPLIES GET INPUT
;	FROM VARIABLE, WHICH IS HANDLED BY 'VARINP'
;
;
..IN:	.IN				; GET INPUT FROM ?
	CMP	R0,#'!			; INPUT FROM FRMIO?
	BEQ	100$			; YES -> 100$
	CALL	VARINP			; NO -> GET FROM VARIABLE
100$:	RETURN				; HOME
;
;

;
;
;
.SBTTL	PRIMITIVE - '..COMM'	...	PROCESSING A COMMENT
;
;
;	DISCARD ALL OUTPUT UNTILL DETECT A <LF>
;
;
..COMM:	MOV	#100$,OUTVEC		; DIRRECT OUTPUT TO NULL HANDLER
	RETURN				; HOME
100$:	CMP	R0,#<'J-'@>		; <LF>?
	BNE	110$			; NO -> 110$
	MOV	#RPTOUC,OUTVEC		; REDIRRECT OUTPUT
110$:	RETURN				; DONE
;
;
;

;
;
;
.SBTTL	PRIMITIVE	- '..SUPP'	...	SUPPRESS OUTPUT TO !
;
;
;	THIS ALLOWS COMMANDS TO BE FORMATTED WITH TABS ETC, AND THE
;	FORMATTING CHARACTERS NOT SENT TO THE REPORT FILE. WE TOGGLE
;	A BIT IN THE 'RPTFLG' BYTE (LSB). WHEN THE BIT IS SET ALL
;	OUTPUT TO STREAM ! IS DISCARDED.
;
;
..SUPP:	INCB	RPTFLG			; TOGGLE
	RETURN				; HOME
;
;

;
;
;
.SBTTL	PRIMITIVE - '..DATA'	...	RETURN FIELD DATA
;
;
..DATA:	.PUSH	<R1,R2,R3,R4,R5>	; SAVE R1
	MOV	#BUF,R1			; R1 --> BUFFER
100$:	CALL	FRMINC			; GET A CHARACTER
	BCS	500$			; FAILED -> 500$
	CMP	R0,#'>			; EOS?
	BEQ	1000$			; YES -> 1000$
	CMP	R0,#'<			; EOS?
	BEQ	1000$			; YES -> 1000$
	MOVB	R0,(R1)+		; SAVE
	BR	100$			; LOOP
;
500$:	ERROR	<'..DATA' - UNEXPECTEDLY HIT EOF?>
;
;
1000$:	MOV	R0,R5			; SAVE TERMINATOR
	CLRB	(R1)+			; SET EOF
	MOV	#FORM$$,R0		; R0 -> FORM TABLE
	MOV	#BUF,R1			; R1 -> FIELD NAME
	CALL	LOCFLD			; LOCATE FIELD
	BCS	10000$			; FAILED -> 10000$
	MOV	(R1),R0			; R0 = TYPE OF FIELD
	BIC	#^C7,R0			; DISCARD FLAGS
	CMP	R0,#2			; ASCII TYPE FIELD?
	BLOS	4000$			; YES -> 4000$
	CMP	R0,#3			; NUMBER TYPE FIELD?
	BEQ	5000$			; YES -> 5000$
	BR	6000$			; PROCEED
;
;

;
;
;	---	RETURNING CONTENTS OF ASCII FIELD --
;
;
4000$:	MOV	2(R1),R0		; SIZE = R1
	MOV	4(R1),R2		; R2 == OFFSET
	ADD	FORM$$,R2		; R2 --> FIELD
	MOV	#BUF,R1			; R1 --> BUFFER
4100$:	MOVB	(R2)+,(R1)+		; COPY
	SOB	R0,4100$		; LOOP
	BR	7000$			; EXIT
;

;
;
;
;	---	RETURNING CONTENTS OF NUMERIC FIELD ---
;
;
5000$:	MOV	4(R1),R0		; R0 == OFFSET
	ADD	FORM$$,R0		; R0 --> FIELD
	MOVB	(R0),BUF		; LOB -> BUF
	MOVB	1(R0),BUF+1		; HOB -> BUF+1
	MOV	#^B0101010100000000,R0	; R0 = FORMAT WORD
	MOV	#BUF,R1			; R1 -> BUFFER
	MOV	(R1),R2			; R2 = VALUE
	CALL	CNAFD5			; GET AS ASCII
	BR	7000$			; LOOP
;

;
;
;	---	RETURNING CONTENTS OF 'DATE' FIELD ---
;
;
6000$:	MOV	4(R1),R0			; R0 == OFFSET
	ADD	FORM$$,R0			; R0 --> FIELD
	MOVB	(R0),BUF			; LOB -> BUF
	MOVB	1(R0),BUF+1			; HOB -> BUF+1
;
	MOV	#0,R0				; R0 = FORMAT
	MOV	#BUF,R1				; R1 -> BUFFER
	MOV	BUF,R3				; DATE -> R3
;
	MOV	R3,R2				; DATE -> R2
	BIC	#^C31.,R2			; DAY COUNT -> R2
	CALL	CNAFD2				; GET DAY AS ASCII
	MOVB	#'-,(R1)+			; DELIMIT
;
	MOV	R3,R2				; DATE -> R2
	.REPT	5				; --- 2^5=32 ---
		ASR	R2			;   MONTH -> LOB
	.ENDR					; --------------
	BIC	#^C15.,R2			; LEAVE MONTH ONLY
	CALL	CNAFD2				; ASCII
	MOVB	#'-,(R1)+			; DELIMIT
;
	MOV	R3,R2				; DATE -> R2
	.REPT	11.				; --- 2^11 = LOT ---
		ASR	R2			;   YEAR -> LOB
	.ENDR					; ------------------
	BIC	#^C31.,R2			; LEAVE YEAR ONLY
	ADD	#72.,R2				; CHANGE BASE FROM 1972 -> 1900
	CALL	CNAFD2				; TO ASCII
	BR	7000$				; CONTINUE
;
;

;
;
;		--- THIS SECTION DOES TRIMMING
;
;
7000$:	CMP	R5,#'<			; 'TRIM' TERMINATOR?
	BNE	7040$			; NO -> SKIP
7010$:	CMP	R1,#BUF			; ALL TRIMMED OF?
	BLOS	7040$			; YES -> SKIP
	CMPB	-(R1),#40		; TRAILING SPACE?
	BEQ	7010$			; YES -> IGNORE IT
	INC	R1			; POINT TO NEXT CHAR PLACE
;
;
;
;		--- THIS SECTION SENDS DATA TO OUTPUT STREAM ---
;
;
7040$:	CLRB	(R1)			; TERMINATE
	MOV	#BUF,R1			; R1 -> BUFFER
7100$:	MOVB	(R1)+,R0		; GET CHAR
	BEQ	7200$			; EOS -> 7200$
	.OUT	R0			; SEND OUT
	BR	7100$			; LOOP
;
7200$:	.POP	<R5,R4,R3,R2,R1>	; RESTORE
	RETURN				; EXIT
;
;
;
;		--- COULD NOT LOCATE FIELD ---
;
;
10000$:	.OUT	<#'?,#'?,#'?>		; SHOW NOT FOUND
	BR	7200$			; EXIT
;
;

;
;
;
.SBTTL	PRIMITIVE - '..NEXT'	...	GET NEXT RECORD NUM
;
;
;
..NEXT:	.PUSH	<R0,R1>				; SAVE
	MOV	RECRPT,R0			; GET RECORD TO REPEAT
	CLR	RECRPT				; CLEAR - HAVE GOT IT
	TST	R0				; GOT A LEGAL ONE?
	BNE	100$				; YES -> 100$
;
	TST	RECNUM				; ANY LEFT?
	BEQ	1000$				; NO -> SKIP
	MOV	SFLPAR,R1			; SIZE
	MOV	#BUF,R0				; BUFFER
	CALL	SFLINP				; GET NEXT
	MOV	BUF,R0				; GET REC NUMBER
;
100$:	DEC	RECNUM				; ONE LESS
	MOV	R0,RECORD			; SAVE RECORD NUMBER
	CALL	FIL.DR				; READ IN NEW DATA
1000$:	.POP	<R1,R0>				; RESTORE
	RETURN					; HOME
;
;
;
RECNUM:	.WORD	0				; NUMBER OF RECORDS
RECORD:	.WORD					; RECORD BEING PROCESSED
RECRPT:	.WORD	0				; RECORD TO REPEAT
;
;

;
;
;
.SBTTL	PRIMITIVE - '..DATE'	...	OUTPUT DATE
;
;
..DATE:	.PUSH	<R1,R2,R3,R5>			; SAVE
	.DATE					; GET DATE
	MOV	R0,R5				; SAVE FOR LATER
	CLR	R0				; NO FORMAT CONTROL
	MOV	#BUF,R1				; R1 -> BUFFER
;
	MOV	R5,R2				; DATE -> R2
	.REPT	5				; --- 2^5 = 32 ---
		ASR	R2			;   MOVE DAY -> LOB
	.ENDR					; ----------------
	BIC	#^C31.,R2			; LEAVE DAY ONLE
	CALL	CNAFD2				; BIN -> ASC
	.OUT	<BUF,BUF+1,#'->			; SEND OUT
;
	MOV	#BUF,R1				; R1 --> BUFFER
	MOV	R5,R2				; DATE -> R2
	.REPT	10.				; --- 2^10 = ?? ---
		ASR	R2			;   MONTH -> LOB
	.ENDR					; -----------------
	BIC	#^C15.,R2			; LEAVE MONTH COUNT
	CALL	CNAFD2				; CHANGE TO ASCII
	.OUT	<BUF,BUF+1,#'->			; OUTPUT
;
	MOV	#BUF,R1				; R1 --> BUFFER
	MOV	R5,R2				; DATE -> R2
	BIC	#^C31.,R2			; LEAVE YEAR ONLY
	ADD	#72.,R2				; CONVERT
	CALL	CNAFD2				; TO ASCII
	.OUT	<BUF,BUF+1>			; OUTPUT
;
	.POP	<R5,R3,R2,R1>			; RESTORE
	RETURN					; HOME
;
;

;
;
;
.SBTTL	PRIMITIVE - '..PAGE'	...	OUTPUT PAGE
;
;
..PAGE:	.PUSH	<R0,R1,R2>			; SAVE
	MOV	(PC)+,R2			; R2 == PAGE NUMBER
PAGENO:	.WORD	0				; PAGE NUMBER
	MOV	#BUF,R1				; R1 --> BUFFER
	MOV	#^B0101010100000000,R0		; FORMAT (B -> S)
	CALL	CNAFD5				; CHANGE TO ASCII
	.OUT	<BUF,BUF+1,BUF+2,BUF+3,BUF+4>	; OUTPUT
	.POP	<R2,R1,R0>			; RESTORE
	RETURN					; EXIT
;
;
;

;
;
;
.SBTTL	ROUTINE	- '..EQ'	...	END PHASE IF VARIABLES EQUAL
;
;
;
;	IF THE CONTENTS OF THE TWO VARIABLES SPECIFIED (NEXT TWO CHARS)
;	ARE THE SAME END PHASE!
;
;
..EQ:	CALL	...CMP			; COMPARE TIME
	BCS	1000$			; ERROR -> 1000$
	BNE	1000$			; NO MATCH -> 1000$
	MOV	RECORD,RECRPT		; REPEAT RECORD
	INC	RECNUM			; ONE MORE RECORD TO DO
	JMP	..PEND			; END PHASE
1000$:	RETURN				; HOME
;
;

;
;
;
.SBTTL	ROUTINE	- '..NE'	...	IF VARS NOT SAME THEN END PHASE
;
;
;
;	THIS ROUTINE WILL TERMINATE THE PHASE IFF THE TWO SPECIFIED
;	VARIABLES ARE NOT THE SAME.
;
;
..NE:	CALL	...CMP			; COMPARE TIME
	BCS	1000$			; FAIL -> 1000$
	BEQ	1000$			; MATCH -> 1000$
	MOV	RECORD,RECRPT		; REPEAT RECORD
	INC	RECNUM			; ONE MORE RECORD TO DO
	JMP	..PEND			; END PHASE
1000$:	RETURN				; HOME
;
;

;
;
.SBTTL	ROUTINE	- '...CMP'	...	COMPARE VARIABLES
;
;
;	THIS ROUTINE COMPARES THE CONTENTS OF THE TWO VARIABLES, WHOSE
;	NAMES ARE THE NEXT TWO LETTERS. ON EXIT-
;
;		C	IS SET IF ILLEGAL VAR NAME WAS ENCOUNTERED
;		Z	IS SET IFF TWO VARS HOLD SAME DATA
;
...CMP:	.PUSH	<R1,R2>			; SAVE
	.IN				; GET NEXT CHARACTER
	SUB	#'A,R0			; A -> 0
	BLO	10000$			; NOT ALPHA -> ABORT
	CMP	R0,#<'Z-'A>		; IN RANGE?
	BHI	10000$			; NO -> ABORT
	SWAB	R0			; R0*1000 -> R0
	ADD	#VAR.DT,R0			; POINT TO DATA
	MOV	R0,R1			; VAR 1 ADDRESS -> R1
;
	.IN				; GET NEXT CHARACTER
	SUB	#'A,R0			; A -> 0
	BLO	10000$			; NOT ALPHA -> ABORT
	CMP	R0,#<'Z-'A>		; IN RANGE?
	BHI	10000$			; NO -> ABORT
	SWAB	R0			; R0*1000 -> R0
	ADD	#VAR.DT,R0			; POINT TO DATA
	MOV	R0,R2			; VAR 2 ADDRESS -> R2
;
100$:	CMPB	(R1)+,(R2)+		; MATCH?
	BNE	1000$			; <> -> 1000$
	TSTB	-1(R1)			; NULL?
	BNE	100$			; NO -> SKIP
	.POP	<R2,R1>			; RESTORE
	CLR	R0			; SET 'MATCH' FLAG
	RETURN				; HOME
;
;
1000$:	.POP	<R2,R1>			; RESTORE
	CLR	R0			; CLEAR 'C'
	MOV	#1,R0			; NO MATCH
	RETURN				; HOME
;
;
10000$:	.POP	<R2,R1>			; RESTORE
	SEC				; SET FAIL
	RETURN				; HOME
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'DECODE'	...	DECODE COMMANDS
;
;
;
DECODE:	.PUSH	R5			; SAVE
10$:	TST	(R5)			; EOT?
	BEQ	1000$			; YES -> 1000$
	CMP	R0,(R5)			; MATCH?
	BEQ	2000$			; YES -> 2000$
	ADD	#4,R5			; BUMP UP
	BR	10$			; LOOP
;
1000$:	.POP	R5			; RESTORE
	SEC				; SET FAIL
	RETURN				; EXIT
;
;
2000$:	MOV	2(R5),R0		; GET ARG
	.POP	R5			; RESTORE
	CLC				; CLEAR FAIL
	RETURN				; EXIT
;
;

.SBTTL	*******************
.SBTTL	*                 *
.SBTTL	* MODULE 'SFLEIO' *
.SBTTL	*                 *
.SBTTL	*******************
;
;
;	THE FUNCTION OF THIS MODULE IS TO ALLOW DATA TO BE TRANSFERED
;	TO AND FROM THE SORT DATA FILE. THE ROUTINES PROVIDED ARE
;
;		SFLINT	...	MUST BE CALLED FIRST TO INITIALIZE
;				THE BUFFERS, CHANNELS ETC. DESTROYS
;				ALL REGISTERS.
;
;		SFLRST	...	MUST BE CALLED AT THE END OF A PASS
;				TO RESET THE INPUT/OUTPUT BUFFERS.
;				EFFECTIVELY REWINDS THE FILE.
;
;		SFLEND	...	CALLED AT END OF I/O TO CLOSE OUT
;				THE FILE.
;
;		SFLINP	...	INPUT 'R1' CHARACTERS INTO BUFFER @ 'R0'
;
;		SFLOUT	...	OUTPUTS 'R1' CHARS FROM BUFFER @ 'R0'
;
;		SFLINC	...	RETURNS NEXT INPUT CHARACTER IN R0.
;
;		SFLOUC	...	OUTPUTS CHARACTER IN R0 TO FILE.
;
;
;	THE FOLLOWING VARIABLES/BUFFERS ARE USED BY THE PROGRAM. THE
;	USER MAY ACCESS ONLY THOSE MARKED WITH AN ** SYMBOL-
;
;	SFLPAR **	HOLDS BLOCK '0' OF THE SORT DATA FILE. THE
;			CONTENTS THIS 400 (OCTAL) WORD BUFFER CAN BE
;			MODIFIED TO SEND INFORMATION TO THE NEXT
;			PROGRAM TO USE THE FILE AS IT WILL BE WRITTEN
;			OUT WHEN THE FILE IS CLOSED.
;
;	INPBUF/OUTBUF	400 (OCTAL) WORDS FILE BUFFERS
;
;	INPCNT/OUTCNT	COUNT OF CHARACTERS IN INPUT/OUTPUT BUFFERS
;
;	FDB		FILE DESCRIPTOR BLOCK FOR DATA FILE.
;
;

;
;
;
.SBTTL	DOCUMENTATION	- 'SFLEIO' FILE STRUCTURE
;
;
;	'SFLEIO' IS USED TO ACCESS THE DATA IN THE "DK:SORTER.DAT"
;	DATA FILE. THIS FILE IS TREATED AS COMPOSED OF TWO SECTIONS.
;	THESE SECTIONS ARE -
;
;	BLOCK 0		A 400 (OCTAL) WORD PARAMETER BLOCK WHICH
;			WILL BE READ INTO 'SFLPAR' AT INITIALIZATION
;			TIME, AND REWRITTEN OUT AT EXIT TIME. THIS
;			PARAMETER BLOCK CAN BE USED BY THE CALLER
;			TO PASS ARGUMENTS BETWEEN PROGRAMS.
;
;	BLOCK 1-N	A STREAM OF 8 BIT BYTES, WHICH CAN BE READ/WRITTEN
;			SEQUENTIALLY. THE FILE CAN BE REWOUND AT ANY
;			TIME BY CALLING 'SFLRST'
;
;
;
.SBTTL	DOCUMENTATION	- 'SFLEIO' FILE I/O
;
;
;	ALL FILE I/O IS DONE USING THE 'FILSPT' MODULE. ANY FILE I/O
;	ERRORS WILL CAUSE A FATAL ERROR MESSAGE TO BE PRINTED.
;
;

;
;
;
.SBTTL	DECLARATIONS FOR 'SFLEIO'
;
;
.MCALL	.PUSH,.POP			; STACKING
.MCALL	.PRINT,.EXIT			; ERROR HANDLING
.MCALL	FILSPT				; FILE SUPPORT
;
	FILSPT				; INVOKE FILE SUPPORT
;
;
.GLOBL	FIL.ST				; "FILSPT"
;
;
	.PSECT	CODE			; OPEN CODE AREA
;	------	----
;

;
;
;
.SBTTL	INITIALIZATION SECTION
;
;
;	OPEN SORT DATA FILE FOR INPUT, AND READ IN THE PARAMETER
;	BLOCK. RESET THE INPUT/OUTPUT BLOCK/CHARACTER COUNTER.
;
SFLINT:	NAME	#FDB,#100$,ERROR=40$	; INSERT FILE NAME
	PURGE				; ENSURE CHANNEL OK
	LOOKUP	ERROR=60$		; LOOKUP FILE
	READ	BLOCK=#0,SIZE=#400,BUFFER=#SFLPAR,ERROR=60$
	MOV	#1,INPBLK		; START INPUTING BLOCK 1
	MOV	#1,OUTBLK		; START OUTPUTING BLOCK 1
	MOV	#1000,INPCNT		; 1000 BYTES IN INPUT BUFFER
	MOV	#0,OUTCNT		; NO BYTES IN OUTPUT BUFFER
	READ	FDB=#FDB,BLOCK=#1,BUFFER=#INPBUF,ERROR=60$
	RETURN				; ALL DONE
;
40$:	.PRINT	110$			; TELL FAILED
60$:	CALL	FIL.ST			; DISPLAY FILE STATUS
	.EXIT				; QUIT
;
.NLIST	BIN
.ENABL	LC
100$:	.ASCIZ	/DK:SORTER.DAT/		; FILE SPEC
	.BLKB	120-.+100$		; PAD OUT (***FIL.NM***)
110$:	.ASCIZ	/"SFLEIO" -??????- HOW CAN NAMING FAIL?/
.EVEN
.DSABL	LC
.LIST	BIN
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'SFLRST'	...	RESET SORT FILE
;
;
;	THIS ROUTINE IS CALLED TO RESET THE SORT FILE PRIOR TO A NEW
;	PASS. IT EMPTIES OUT THE CONTENTS OF THE OUTPUT BUFFER AND
;	RESETS THE INPUT/OUTPUT BLOCK/CHARACTER COUNTERS.
;
;
;
SFLRST:	.PUSH	R5			; SAVE R5
	TST	OUTCNT			; ANY CHARACTERS TO OUTPUT?
	BEQ	100$			; NO -> 100$
	WRITE	FDB=#FDB,BLOCK=OUTBLK,BUFFER=#OUTBUF,ERROR=1000$
100$:	MOV	#1,INPBLK		; START INPUTING BLOCK 1
	MOV	#1,OUTBLK		; START OUTPUTING BLOCK 1
	MOV	#1000,INPCNT		; 1000 BYTES IN INPUT BUFFER
	MOV	#0,OUTCNT		; NO BYTES IN OUTPUT BUFFER
	READ	FDB=#FDB,BLOCK=#1,BUFFER=#INPBUF,ERROR=1000$
	.POP	R5			; RESTORE
	RETURN				; EXIT
;
1000$:	CALL	FIL.ST			; SHOW CAUSE OF ERROR
	.EXIT				; QUIT
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'SFLEND'	...	CLOSE SORT FILE
;
;
;	THIS ROUTINE IS CALLED WHEN THE USER IS FINISHED WITH THE
;	SORT FILE. IT EMPTIES OUT THE CONTENTS OF THE OUTPUT BUFFER
;	AND CLOSES OUT THE FILE.
;
;
;
SFLEND:	.PUSH	R5			; SAVE R5
	TST	OUTCNT			; ANY CHARACTERS TO OUTPUT?
	BEQ	100$			; NO -> 100$
	WRITE	FDB=#FDB,BLOCK=OUTBLK,BUFFER=#OUTBUF,ERROR=1000$
100$:	WRITE	FDB=#FDB,BLOCK=#0,BUFFER=#SFLPAR,ERROR=1000$
	PURGE	#FDB			; PURGE CHANNEL
	.POP	R5			; RESTORE
	RETURN				; BYE
;
;
1000$:	CALL	FIL.ST			; SHOW ERROR CAUSE
	.EXIT				; EXIT TIME
;
;

;
;
;
.SBTTL	ROUTINES	- 'SLFINP'/'SLFOUT'	...	INPUT/OUTPUT
;
;
;	THESE ROUTINES TRANSFER 'R1' CHARACTERS TO/FROM THE USER
;	BUFFER TO/FROM THE SORT DATA FILE. THE ADDRESS OF THE USER
;	BUFFER MUST BE IN 'R0'.
;
;
;
SFLINP:	.PUSH	<R0,R1,R2>		; SAVE
	MOV	R0,R2			; BUFFER ADDRESS -> R2
10$:	CALL	SFLINC			; GET ONE CHARACTER
	MOVB	R0,(R2)+		; SAVE IT
	SOB	R1,10$			; LOOP
	.POP	<R2,R1,R0>		; RESTORE
	RETURN				; EXIT
;
;
;
SFLOUT:	.PUSH	<R0,R1,R2>		; SAVE
	MOV	R0,R2			; BUFFER ADDRESS -> R2
10$:	MOVB	(R2)+,R0		; CHAR -> R0
	CALL	SFLOUC			; OUTPUT A CHARACTER
	SOB	R1,10$			; LOOP
	.POP	<R2,R1,R0>		; SAVE
	RETURN				; EXIT
;
;

;
;
;
.SBTTL	PRIMITIVE	- 'SFLINC'	...	INPUT A CHARACTER
;
;
;	THIS ROUTINE READS IN THE NEXT CHARACTER FROM THE SORT
;	DATA FILE IN RETURNS IT IN R0.
;
;
;
SFLINC:	.PUSH	<R4,R5>			; SAVE
	MOV	INPCNT,R4		; INPUT COUNT -> R4
	BNE	100$			; YES -> SKIP
	INC	INPBLK			; UP BLOCK COUNT
	READ	FDB=#FDB,BLOCK=INPBLK,BUFFER=#INPBUF,ERROR=1000$
	MOV	#1000,R4		; RESET COUNTER
;
100$:	MOV	#INPBUF+1000,R5		; R5 -> END OF BUFFER
	SUB	R4,R5			; R5 -> CHARACTER TO GET
	MOVB	(R5),R0			; GET CHAR
	DEC	R4			; ONE LESS CHAR IN BUFFER
	MOV	R4,INPCNT		; SAVE
	.POP	<R5,R4>			; RESTORE
	RETURN				; EXIT
;
1000$:	CALL	FIL.ST			; DISPLAY ERROR CAUSE
	.EXIT				; QUIT
;
;

;
;
;
.SBTTL	PRIMITIVE	- 'SFLOUC'	...	OUTPUT A CHARACTER
;
;
;	THIS ROUTINE WILL PLACE THE CHARACTER IN R0 INTO THE DATA
;	FILE.
;
;
;
SFLOUC:	.PUSH	<R4,R5>			; SAVE
	MOV	OUTCNT,R4		; NUM CHRS IN BUF -> R4
	MOVB	R0,OUTBUF(R4)		; SAVE CHAR
	INC	R4			; UP COUNT
	CMP	R4,#1000		; FULL BUFFER?
	BGT	1000$			; > -> HARD ERROR
	BNE	100$			; < -> OK
	WRITE	FDB=#FDB,BLOCK=OUTBLK,BUFFER=#OUTBUF,ERROR=1100$
	INC	OUTBLK			; BUMP UP BLOCK NUMBER
	CLR	R4			; RESET CHAR COUNTER
;
100$:	MOV	R4,OUTCNT		; SAVE COUNT
	.POP	<R5,R4>			; SAVE
	RETURN				; EXIT
;
;
1000$:	.PRINT	#1777$			; TELL FAILED
1100$:	CALL	FIL.ST			; SHOW ERROR CAUSE
	.EXIT				; QUIT
;
.NLIST	BIN
1777$:	.ASCIZ	/"SFLEIO" -?????- WE IS STUFFED!/
.EVEN
.LIST	BIN
;
;

;
.SBTTL	DATA AREA FOR 'SFLEIO'
;
;
;
	.PSECT	$$$FDB			; OPEN FILES AREA
;	------	------
;
;
	FDB	NAME=FDB,CHANNEL=15.,BUFFER=SFLPAR,SIZE=400
;
;
INPCNT:	.WORD	0			; NUMBER WORDS IN BUFFER
OUTCNT:	.WORD	0
INPBLK:	.WORD	-1			; BLOCK NUMBER
OUTBLK:	.WORD	-1			; BLOCK NUMBER
SFLPAR:	.BLKW	400			; PARAMETER BLOCK
INPBUF:	.BLKW	400			; INPUT BUFFER
OUTBUF:	.BLKW	400			; OUTPUT BUFFER
;

.SBTTL	***************************
.SBTTL	*                         *
.SBTTL	* MODULE 'FRMIO           *
.SBTTL	*                         *
.SBTTL	***************************
;
;
;	THIS MODULE CONTAINS THE ROUTINES-
;
;		FRMINT	...	INITIALIZE THE FORM DEFINITION FILE
;				THIS ROUTINE MUST BE CALLED AT START UP
;
;		FRMINC	...	RETURNS NEXT CHARACTER FRM FILE
;
;		FRMRST	...	RESETS THE FILE - EFFECTIVELY DOES A
;				REWIND!
;
;
;
;
;
.SBTTL	DOCUMENTATION	- MODULE FUNCTION
;
;
;	THE 'FRMINT' ROUTINE MEST BE CALLED FIRST. THIS ROUTINE WILL OPEN
; THE FILE'DK:REPORT.FRM' FOR INPUT. IF THE FILE IS CANNOT BE FOUND, A
; WARNING MESSAGE IS TYPED, AND THE OPERATOR IS REQUESTED TO SPECIFY THE
; NAME OF THE FORMAT FILE.
;
;
;	THE 'FRMINC' ROUTINE IS USED TO RETURN CHARACTERS SEQUENTIALLY
; FROM THE FILE. THE CHARACTERS ARE RETURNED IN 'R0'. THE 'C' FLAG WILL
; BE SET IFF AN EOF ERROR OCCURS. THIS ROUTINE SUPPORTS THE '^'/'_'
; PREFIX CHARACTERS, AND WILL INTERPRET A '^Z' AS AN EOF MARKER.
;
;
;	THE 'FRMRST' ROUTINE IS CALLED TO REWIND THE FILE.
;
;

;
;
;
.SBTTL	DECLARATIIONS
;
;
;
.MCALL	.PUSH,.POP			; STACKING
.MCALL	.EXIT,.PRINT,.GTLINE		; RT-11 ERROR REPORTING
.MCALL	FILSPT				; FILE SUPPORT
;
	FILSPT				; SET UP FOR FILE SUPPORT
;
;
;
;
;
;
	.PSECT	CODE			; OPEN CODE SECTION
;	------	----
;
;

;
;
;
.SBTTL	INITIALIZATION SECTION
;
;
;
;	RESET 'EOF' FLAG AND FETCH POINTER. OPEN FILE, AND READ IN FIRST
;	TWO BLOCKS. SET UP 'COUNT OF CHARS IN BUFFER'.
;
;
FRMINT:	CLR	FRMFLG			; CLEAR 'EOF' MARKER
	MOV	#FRMBUF,FRMPNT		; RESET POINTER
;
	COPY	<<<FORM$$+40>>>,101$	;; SET UP FILE NAME
	COPY	130$			;; SET UP EXTENSION
	NAME	#FDBFRM,#100$		; ATTEMP USE DEFAULT NAME
	BCC	20$			; FAIL -> 20$
;
10$:	.GTLIN	#100$,#120$		; GET NAME OF FORMAT FILE
	NAME	#FDBFRM,#100$,ERROR=40$	; INSERT FILE NAME
;
20$:	PURGE				; ENSURE CHANNEL OK
	LOOKUP	ERROR=10$		; LOOKUP FILE
	READ	FDB=#FDBFRM,BLOCK=#0,BUFFER=#FRMBUF,SIZE=#1000,ERROR=60$
	MOV	FDBFRM+FDB.CP,R0	; NUMBER WORDS -> R0
	ASL	R0			; WORDS -> BYTES
	MOV	R0,FRMCNT		; SET UP COUNTER
	RETURN				; ALL DONE
;
40$:	.PRINT	110$			; TELL FAILED
60$:	STATUS				; SHOW ERROR CAUSE
	.EXIT				; QUIT
;
.NLIST	BIN
.ENABL	LC
100$:	.ASCII	/DK:/			; DEVICE
101$:	.BLKB	120-.+100$		; PAD OUT (***FIL.NM***)
110$:	.ASCIZ	/"FRMIO" -??????- HOW CAN NAMING FAIL?/
120$:	.ASCII	/ENTER NAME OF FORMAT FILE > /<200>
130$:	.ASCIZ	/.FRM/			; FILE EXTENSION
.EVEN
.DSABL	LC
.LIST	BIN
;
;

;
;
.SBTTL	ROUTINE	- 'FRMINC'	...	GET A CHARACTER FROM FORM BUFFER
;
;
;	THIS ROUTINE PASSES BACK THE NEXT CHAR IN THE FILE. IT USES THE
;	'FRMINP' ROUTINE TO FETCH THE CHARACTER. THIS ROUTINE HANDLES
;	THE FOLLOWING
;
;		A '_' CHARACTER WILL CAUSE THE NEXT CHARACTER TO BE
;		RETURNED, WITH OUT ANY TRANSLATION.
;
;		A '^' WILL CAUSE THE NEXT CHARACTER TO BE INPUT, AND
;		CONVERTED TO ITS EQUIVALENT CONTROL CODE.
;
;		A '^Z' WILL CAUSE AN EOF CONDITION TO BE FLAGGED.
;
;
FRMINC:	.PUSH	<R1,R2,R3,R4,R5>	; SAVE
	TST	FRMFLG			; AT EOF?
	BNE	1000$			; YES -> 1000$
;
	CALL	FRMINP			; INPUT NEXT CHAR
	BCS	1000$			; AT EOF -> 1000$
	CMP	R0,#<'Z-'@>		; EOF MARKER?
	BEQ	1000$			; YES -> 1000$
	CMP	R0,#'_			; SPECIAL PREFIX?
	BEQ	400$			; YES -> 400$
	CMP	R0,#'^			; SPECIAL?
	BEQ	500$			; YES -> 500$
	.POP	<R5,R4,R3,R2,R1>	; RESTORE
	CLC				; CLEAR FAIL
	RETURN				; ALL DONE
;
;
400$:	CALL	FRMINP			; GET NEXT CHAR
	BCS	1000$			; FAILED -> 1000$
	.POP	<R5,R4,R3,R2,R1>	; RESTORE
	CLC				; CLEAR FAIL FLAG
	RETURN				; HOME
;
500$:	CALL	FRMINP			; GET NEXT CHAR
	BCS	1000$			; FAILED -> 1000$
	.POP	<R5,R4,R3,R2,R1>	; RESTORE
	BIC	#^C37,R0		; TRANSLATE TO A CONTROL CODE
	RETURN				; HOME
;
1000$:	.POP	<R5,R4,R3,R2,R1>	; RESTORE
	MOV	#1,FRMFLG		; SET EOF MARKER
	SEC				; ENSURE 'C' SET
	RETURN				; HOME
;
;
;

;
;
.SBTTL	ROUTINE	- 'FRMINP'	...	INPUTS NEXT SEQUENTIAL CHAR
;
;
;	THIS ROUTINE RETURNS THE NEXT CHARACTER IN THE FILE. THE 'C'
;	FLAG IS SET ON EXIT IFF AN EOF WAS DETECTED. THIS ROUTINE
;	BELIEVES AN EOF OCCURES WHEN-
;
;		AN ATTEMPTED READ FAILS DUE TO RT11 EOF
;
;		A <NULL> IS ENCOUNTERED (IN AN ASCII FILE)
;
FRMINP:	DEC	FRMCNT			; ONE LESS BYTE IN BUFFER
	BGE	100$			; OK -> 100$
	ADD	#2,FDBFRM+FDB.BL	; READ IN NEXT TWO BLOCKS
	READ	FDB=#FDBFRM,BUFFER=#FRMBUF,SIZE=#1000,ERROR=60$
	MOV	FDBFRM+FDB.CP,R0	; NUMBER WORDS -> R0
	ASL	R0			; WORDS -> BYTES
	MOV	R0,FRMCNT		; SET UP COUNTER
	BEQ	1000$			; AT EOF -> 1000$
	MOV	#FRMBUF,FRMPNT		; RESET POINTER
	BR	FRMINC			; INPUT A CHAR
60$:	STATUS				; SHOW CAUSE OF ERROR
	.EXIT				; BYE
;
100$:	MOVB	@FRMPNT,R0		; GET CHARACTER
	BIC	#^C377,R0		; MUST BE LOB ONLY
	BEQ	1000$			; <NULL> -> EOF
	INC	FRMPNT			; UP POINTER
	CLC				; CLEAR FAIL
	RETURN				; EXIT
;
;
1000$:	CLR	R0			; ENSURE NULL
	SEC				; SET FAIL
	RETURN				; EXIT
;
;
;

;
;
.SBTTL	ROUTINE	- 'FRMRST'	...	RESET INPUT POINTER
;
;
;
FRMRST:	.PUSH	<R0,R5>			; SAVE
	READ	FDB=#FDBFRM,BUFFER=#FRMBUF,SIZE=#1000,BLOCK=#0,ERROR=60$
	MOV	FDBFRM+FDB.CP,R0	; NUMBER WORDS -> R0
	ASL	R0			; WORDS -> BYTES
	MOV	R0,FRMCNT		; SET UP COUNTER
	MOV	#FRMBUF,FRMPNT		; RESET POINTER
	CLR	FRMFLG			; CLEAR 'EOF' MARKER
	.POP	<R5,R0>			; RESTORE
	RETURN				; ALL DONE
;
60$:	STATUS				; SHOW CAUSE OF ERROR
	.EXIT				; BYE
;
;
;

;
;
;
.SBTTL	VARIABLES USED BY 'FRMIO'
;
;
;
FDB	NAME=FDBFRM,CHANNEL=14.,BUFFER=FRMBUF,SIZE=1000
FRMBUF:	.BLKW	1000			; I/O BUFFER
FRMFLG:	.WORD	0			; <> 0 --> EOF
FRMCNT:	.WORD	0			; COUNT OF CHARS IN BUFFER
FRMPNT:	.WORD	FRMBUF			; FETCH POINTER
;
;

.SBTTL	*******************
.SBTTL	*                 *
.SBTTL	* MODULE 'RPTEIO' *
.SBTTL	*                 *
.SBTTL	*******************
;
;
;	THE FUNCTION OF THIS MODULE IS TO ALLOW DATA TO BE TRANSFERED
;	TO THE REPORT DATA FILE. THE ROUTINES PROVIDED ARE
;
;		RPTINT	...	MUST BE CALLED FIRST TO INITIALIZE
;				THE BUFFERS, CHANNELS ETC. DESTROYS
;				ALL REGISTERS.
;
;		RPTEND	...	CALLED AT END OF I/O TO CLOSE OUT
;				THE FILE.
;
;		RPTOUC	...	OUTPUTS CHARACTER IN R0 TO FILE.
;
;
;	THE FOLLOWING VARIABLES/BUFFERS ARE USED BY THE PROGRAM.
;
;	RPTBUF		400 (OCTAL) WORDS FILE BUFFER
;
;	RPTCNT		COUNT OF CHARACTERS IN OUTPUT BUFFER
;
;	RPTFDB		FILE DESCRIPTOR BLOCK FOR DATA FILE.
;
;
.SBTTL	DOCUMENTATION	- MODULE OPERATION
;
;
;	THE 'RPTINT' ROUTINE IS USED TO OPEN THE REPORT FILE FOR OUTPUT.
; THE NAME OF THE FILE IS OBTAINED BY PROMPTING THE OPERATOR.
;
;	THE 'RPTOUC' ROUTINE WILL TRANSFER ONE CHARACTER FROM R0 TO THE
; OUTPUT BUFFER.
;
;	THE 'RPTEND' ROUTINE IS CALLED TO CLOSE OUT THE OUTPUT FILE.
;
;

;
;
;
.SBTTL	DECLARATIONS FOR 'RPTEIO'
;
;
.MCALL	.PUSH,.POP			; STACKING
.MCALL	.PRINT,.EXIT			; ERROR HANDLING
.MCALL	FILSPT				; FILE SUPPORT
;
	FILSPT				; INVOKE FILE SUPPORT
;
;

.GLOBL	FIL.ST				; "FILSPT"
;
;
	.PSECT	CODE			; OPEN CODE AREA
;	------	----

;
;
.SBTTL	ROUTINE	- '..FILE'	...	OPEN NEW OUTPUT FILE
;
;
;
;	THIS IS COMPLEXE. WHEN CALLED, THIS ROUTINE WILL PURGE THE
;	DEFAULT OUTPUT FILE, AND OPEN A NEW ONE. THIS ROUTINE CAN BE
;	CALLED ONCE ONLY. IF CALLED MORE THAN ONCE, IT RETURNS
;	IMMEDIATELY.
;
;
;
..FILE:	TST	7000$				; FLAG SET?
	BNE	70$				; YES -> SKIP
	MOV	#1,7000$			; SET FLAG
;
	.PUSH	<R1,R5>				; SAVE
	MOV	#10000$,R1			; R1 -> BUFFER
10$:	.IN					; GET A CHAR
	CMPB	R0,#15				; CR ?
	BEQ	20$				; YES -> SKIP
	MOVB	R0,(R1)+			; PLACE CHAR IN BUFFER
	BR	10$				; LOOP
20$:	.IN					; GET LF
	CLRB	(R1)				; TERMINATE STRING
;
	NAME	#RPTFDB,#10000$,ERROR=1000$	; INSERT FILE NAME
	PURGE					; ENSURE CHANNEL OK
	ENTER	ERROR=1000$			; LOOKUP FILE
	MOV	#0,RPTBLK			; START OUTPUTING BLOCK 0
	MOV	#0,RPTCNT			; NO BYTES IN OUTPUT BUFFER
	.POP	<R5,R1>				; RESTORE
70$:	RETURN					; ALL DONE
;
1000$:	STATUS					; SHOW CAUSE OF ERROR
	.EXIT					; QUIT
;
;
7000$:	.WORD	0				; SEMAPHOR FLAG
;
.NLIST	BIN
.ENABL	LC
10000$:	.ASCII	//				; FILE NAME GOES HERE
10001$:	.BLKB	120-.+10000$			; PAD OUT (***FIL.NM***)
.EVEN
.DSABL	LC
.LIST	BIN

.SBTTL	INITIALIZATION SECTION

RPTINT:	COPY	<<<FORM$$+40>>>,101$	;; COPY
	COPY	130$			;; EXTENSION
	NAME	#RPTFDB,#100$,ERROR=40$	; INSERT FILE NAME
	PURGE				; ENSURE CHANNEL OK
	ENTER	ERROR=60$		; LOOKUP FILE
	MOV	#0,RPTBLK		; START OUTPUTING BLOCK 0
	MOV	#0,RPTCNT		; NO BYTES IN OUTPUT BUFFER
	RETURN				; ALL DONE
;
40$:	.PRINT	110$			; TELL FAILED
60$:	CALL	FIL.ST			; DISPLAY FILE STATUS
	.EXIT				; QUIT
;
.NLIST	BIN
.ENABL	LC
100$:	.ASCII	/LST:/			; DEVICE

.EVEN

;
.SBTTL	ROUTINE	- 'RPTEND'	...	CLOSE REPORT FILE
;
;
;	THIS ROUTINE IS CALLED WHEN THE USER IS FINISHED WITH THE
;	REPORT FILE. IT EMPTIES OUT THE CONTENTS OF THE OUTPUT BUFFER
;	AND CLOSES OUT THE FILE.
;
;
;
RPTEND:	CLR	R0			; 0 -> R0
	CALL	RPTOUC			; SEND A NULL
	.PUSH	R5			; SAVE R5
	MOV	RPTCNT,R1		; COUNT -> R1
	BEQ	100$			; NO -> 100$
	ASR	R1			; BYTES -> WORDS
	WRITE	FDB=#RPTFDB,BLOCK=RPTBLK,SIZE=R1,BUFFER=#RPTBUF,ERROR=1000$
100$:	CLOSE	#RPTFDB			; PURGE CHANNEL
	.POP	R5			; RESTORE
	RETURN				; BYE
;
;
1000$:	CALL	FIL.ST			; SHOW ERROR CAUSE
	.EXIT				; EXIT TIME
;
;

;
;
;
.SBTTL	PRIMITIVE	- 'RPTOUC'	...	OUTPUT A CHARACTER
;
;
;	THIS ROUTINE WILL PLACE THE CHARACTER IN R0 INTO THE DATA
;	FILE.
;
;
;
RPTOUC:	BIT	#1,RPTFLG		;; OUTPUT SUPPRESSED?
	BNE	500$			;; YES -> 500$
	.PUSH	<R4,R5>			; SAVE
	MOV	RPTCNT,R4		; NUM CHRS IN BUF -> R4
	MOVB	R0,RPTBUF(R4)		; SAVE CHAR
	INC	R4			; UP COUNT
	CMP	R4,#1000		; FULL BUFFER?
	BGT	1000$			; > -> HARD ERROR
	BNE	100$			; < -> OK
	WRITE	FDB=#RPTFDB,BLOCK=RPTBLK,SIZE=#400,BUFFER=#RPTBUF,ERROR=1100$
	INC	RPTBLK			; BUMP UP BLOCK NUMBER
	CLR	R4			; RESET CHAR COUNTER
;
100$:	MOV	R4,RPTCNT		; SAVE COUNT
	.POP	<R5,R4>			; SAVE
500$:	RETURN				; EXIT
;
;
1000$:	.PRINT	#1777$			; TELL FAILED
1100$:	CALL	FIL.ST			; SHOW ERROR CAUSE
	.EXIT				; QUIT
;
.NLIST	BIN
1777$:	.ASCIZ	/"RPTIO" -?????- WE IS STUFFED!/
.EVEN
.LIST	BIN
;
;
RPTFLG:	.WORD	0			; 0 = NO SUPPRESSION

.SBTTL	*******************
.SBTTL	*                 *
.SBTTL	* MODULE 'VARMNG' *
.SBTTL	*                 *
.SBTTL	*******************
;
;
;
;
; THIS MODULE CONTAINS THE TWO ROUTINE REQUIRED TO DO I/O TO AND FROM
; MEMORY STORAGE VARIABLES. THE ROUTINES ARE-
;
; VAROUT	...	SEND FURTHER OUTPUT TO SPECIFIED VARIABLE. THE NAME
;			OF THE OUTPUT VARIABLE IS IN 'R0'
;
; VARINP	...	INPUT FROM THE VARIABLE WHOSE NAME IS IN 'R0'
;
;

;
;
;
.SBTTL	DOCUMENTATION	- ARGUMENT PASSING
;
;
; FOR BOTH ROUTINES THE NAME OF THE VARIABLES IS HELD IN 'R0' AT ENTRY.
;
;
;
.SBTTL	DOCUMENTATION	- VARIABLE NAMES
;
;
; THE LEGAL VARIABLE NAMES ARE -
;
;	@	FOR TERMINAL
;	A-Z	FOR MEMORY VARIABLES
;
; AN INPUT OPERATION ON '@' CAUSES A '^' TO APPEAR ON THE TERMINAL, AND ALL
; CHARACTERS TYPED WILL BE INPUT AND SENT TO THE ROUTINE WHOSE ADDRESS IS
; IN THE 'OUTVEC' WORD. THE INPUT IS TERMINATED WITH A '^Z' (CONTROL Z).
;
; AN INPUT FROM A VARIABLE A-Z WILL CUASE THE CONTENTS OF THAT VARIABLE TO
; BE SENT TO THE OUTPUT ROUTINE.
;
; AN OUTPUT OPERATION TO THE '@' VARIABLE CUASES SEBSEQUENT OUTPUT TO BE SEND
; TO THE CONSOLE. AN OUTPUT TO A VARIABLE 'A'-'Z' CAUSES SESEQUENT OUTPUT
; TO BE STORED IN THE SPECIFIED VARIABLE. EACH VARIABLE CAN HOLD A MAXIMUM OF
; '254' CHARACTERS. THE OUTPUT OPERATION CAUSES THE 'OUTVEC' WORD TO BE POINTED
; TO AN INTERNAL HANDLER WITHIN THE MODULE. OUTPUT MUST BE REDIRECTED TO
; ANOTHER OBJECT TO TERMINATE THE OPERATION.
;
;
;
.SBTTL	DOCUMENTATION	- VARIABLE STORAGE
;
;
; THIS MODULE RESERVES A 256 BYTE BUFFER PER VARIABLE. EACH STRING IS
; STORED IN ASCIZ FORMAT.
;
;

;
;
.SBTTL	DECLARATIONS
;
;
.MCALL	.PRINT,.TTYIN,.TTYOUT			; I/O
.MCALL	.PUSH,.POP				; STACKING
.MCALL	.EXIT					; KILL JOB ON ERROR
;
;
;
;
	.PSECT	CODE				; OPEN CODE SECTION
;	------	----
;
;
;

;
;
;
.SBTTL	MACRO DEFINITIONS
;
;
.MACRO	ERROR	MES,?A,?B,?C
	.PRINT	#A
	.EXIT
A:	.ASCIZ	/'VARMNG' -FATAL- //MES/
.EVEN
.ENDM	ERROR
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'VAROUT'	...	OUTPUT TO VARIABLE
;
;
;
VAROUT:	BIC	#^C177,R0			; DISCARD HIGH ORDER BITS
	CMP	R0,#'@				; SPECIAL = TTY
	BNE	100$				; NO -> 100$
	MOV	#VARTTO,OUTVEC			; SET UP OUTPUT VECTOR
	RETURN					; HOME
;
;
100$:	CMP	R0,#'A				; LOWEST VAR IS 'A'
	BLO	10000$				; TO LOW -> ERROR
	CMP	R0,#'Z				; TO HIGH?
	BHI	10000$				; YES -> ERROR
;
	SUB	#'A,R0				; A -> 0
	SWAB	R0				; R0*256 -> R0
	ADD	#VAR.DT,R0			; POINT TO DATA AREA
	MOV	R0,VAR.SP			; SAVE STORE POINTER
	CLR	VAR.CT				; SET UP OUTPUT COUNTER
	MOV	#VARVRO,OUTVEC			; SET UP ROUTINE ADDRESS
	RETURN					; HOME
;
;
10000$:	ERROR	<'VAROUT'  ILLEGAL VARIABLE NAME>; TELL FAIED
;
;

;
;
;
.SBTTL	ROUTINE	- 'VARTTO'	...	OUTPUT TO TTY
;
;
;
VARTTO:	.TTYOUT					; SEND OUT CHARACTER
	RETURN					; ALL DONE
;
;
;

;
;
;
.SBTTL	ROUTINE	- 'VARVRO'	...	CHAR (R0) --> VARIABLE
;
;
VARVRO:	CMP	VAR.CT,#254.			; 254 CHARACTERS IN BUFFER?
	BHIS	1000$				; YES -> ERROR
	MOVB	R0,@VAR.SP			; STORE CHARACTER
	INC	VAR.SP				; UP POINTER
	CLRB	@VAR.SP				; <NULL> = EOI
	INC	VAR.CT				; UPDATE COUNT
	RETURN					; ALL DONE
;
1000$:	ERROR	<'VARVRO'  TOO MANY CHARACTERS IN VARIABLE!>
;
;

;
;
;
.SBTTL	ROUTINE	- 'VARINP'	...	OPEN VARIABLE FOR INPUT
;
;
;
VARINP:	BIC	#^C177,R0			; DISCARD HIGH ORDER BITS
	CMP	R0,#'@				; SPECIAL = TTY
	BNE	100$				; NO -> 100$
;
	.TTYOUT	#'^				; SHOW WAITING FOR INPUT
10$:	.TTYIN					; GET CHAR
	CMP	R0,#<'Z-'@>			; ^Z
	BEQ	20$				; YES -> 20$
	CALL	@OUTVEC				; OUTPUT IT
	BR	10$				; LOOP
20$:	RETURN					; HOME
;
;
100$:	CMP	R0,#'A				; LOWEST VAR IS 'A'
	BLO	10000$				; TO LOW -> ERROR
	CMP	R0,#'Z				; TO HIGH?
	BHI	10000$				; YES -> ERROR
;
	SUB	#'A,R0				; A -> 0
	SWAB	R0				; R0*256 -> R0
	ADD	#VAR.DT,R0			; POINT TO DATA AREA
	.PUSH	R1				; SAVE
	MOV	R0,R1				; SAVE IN R1
110$:	MOVB	(R1)+,R0			; GET CHAR
	BEQ	120$				; EOS -> 120$
	CALL	@OUTVEC				; OUTPUT
	BR	110$				; LOOP
120$:	.POP	R1				; RESTORE
	RETURN					; HOME
;
;
10000$:	ERROR	<'VAROPN'  ILLEGAL VARIABLE NAME>; TELL FAIED
;
;

;
;
;
.SBTTL	STORAGE AREA
;
;
VAR.SP:	.WORD	VAR.DT				; STORE POINTER
VAR.CT:	.WORD	256.				; STORE COUNT
VAR.DT:	.IRPC	X,<ABCDEFGHIJKLMNOPQRSTUVWXYZ>	; --- HAVE VARS A-Z ---
	.BYTE	0				; INITIALZE TO NULL STRING
	.BLKB	255.				; RESERVE 255 BYTES
	.ENDR					; ----------------------

	BCS	10000$				; NONE -> ERROR

	RETURN					; HOME
